99void * infix_arena_alloc(infix_arena_t * arena, size_t size, size_t alignment) {
100 // Fail immediately if the arena is null or already in an error state.
101 if (arena == nullptr || arena->error)
102 return nullptr;
103
104 // Alignment must be a power of two. This is a common check.
105 if (alignment == 0 || (alignment & (alignment - 1)) != 0) {
106 arena->error = true;
107 return nullptr;
108 }
109
110 // An allocation of zero bytes should return a valid, unique pointer per the C standard.
111 // We return a pointer to the current position without advancing it.
112 if (size == 0)
113 return (void *)(arena->buffer + arena->current_offset);
114
115 // Calculate the padding required to meet the alignment.
116 uintptr_t current_ptr = (uintptr_t)arena->buffer + arena->current_offset;
117 uintptr_t aligned_ptr = (current_ptr + alignment - 1) & ~(alignment - 1);
118 size_t padding = aligned_ptr - current_ptr;
119
120 // Check if adding padding would overflow size_t.
121 if (arena->current_offset > SIZE_MAX - padding) {
122 arena->error = true;
123 return nullptr;
124 }
125 size_t aligned_offset = arena->current_offset + padding;
126 // Check if adding the requested size would overflow size_t.
127 if (aligned_offset > SIZE_MAX - size) {
128 arena->error = true;
129 return nullptr;
130 }
131
132 // Check if there is enough space left in the arena.
133 if (aligned_offset + size > arena->capacity) {
134 arena->error = true;
135 return nullptr;
136 }
137
138 // Bump the pointer and return the aligned address.
139 arena->current_offset = aligned_offset + size;
140 return (void *)aligned_ptr;
141}

156void * infix_arena_calloc(infix_arena_t * arena, size_t num, size_t size, size_t alignment) {
157 // Security: Check for integer overflow in the size calculation.
158 if (size > 0 && num > SIZE_MAX / size) {
159 if (arena)
160 arena->error = true;
161 return nullptr;
162 }
163
164 size_t total_size = num * size;

166
167 // If allocation was successful, zero out the memory.
168 if (ptr != nullptr)
169 memset(ptr, 0, total_size);
