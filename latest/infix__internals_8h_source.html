255static inline size_t _infix_align_up(size_t value, size_t alignment) {
256 // This is a standard bit-twiddling hack for alignment.
257 // It is equivalent to: (value + alignment - 1) / alignment * alignment
258 return (value + alignment - 1) & ~(alignment - 1);
259}

397static inline bool is_passed_by_reference(infix_type * type) {
398 // On Windows x64, aggregates whose size is not a power of two (1, 2, 4, 8 bytes)
399 // are passed by reference. This also applies to sizes larger than 8 bytes on MSVC.
400 // However, GCC/Clang on Windows passes 16-byte types by value on the stack.
401 // To ensure compatibility, we treat 16-byte types as pass-by-value.
402 //~ if (type->size == 16)
403 //~ return false;
404 return type->size != 1 && type->size != 2 && type->size != 4 && type->size != 8;
405}

415static inline bool return_uses_hidden_pointer_abi(c23_maybe_unused infix_type * type) {
416#if defined(INFIX_ABI_WINDOWS_X64)
417#if defined(INFIX_COMPILER_GCC)
418 // On GCC for Windows, its 16-byte long double is a special case returned by reference.
419 if (is_long_double(type))
420 return true;
421#endif
422 // For all other compilers (MSVC, Clang) and for aggregate types on GCC, the rule is size-based.
423 if (type->category == INFIX_TYPE_STRUCT || type->category == INFIX_TYPE_UNION || type->category == INFIX_TYPE_ARRAY)
424 return is_passed_by_reference(type);
425 return false; // Other primitives (including __int128_t) are returned by value.
426#elif defined(INFIX_ABI_SYSV_X64)
427 // long double is not returned via hidden pointer on SysV, but on the x87 stack.
428 // The check here is just for large aggregates.
429 return (type->category == INFIX_TYPE_STRUCT || type->category == INFIX_TYPE_UNION ||
430 type->category == INFIX_TYPE_ARRAY) &&
431 type->size > 16;
432#elif defined(INFIX_ABI_AAPCS64)
433 return (type->category == INFIX_TYPE_STRUCT || type->category == INFIX_TYPE_UNION ||
434 type->category == INFIX_TYPE_ARRAY) &&
435 type->size > 16;
436#else
437 return false;
438#endif
439}

473static inline bool are_all_stack_args_homogeneous(infix_call_frame_layout * layout,
474 infix_type ** arg_types,
475 size_t num_args) {
476#if defined(BULK_MOVE_THRESHOLD)
477 if (layout->num_stack_args < BULK_MOVE_THRESHOLD)
478 return false;
479#endif
480
481 infix_type * first_stack_type = nullptr;
482 size_t first_stack_idx = 0;
483
484 // Find the first argument passed on the stack
485 for (size_t i = 0; i < num_args; ++i) {
486 if (layout->arg_locations[i].type == ARG_LOCATION_STACK) {
487 first_stack_type = arg_types[i];
488 first_stack_idx = i;
489 break;
490 }
491 }
492
493 if (!first_stack_type)
494 return false; // No stack arguments
495
496#if defined(INFIX_ABI_WINDOWS_X64)
497 if (is_passed_by_reference(first_stack_type) || first_stack_type->size != 8)
498 return false;
499#else
500 // This optimization is only safe for simple 8-byte types (double, uint64_t, pointers, etc.).
501 // It correctly excludes aggregates passed on the stack.
502 if (first_stack_type->size != 8)
503 return false;
504#endif
505
506 // Check that all subsequent stack arguments are of the exact same type.
507 for (size_t i = first_stack_idx + 1; i < num_args; ++i) {
508 if (layout->arg_locations[i].type == ARG_LOCATION_STACK) {
509 if (arg_types[i] != first_stack_type)
510 return false;
511 }
512 }
513
514 return true;
515}
