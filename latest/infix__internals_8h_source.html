378static inline bool is_passed_by_reference(infix_type * type) {
379 // On Windows x64, aggregates whose size is not a power of two (1, 2, 4, 8 bytes)
380 // are passed by reference. This also applies to sizes larger than 8 bytes on MSVC.
381 // However, GCC/Clang on Windows passes 16-byte types by value on the stack.
382 // To ensure compatibility, we treat 16-byte types as pass-by-value.
383 //~ if (type->size == 16)
384 //~ return false;
385 return type->size != 1 && type->size != 2 && type->size != 4 && type->size != 8;
386}

396static inline bool return_uses_hidden_pointer_abi(c23_maybe_unused infix_type * type) {
397#if defined(INFIX_ABI_WINDOWS_X64)
398#if defined(INFIX_COMPILER_GCC)
399 // On GCC for Windows, its 16-byte long double is a special case returned by reference.
400 if (is_long_double(type))
401 return true;
402#endif
403 // For all other compilers (MSVC, Clang) and for aggregate types on GCC, the rule is size-based.
404 if (type->category == INFIX_TYPE_STRUCT || type->category == INFIX_TYPE_UNION || type->category == INFIX_TYPE_ARRAY)
405 return is_passed_by_reference(type);
406 return false; // Other primitives (including __int128_t) are returned by value.
407#elif defined(INFIX_ABI_SYSV_X64)
408 // long double is not returned via hidden pointer on SysV, but on the x87 stack.
409 // The check here is just for large aggregates.
410 return (type->category == INFIX_TYPE_STRUCT || type->category == INFIX_TYPE_UNION ||
411 type->category == INFIX_TYPE_ARRAY) &&
412 type->size > 16;
413#elif defined(INFIX_ABI_AAPCS64)
414 return (type->category == INFIX_TYPE_STRUCT || type->category == INFIX_TYPE_UNION ||
415 type->category == INFIX_TYPE_ARRAY) &&
416 type->size > 16;
417#else
418 return false;
419#endif
420}

454static inline bool are_all_stack_args_homogeneous(infix_call_frame_layout * layout,
455 infix_type ** arg_types,
456 size_t num_args) {
457#if defined(BULK_MOVE_THRESHOLD)
458 if (layout->num_stack_args < BULK_MOVE_THRESHOLD)
459 return false;
460#endif
461
462 infix_type * first_stack_type = nullptr;
463 size_t first_stack_idx = 0;
464
465 // Find the first argument passed on the stack
466 for (size_t i = 0; i < num_args; ++i) {
467 if (layout->arg_locations[i].type == ARG_LOCATION_STACK) {
468 first_stack_type = arg_types[i];
469 first_stack_idx = i;
470 break;
471 }
472 }
473
474 if (!first_stack_type)
475 return false; // No stack arguments
476
477#if defined(INFIX_ABI_WINDOWS_X64)
478 if (is_passed_by_reference(first_stack_type) || first_stack_type->size != 8)
479 return false;
480#else
481 // This optimization is only safe for simple 8-byte types (double, uint64_t, pointers, etc.).
482 // It correctly excludes aggregates passed on the stack.
483 if (first_stack_type->size != 8)
484 return false;
485#endif
486
487 // Check that all subsequent stack arguments are of the exact same type.
488 for (size_t i = first_stack_idx + 1; i < num_args; ++i) {
489 if (layout->arg_locations[i].type == ARG_LOCATION_STACK) {
490 if (arg_types[i] != first_stack_type)
491 return false;
492 }
493 }
494
495 return true;
496}
