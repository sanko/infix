189c23_nodiscard infix_status infix_type_create_struct(infix_arena_t * arena,
190 infix_type ** out_type,
191 infix_struct_member * members,
192 size_t num_members) {
193 if (out_type == nullptr)
194 return INFIX_ERROR_INVALID_ARGUMENT;
195
196 // Validate that all member types are non-null.
197 for (size_t i = 0; i < num_members; ++i) {
198 if (members[i].type == nullptr) {
199 *out_type = nullptr;
200 return INFIX_ERROR_INVALID_ARGUMENT;
201 }
202 }
203
204 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
205 if (type == nullptr) {
206 *out_type = nullptr;
207 return INFIX_ERROR_ALLOCATION_FAILED;
208 }
209 type->is_arena_allocated = true;
210 type->category = INFIX_TYPE_STRUCT;
211 type->meta.aggregate_info.members = members;
212 type->meta.aggregate_info.num_members = num_members;
213
214 size_t preliminary_size = 0;
215 size_t max_alignment = 1;
216
217 // First pass: determine the natural size and max alignment.
218 for (size_t i = 0; i < num_members; ++i) {
219 infix_struct_member * member = &members[i];
220
221 // Security: Check for integer overflow when calculating member end.
222 if (member->offset > SIZE_MAX - member->type->size) {
223 *out_type = nullptr;
224 return INFIX_ERROR_INVALID_ARGUMENT;
225 }
226
227 size_t member_end = member->offset + member->type->size;
228 if (member_end > preliminary_size)
229 preliminary_size = member_end;
230 if (member->type->alignment > max_alignment)
231 max_alignment = member->type->alignment;
232 }
233
234 type->alignment = max_alignment;
235
236 // Check for overflow when calculating padding.
237 if (max_alignment > 0 && preliminary_size > SIZE_MAX - (max_alignment - 1)) {
238 *out_type = nullptr;
239 return INFIX_ERROR_INVALID_ARGUMENT;
240 }
241
242 // Final size is the preliminary size rounded up to the nearest multiple of the max alignment.
243 if (max_alignment > 0)
244 type->size = (preliminary_size + max_alignment - 1) & ~(max_alignment - 1);
245 else
246 type->size = preliminary_size;
247
248 INFIX_DEBUG_PRINTF("Created struct type. Size: %llu, Alignment: %llu",
249 (unsigned long long)type->size,
250 (unsigned long long)type->alignment);
251 *out_type = type;
252 return INFIX_SUCCESS;
253}

337c23_nodiscard infix_status infix_type_create_union(infix_arena_t * arena,
338 infix_type ** out_type,
339 infix_struct_member * members,
340 size_t num_members) {
341 if (out_type == NULL) {
342 return INFIX_ERROR_INVALID_ARGUMENT;
343 }
344
345 // Validate that all member types are non-null before proceeding.
346 for (size_t i = 0; i < num_members; ++i) {
347 if (members[i].type == NULL) {
348 *out_type = NULL;
349 return INFIX_ERROR_INVALID_ARGUMENT;
350 }
351 }
352
353 // Allocate the infix_type struct itself from the arena.
354 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
355 if (type == NULL) {
356 *out_type = NULL;
357 return INFIX_ERROR_ALLOCATION_FAILED;
358 }
359
360 // Mark this type as arena-allocated so infix_type_destroy will ignore it.
361 type->is_arena_allocated = true;
362 type->category = INFIX_TYPE_UNION;
363 type->meta.aggregate_info.members = members;
364 type->meta.aggregate_info.num_members = num_members;
365
366 size_t max_size = 0;
367 size_t max_alignment = 1;
368
369 // A union's size is determined by its largest member, and its alignment
370 // by the strictest alignment of any member.
371 for (size_t i = 0; i < num_members; ++i) {
372 if (members[i].type->size > max_size) {
373 max_size = members[i].type->size;
374 }
375 if (members[i].type->alignment > max_alignment) {
376 max_alignment = members[i].type->alignment;
377 }
378 }
379 type->alignment = max_alignment;
380
381 // Security: Check for integer overflow before calculating the final padded size.
382 if (max_alignment > 0 && max_size > SIZE_MAX - (max_alignment - 1)) {
383 *out_type = NULL;
384 return INFIX_ERROR_INVALID_ARGUMENT; // Overflow would occur
385 }
386
387 // The final size is the size of the largest member, rounded up to a
388 // multiple of the union's overall alignment.
389 type->size = (max_size + max_alignment - 1) & ~(max_alignment - 1);
390
391 INFIX_DEBUG_PRINTF("Created arena union type. Size: %llu, Alignment: %llu",
392 (unsigned long long)type->size,
393 (unsigned long long)type->alignment);
394
395 *out_type = type;
396 return INFIX_SUCCESS;
397}

412c23_nodiscard infix_status infix_type_create_array(infix_arena_t * arena,
413 infix_type ** out_type,
414 infix_type * element_type,
415 size_t num_elements) {
416 if (out_type == NULL || element_type == NULL)
417 return INFIX_ERROR_INVALID_ARGUMENT;
418
419 // Security: Check for integer overflow before calculating the total array size.
420 // This is critical when dealing with inputs from a parser or other untrusted source.
421 if (element_type->size > 0 && num_elements > SIZE_MAX / element_type->size) {
422 *out_type = NULL;
423 return INFIX_ERROR_INVALID_ARGUMENT; // Calculation would overflow.
424 }
425
426 // Allocate the infix_type struct itself from the arena.
427 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
428 if (type == NULL) {
429 *out_type = NULL;
430 return INFIX_ERROR_ALLOCATION_FAILED;
431 }
432
433 // Mark this type as arena-allocated so infix_type_destroy will ignore it.
434 type->is_arena_allocated = true;
435 type->category = INFIX_TYPE_ARRAY;
436 type->meta.array_info.element_type = element_type;
437 type->meta.array_info.num_elements = num_elements;
438
439 // An array's alignment is the same as its element's alignment.
440 type->alignment = element_type->alignment;
441 type->size = element_type->size * num_elements;
442
443 INFIX_DEBUG_PRINTF("Created arena array type. Size: %llu, Alignment: %llu",
444 (unsigned long long)type->size,
445 (unsigned long long)type->alignment);
446
447 *out_type = type;
448 return INFIX_SUCCESS;
449}
