228 type->size = base_type->size * 2;

230 type->meta.complex_info.base_type = base_type;

254c23_nodiscard infix_status infix_type_create_struct(infix_arena_t * arena,
255 infix_type ** out_type,
256 infix_struct_member * members,
257 size_t num_members) {
258 if (out_type == nullptr)
259 return INFIX_ERROR_INVALID_ARGUMENT;
260
261 // Validate that all member types are non-null before proceeding.
262 for (size_t i = 0; i < num_members; ++i) {
263 if (members[i].type == nullptr) {
264 *out_type = nullptr;
265 return INFIX_ERROR_INVALID_ARGUMENT;
266 }
267 }
268
269 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
270 if (type == nullptr) {
271 *out_type = nullptr;
272 return INFIX_ERROR_ALLOCATION_FAILED;
273 }
274
275 // Robustness: Copy the caller's member data into the arena to make this
276 // type object self-contained and immune to use-after-free errors.
277 infix_struct_member * arena_members = nullptr;
278 if (num_members > 0) {
279 arena_members =
280 infix_arena_alloc(arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
281 if (arena_members == nullptr) {
282 *out_type = nullptr;
283 return INFIX_ERROR_ALLOCATION_FAILED;
284 }
285 memcpy(arena_members, members, sizeof(infix_struct_member) * num_members);
286 }
287
288 type->is_arena_allocated = true;
289 type->category = INFIX_TYPE_STRUCT;
290 type->meta.aggregate_info.members = arena_members;
291 type->meta.aggregate_info.num_members = num_members;
292
293 size_t current_offset = 0;
294 // Initialize max_alignment to 1. This is the alignment of an empty struct
295 // and a safe starting point for calculations.
296 size_t max_alignment = 1;
297
298 // Calculate layout based on the safe, arena-allocated copy of members.
299 for (size_t i = 0; i < num_members; ++i) {
300 infix_struct_member * member = &arena_members[i];
301 size_t member_align = member->type->alignment;
302
303 if (member_align == 0) { // Should not happen for valid types.
304 *out_type = nullptr;
305 return INFIX_ERROR_INVALID_ARGUMENT;
306 }
307
308 // Calculate padding required to align the current offset for this member.
309 size_t padding = (member_align - (current_offset % member_align)) % member_align;
310
311 // Security: Check for integer overflow before adding padding.
312 if (current_offset > SIZE_MAX - padding) {
313 *out_type = nullptr;
314 return INFIX_ERROR_INVALID_ARGUMENT;
315 }
316 current_offset += padding;
317
318 // Set the final calculated offset for this member.
319 member->offset = current_offset;
320
321 // Security: Check for integer overflow before adding the member's size.
322 if (current_offset > SIZE_MAX - member->type->size) {
323 *out_type = nullptr;
324 return INFIX_ERROR_INVALID_ARGUMENT;
325 }
326 current_offset += member->type->size;
327
328 // Update the maximum alignment required by the struct.
329 if (member_align > max_alignment)
330 max_alignment = member_align;
331 }
332
333 type->alignment = max_alignment;
334
335 // Security: Check for overflow when calculating the final rounded-up size.
336 if (max_alignment > 0 && current_offset > SIZE_MAX - (max_alignment - 1)) {
337 *out_type = nullptr;
338 return INFIX_ERROR_INVALID_ARGUMENT;
339 }
340
341 // The final size is the calculated offset rounded up to the nearest multiple
342 // of the struct's overall alignment to account for trailing padding.
343 type->size = (current_offset + max_alignment - 1) & ~(max_alignment - 1);
344
345 INFIX_DEBUG_PRINTF("Created struct type. Size: %llu, Alignment: %llu",
346 (unsigned long long)type->size,
347 (unsigned long long)type->alignment);
348 *out_type = type;
349 return INFIX_SUCCESS;
350}

441c23_nodiscard infix_status infix_type_create_union(infix_arena_t * arena,
442 infix_type ** out_type,
443 infix_struct_member * members,
444 size_t num_members) {
445 if (out_type == nullptr)
446 return INFIX_ERROR_INVALID_ARGUMENT;
447
448 // Validate that all member types are non-null before proceeding.
449 for (size_t i = 0; i < num_members; ++i) {
450 if (members[i].type == nullptr) {
451 *out_type = nullptr;
452 return INFIX_ERROR_INVALID_ARGUMENT;
453 }
454 }
455
456 // Allocate the infix_type struct itself from the arena.
457 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
458 if (type == nullptr) {
459 *out_type = nullptr;
460 return INFIX_ERROR_ALLOCATION_FAILED;
461 }
462
463 // Robustness: Copy the caller's member data into the arena.
464 infix_struct_member * arena_members = nullptr;
465 if (num_members > 0) {
466 arena_members =
467 infix_arena_alloc(arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
468 if (arena_members == nullptr) {
469 *out_type = nullptr;
470 return INFIX_ERROR_ALLOCATION_FAILED;
471 }
472 memcpy(arena_members, members, sizeof(infix_struct_member) * num_members);
473 }
474
475 // Mark this type as arena-allocated so infix_type_destroy will ignore it.
476 type->is_arena_allocated = true;
477 type->category = INFIX_TYPE_UNION;
478 type->meta.aggregate_info.members = arena_members;
479 type->meta.aggregate_info.num_members = num_members;
480
481 size_t max_size = 0;
482 size_t max_alignment = 1;
483
484 // A union's size is determined by its largest member, and its alignment
485 // by the strictest alignment of any member.
486 for (size_t i = 0; i < num_members; ++i) {
487 arena_members[i].offset = 0; // All union members are at offset 0.
488 if (arena_members[i].type->size > max_size)
489 max_size = arena_members[i].type->size;
490 if (arena_members[i].type->alignment > max_alignment)
491 max_alignment = arena_members[i].type->alignment;
492 }
493 type->alignment = max_alignment;
494
495 // Security: Check for integer overflow before calculating the final padded size.
496 if (max_alignment > 0 && max_size > SIZE_MAX - (max_alignment - 1)) {
497 *out_type = nullptr;
498 return INFIX_ERROR_INVALID_ARGUMENT; // Overflow would occur
499 }
500
501 // The final size is the size of the largest member, rounded up to a
502 // multiple of the union's overall alignment.
503 type->size = (max_size + max_alignment - 1) & ~(max_alignment - 1);
504
505 INFIX_DEBUG_PRINTF("Created arena union type. Size: %llu, Alignment: %llu",
506 (unsigned long long)type->size,
507 (unsigned long long)type->alignment);
508
509 *out_type = type;
510 return INFIX_SUCCESS;
511}

526c23_nodiscard infix_status infix_type_create_array(infix_arena_t * arena,
527 infix_type ** out_type,
528 infix_type * element_type,
529 size_t num_elements) {
530 if (out_type == nullptr || element_type == nullptr)
531 return INFIX_ERROR_INVALID_ARGUMENT;
532
533 // Security: Check for integer overflow before calculating the total array size.
534 // This is critical when dealing with inputs from a parser or other untrusted source.
535 if (element_type->size > 0 && num_elements > SIZE_MAX / element_type->size) {
536 *out_type = nullptr;
537 return INFIX_ERROR_INVALID_ARGUMENT; // Calculation would overflow.
538 }
539
540 // Allocate the infix_type struct itself from the arena.
541 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
542 if (type == nullptr) {
543 *out_type = nullptr;
544 return INFIX_ERROR_ALLOCATION_FAILED;
545 }
546
547 // Mark this type as arena-allocated so infix_type_destroy will ignore it.
548 type->is_arena_allocated = true;
549 type->category = INFIX_TYPE_ARRAY;
550 type->meta.array_info.element_type = element_type;
551 type->meta.array_info.num_elements = num_elements;
552
553 // An array's alignment is the same as its element's alignment.
554 type->alignment = element_type->alignment;
555 type->size = element_type->size * num_elements;
556
557 INFIX_DEBUG_PRINTF("Created arena array type. Size: %llu, Alignment: %llu",
558 (unsigned long long)type->size,
559 (unsigned long long)type->alignment);
560
561 *out_type = type;
562 return INFIX_SUCCESS;
563}
