222 }

224 type->is_arena_allocated = true;

226

228 type->size = base_type->size * 2;

230 type->meta.complex_info.base_type = base_type;

232 *out_type = type;
233 return INFIX_SUCCESS;

254c23_nodiscard infix_status infix_type_create_struct(infix_arena_t * arena,
255 infix_type ** out_type,
256 infix_struct_member * members,
257 size_t num_members) {
258 if (out_type == nullptr)
259 return INFIX_ERROR_INVALID_ARGUMENT;
260
261 // Validate that all member types are non-null before proceeding.
262 for (size_t i = 0; i < num_members; ++i) {
263 if (members[i].type == nullptr) {
264 *out_type = nullptr;
265 return INFIX_ERROR_INVALID_ARGUMENT;
266 }
267 }
268
269 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
270 if (type == nullptr) {
271 *out_type = nullptr;
272 return INFIX_ERROR_ALLOCATION_FAILED;
273 }
274
275 // Robustness: Copy the caller's member data into the arena to make this
276 // type object self-contained and immune to use-after-free errors.
277 infix_struct_member * arena_members = nullptr;
278 if (num_members > 0) {
279 arena_members =
280 infix_arena_alloc(arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
281 if (arena_members == nullptr) {
282 *out_type = nullptr;
283 return INFIX_ERROR_ALLOCATION_FAILED;
284 }
285 memcpy(arena_members, members, sizeof(infix_struct_member) * num_members);
286 }
287
288 type->is_arena_allocated = true;
289 type->category = INFIX_TYPE_STRUCT;
290 type->meta.aggregate_info.members = arena_members;
291 type->meta.aggregate_info.num_members = num_members;
292
293 size_t current_offset = 0;
294 // Initialize max_alignment to 1. This is the alignment of an empty struct
295 // and a safe starting point for calculations.
296 size_t max_alignment = 1;
297
298 // Calculate layout based on the safe, arena-allocated copy of members.
299 for (size_t i = 0; i < num_members; ++i) {
300 infix_struct_member * member = &arena_members[i];
301 size_t member_align = member->type->alignment;
302
303 if (member_align == 0) { // Should not happen for valid types.
304 *out_type = nullptr;
305 return INFIX_ERROR_INVALID_ARGUMENT;
306 }
307
308 // Calculate padding required to align the current offset for this member.
309 size_t padding = (member_align - (current_offset % member_align)) % member_align;
310
311 // Security: Check for integer overflow before adding padding.
312 if (current_offset > SIZE_MAX - padding) {
313 *out_type = nullptr;
314 return INFIX_ERROR_INVALID_ARGUMENT;
315 }
316 current_offset += padding;
317
318 // Set the final calculated offset for this member.
319 member->offset = current_offset;
320
321 // Security: Check for integer overflow before adding the member's size.
322 if (current_offset > SIZE_MAX - member->type->size) {
323 *out_type = nullptr;
324 return INFIX_ERROR_INVALID_ARGUMENT;
325 }
326 current_offset += member->type->size;
327
328 // Update the maximum alignment required by the struct.
329 if (member_align > max_alignment)
330 max_alignment = member_align;
331 }
332
333 type->alignment = max_alignment;
334
335 // Security: Check for overflow when calculating the final rounded-up size.
336 if (max_alignment > 0 && current_offset > SIZE_MAX - (max_alignment - 1)) {
337 *out_type = nullptr;
338 return INFIX_ERROR_INVALID_ARGUMENT;
339 }
340
341 // The final size is the calculated offset rounded up to the nearest multiple
342 // of the struct's overall alignment to account for trailing padding.
343 type->size = (current_offset + max_alignment - 1) & ~(max_alignment - 1);
344
345 INFIX_DEBUG_PRINTF("Created struct type. Size: %llu, Alignment: %llu",
346 (unsigned long long)type->size,
347 (unsigned long long)type->alignment);
348 *out_type = type;
349 return INFIX_SUCCESS;
350}

425c23_nodiscard infix_status infix_type_create_vector(infix_arena_t * arena,
426 infix_type ** out_type,
427 infix_type * element_type,
428 size_t num_elements) {
429 if (out_type == nullptr || element_type == nullptr || element_type->category != INFIX_TYPE_PRIMITIVE)
430 return INFIX_ERROR_INVALID_ARGUMENT;
431
432 // Security: Check for integer overflow before calculating the total vector size.
433 if (element_type->size > 0 && num_elements > SIZE_MAX / element_type->size) {
434 *out_type = nullptr;
435 return INFIX_ERROR_INVALID_ARGUMENT;
436 }
437
438 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
439 if (type == nullptr) {
440 *out_type = nullptr;
441 return INFIX_ERROR_ALLOCATION_FAILED;
442 }
443
444 type->is_arena_allocated = true;
445 type->category = INFIX_TYPE_VECTOR;
446 type->meta.vector_info.element_type = element_type;
447 type->meta.vector_info.num_elements = num_elements;
448
449 // A vector's size is the total size of its elements.
450 type->size = element_type->size * num_elements;
451 // Common ABIs require 128-bit vectors to be 16-byte aligned.
452 // We will enforce this alignment, as it's the strictest requirement.
453 type->alignment = type->size > 8 ? 16 : type->size;
454
455 *out_type = type;
456 return INFIX_SUCCESS;
457}

476c23_nodiscard infix_status infix_type_create_union(infix_arena_t * arena,
477 infix_type ** out_type,
478 infix_struct_member * members,
479 size_t num_members) {
480 if (out_type == nullptr)
481 return INFIX_ERROR_INVALID_ARGUMENT;
482
483 // Validate that all member types are non-null before proceeding.
484 for (size_t i = 0; i < num_members; ++i) {
485 if (members[i].type == nullptr) {
486 *out_type = nullptr;
487 return INFIX_ERROR_INVALID_ARGUMENT;
488 }
489 }
490
491 // Allocate the infix_type struct itself from the arena.
492 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
493 if (type == nullptr) {
494 *out_type = nullptr;
495 return INFIX_ERROR_ALLOCATION_FAILED;
496 }
497
498 // Robustness: Copy the caller's member data into the arena.
499 infix_struct_member * arena_members = nullptr;
500 if (num_members > 0) {
501 arena_members =
502 infix_arena_alloc(arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
503 if (arena_members == nullptr) {
504 *out_type = nullptr;
505 return INFIX_ERROR_ALLOCATION_FAILED;
506 }
507 memcpy(arena_members, members, sizeof(infix_struct_member) * num_members);
508 }
509
510 // Mark this type as arena-allocated so infix_type_destroy will ignore it.
511 type->is_arena_allocated = true;
512 type->category = INFIX_TYPE_UNION;
513 type->meta.aggregate_info.members = arena_members;
514 type->meta.aggregate_info.num_members = num_members;
515
516 size_t max_size = 0;
517 size_t max_alignment = 1;
518
519 // A union's size is determined by its largest member, and its alignment
520 // by the strictest alignment of any member.
521 for (size_t i = 0; i < num_members; ++i) {
522 arena_members[i].offset = 0; // All union members are at offset 0.
523 if (arena_members[i].type->size > max_size)
524 max_size = arena_members[i].type->size;
525 if (arena_members[i].type->alignment > max_alignment)
526 max_alignment = arena_members[i].type->alignment;
527 }
528 type->alignment = max_alignment;
529
530 // Security: Check for integer overflow before calculating the final padded size.
531 if (max_alignment > 0 && max_size > SIZE_MAX - (max_alignment - 1)) {
532 *out_type = nullptr;
533 return INFIX_ERROR_INVALID_ARGUMENT; // Overflow would occur
534 }
535
536 // The final size is the size of the largest member, rounded up to a
537 // multiple of the union's overall alignment.
538 type->size = (max_size + max_alignment - 1) & ~(max_alignment - 1);
539
540 INFIX_DEBUG_PRINTF("Created arena union type. Size: %llu, Alignment: %llu",
541 (unsigned long long)type->size,
542 (unsigned long long)type->alignment);
543
544 *out_type = type;
545 return INFIX_SUCCESS;
546}
