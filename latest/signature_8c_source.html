94static void skip_whitespace(parser_state * state) {
95 while (true) {
96 // First, consume all standard whitespace characters.
97 while (isspace((unsigned char)*state->p))
98 state->p++;
99 // Next, check for the start of a line comment.
100 if (*state->p == '#') {
101 // If found, consume all characters until the next newline or the end of the string.
102 while (*state->p != '\n' && *state->p != '\0')
103 state->p++;
104 }
105 else
106 // If neither whitespace nor a comment is found, we're done.
107 break;
108 }
109}

120static bool parse_size_t(parser_state * state, size_t * out_val) {
121 const char * start = state->p;
122 char * end;
123 // Use strtoull for battle tested parsing of 64-bit unsigned integers.
124 unsigned long long val = strtoull(start, &end, 10);
125 // If the end pointer hasn't moved, no digits were consumed.
126 if (end == start) {
127 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
128 return false;
129 }
130 *out_val = (size_t)val;
131 state->p = end; // Advance the parser's pointer.
132 return true;
133}

144static const char * parse_identifier(parser_state * state) {
145 skip_whitespace(state);
146 const char * start = state->p;
147 // An identifier must start with an alphabetic character or an underscore.
148 if (!isalpha((unsigned char)*start) && *start != '_')
149 return nullptr;
150 // Consume subsequent alphanumeric characters or underscores.
151 while (isalnum((unsigned char)*state->p) || *state->p == '_')
152 state->p++;
153 size_t len = state->p - start;
154 if (len == 0)
155 return nullptr;
156
157 // Allocate space for the identifier in the arena and copy it.
158 char * name = infix_arena_alloc(state->arena, len + 1, 1);
159 if (!name) {
160 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
161 return nullptr;
162 }
163 memcpy(name, start, len);
164 name[len] = '\0';
165 return name;
166}

180static bool consume_keyword(parser_state * state, const char * keyword) {
181 skip_whitespace(state);
182 size_t len = strlen(keyword);
183 if (strncmp(state->p, keyword, len) == 0) {
184 // This is the "whole word" check. The character after the keyword
185 // cannot be alphanumeric, which would imply it's part of a larger word.
186 if (isalnum((unsigned char)state->p[len]) || state->p[len] == '_')
187 return false;
188
189 // The keyword matched. Advance the pointer and return success.
190 state->p += len;
191 skip_whitespace(state);
192 return true;
193 }
194 return false;
195}

265 if (consume_keyword(state, "double"))
266 return infix_type_create_primitive(INFIX_PRIMITIVE_DOUBLE);
267 if (consume_keyword(state, "float"))
268 return infix_type_create_primitive(INFIX_PRIMITIVE_FLOAT);
269 // If no keyword matches, it's not a primitive type.
270 return nullptr;

289static bool is_function_signature_ahead(const parser_state * state) {
290 const char * p = state->p;
291 // The ambiguity only exists if we start with '('.
292 if (*p != '(')
293 return false;
294 p++;
295
296 // Find the matching closing parenthesis, respecting nested parentheses.
297 int depth = 1;
298 while (*p != '\0' && depth > 0) {
299 if (*p == '(')
300 depth++;
301 else if (*p == ')')
302 depth--;
303 p++;
304 }
305 // If we didn't find a matching parenthesis, it's malformed.
306 if (depth != 0)
307 return false;
308
309 // Skip any whitespace or comments between the ')' and the next token.
310 while (isspace((unsigned char)*p) || *p == '#') {
311 if (*p == '#') {
312 while (*p != '\n' && *p != '\0')
313 p++;
314 }
315 else
316 p++;
317 }
318
319 // The upcoming construct is a function signature if and only if the next
320 // non-whitespace token is the arrow '->'.
321 return (p[0] == '-' && p[1] == '>');
322}

341static infix_type * parse_function_type(parser_state * state) {
342 infix_type * ret_type = nullptr;
343 infix_function_argument * args = nullptr;
344 size_t num_args = 0, num_fixed = 0;
345
346 // Delegate the heavy lifting of parsing the argument list, arrow, and return type.
347 if (parse_function_signature_details(state, &ret_type, &args, &num_args, &num_fixed) != INFIX_SUCCESS)
348 return nullptr;
349
350 // If parsing was successful, allocate a new type descriptor to represent the function signature.
351 infix_type * func_type = infix_arena_alloc(state->arena, sizeof(infix_type), _Alignof(infix_type));
352 if (!func_type) {
353 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
354 return nullptr;
355 }
356
357 // A function signature type itself has the size and alignment of a pointer.
358 *func_type = *infix_type_create_pointer();
359 func_type->is_arena_allocated = true;
360 func_type->category = INFIX_TYPE_REVERSE_TRAMPOLINE;
361
362 // Populate the metadata with the parsed components.
363 func_type->meta.func_ptr_info.return_type = ret_type;
364 func_type->meta.func_ptr_info.args = args;
365 func_type->meta.func_ptr_info.num_args = num_args;
366 func_type->meta.func_ptr_info.num_fixed_args = num_fixed;
367
368 return func_type;
369}

391static infix_struct_member * parse_aggregate_members(parser_state * state, char end_char, size_t * out_num_members) {
392 // A temporary node for building the linked list of members.
393 typedef struct member_node {
394 infix_struct_member m;
395 struct member_node * next;
396 } member_node;
397 member_node *head = nullptr, *tail = nullptr;
398 size_t num_members = 0;
399
400 skip_whitespace(state);
401 // Handle empty aggregates like `{}` or `<>` by checking for the end character immediately.
402 if (*state->p != end_char) {
403 while (1) {
404 skip_whitespace(state);
405 const char * p_before_member = state->p;
406
407 // Tentatively parse an identifier. This could be a member name.
408 const char * name = parse_identifier(state);
409 skip_whitespace(state);
410
411 // A valid named member has the form "name : type". If we found an
412 // identifier and it's followed by a colon, we treat it as a name.
413 if (name && *state->p == ':') {
414 state->p++; // Consume the colon.
415 skip_whitespace(state);
416 }
417 else {
418 // If there's no colon, the identifier was actually the start of a
419 // type keyword (e.g., "int"). We must backtrack the parser to the
420 // position before we tried to parse it as a name.
421 name = nullptr;
422 state->p = p_before_member;
423 }
424
425 // A syntax error: `{name:}` with no type.
426 if (name && (*state->p == end_char || *state->p == ',')) {
427 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
428 return nullptr;
429 }
430
431 // Now, recursively parse the member's type.
432 infix_type * member_type = parse_type(state);
433 if (!member_type) {
434 // Propagate the error from the recursive call.
435 return nullptr;
436 }
437 // The C standard forbids members of type void.
438 if (member_type->category == INFIX_TYPE_VOID) {
439 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
440 return nullptr;
441 }
442
443 // Add the parsed member to our linked list.
444 member_node * node = infix_arena_alloc(state->arena, sizeof(member_node), _Alignof(member_node));
445 if (!node) {
446 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
447 return nullptr;
448 }
449 node->m = infix_struct_member_create(name, member_type, 0); // Offset is calculated later.
450 node->next = nullptr;
451 if (!head)
452 head = tail = node;
453 else {
454 tail->next = node;
455 tail = node;
456 }
457 num_members++;
458
459 skip_whitespace(state);
460 // After a member, we expect either a comma or the end character.
461 if (*state->p == ',') {
462 state->p++;
463 // A trailing comma like `{int,}` is a syntax error.
464 if (*state->p == end_char) {
465 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
466 return nullptr;
467 }
468 }
469 else if (*state->p == end_char)
470 break; // End of the list.
471 else {
472 // Anything else is a syntax error.
473 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
474 return nullptr;
475 }
476 }
477 }
478
479 *out_num_members = num_members;
480 if (num_members == 0)
481 return nullptr;
482
483 // Pass 2: Convert the linked list to a contiguous array.
484 infix_struct_member * members =
485 infix_arena_alloc(state->arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
486 if (!members) {
487 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
488 return nullptr;
489 }
490 member_node * current = head;
491 for (size_t i = 0; i < num_members; i++) {
492 members[i] = current->m;
493 current = current->next;
494 }
495 return members;
496}

508static infix_type * parse_packed_struct(parser_state * state) {
509 size_t alignment = 1; // The default alignment for a packed struct is 1.
510 if (*state->p == '!') {
511 state->p++;
512 // Check for an explicit alignment override, e.g., `!4:{...}`.
513 if (isdigit((unsigned char)*state->p)) {
514 if (!parse_size_t(state, &alignment))
515 return nullptr;
516 if (*state->p != ':') {
517 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
518 return nullptr;
519 }
520 state->p++;
521 }
522 }
523
524 skip_whitespace(state);
525 if (*state->p != '{') {
526 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
527 return nullptr;
528 }
529 state->p++;
530
531 size_t num_members = 0;
532 infix_struct_member * members = parse_aggregate_members(state, '}', &num_members);
533 if (state->last_error != INFIX_SUCCESS && num_members > 0)
534 return nullptr;
535
536 if (*state->p != '}') {
537 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
538 return nullptr;
539 }
540 state->p++;
541
542 infix_type * packed_type = nullptr;
543 // For packed structs, the size is simply the sum of member sizes, as there's
544 // no internal padding. The final layout calculation is handled by the `_create` function.
545 size_t total_size = 0;
546 for (size_t i = 0; i < num_members; ++i)
547 total_size += members[i].type->size;
548
549 infix_status status =
550 infix_type_create_packed_struct(state->arena, &packed_type, total_size, alignment, members, num_members);
551 if (status != INFIX_SUCCESS) {
552 state->last_error = status;
553 return nullptr;
554 }
555 return packed_type;
556}

576static infix_type * parse_type(parser_state * state) {
577 // Security: Prevent stack overflow from deeply nested type definitions.
578 if (state->depth >= MAX_RECURSION_DEPTH) {
579 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
580 return nullptr;
581 }
582 state->depth++;
583
584 skip_whitespace(state);
585 infix_type * result_type = nullptr;
586 const char * p_before_type = state->p; // Save position for backtracking on error.
587
588 // The dispatch logic begins here, checking one token at a time.
589 if (*state->p == '*') {
590 // POINTER TYPE: *<type>
591 state->p++; // Consume the '*'.
592 skip_whitespace(state);
593 // Recursively call parse_type to get the type the pointer points to.
594 infix_type * pointee_type = parse_type(state);
595 if (!pointee_type) {
596 state->depth--;
597 return nullptr;
598 } // Propagate failure.
599 // Create the final pointer type.
600 if (infix_type_create_pointer_to(state->arena, &result_type, pointee_type) != INFIX_SUCCESS) {
601 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
602 result_type = nullptr;
603 }
604 }
605 else if (*state->p == '(') {
606 // GROUPED TYPE or FUNCTION TYPE
607 // This is the ambiguous case. We must look ahead to see if it's a function.
608 if (is_function_signature_ahead(state))
609 result_type = parse_function_type(state);
610 else {
611 // It's a simple grouped type, e.g., *( (int)->void ).
612 state->p++; // Consume '('.
613 skip_whitespace(state);
614 result_type = parse_type(state); // Parse the inner type.
615 if (!result_type) {
616 state->depth--;
617 return nullptr;
618 }
619 skip_whitespace(state);
620 if (*state->p != ')') {
621 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
622 result_type = nullptr;
623 }
624 else {
625 state->p++; // Consume ')'.
626 }
627 }
628 }
629 else if (*state->p == '[') {
630 // ARRAY TYPE: [<size>:<type>]
631 state->p++;
632 skip_whitespace(state);
633 size_t num_elements;
634 if (!parse_size_t(state, &num_elements)) {
635 state->depth--;
636 return nullptr;
637 }
638 skip_whitespace(state);
639 if (*state->p != ':') {
640 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
641 state->depth--;
642 return nullptr;
643 }
644 state->p++;
645 skip_whitespace(state);
646 infix_type * element_type = parse_type(state);
647 if (!element_type) {
648 state->depth--;
649 return nullptr;
650 }
651 // Arrays of void are illegal in C.
652 if (element_type->category == INFIX_TYPE_VOID) {
653 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
654 state->depth--;
655 return nullptr;
656 }
657 skip_whitespace(state);
658 if (*state->p != ']') {
659 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
660 state->depth--;
661 return nullptr;
662 }
663 state->p++;
664 if (infix_type_create_array(state->arena, &result_type, element_type, num_elements) != INFIX_SUCCESS) {
665 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
666 result_type = nullptr;
667 }
668 }
669 else if (*state->p == '!')
670 // PACKED STRUCT TYPE: !<align>?:{...}
671 result_type = parse_packed_struct(state);
672 else if (*state->p == '{')
673 // ANONYMOUS STRUCT TYPE: {...}
674 result_type = parse_aggregate(state, '{', '}', nullptr);
675 else if (*state->p == '<')
676 // ANONYMOUS UNION TYPE: <...>
677 result_type = parse_aggregate(state, '<', '>', nullptr);
678 else if (consume_keyword(state, "struct")) {
679 // NAMED STRUCT or REFERENCE: struct<Name>{...} or struct<Name>
680 if (*state->p != '<') {
681 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
682 state->depth--;
683 return nullptr;
684 }
685 state->p++;
686 const char * name = parse_identifier(state);
687 if (!name) {
688 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
689 state->depth--;
690 return nullptr;
691 }
692 if (*state->p != '>') {
693 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
694 state->depth--;
695 return nullptr;
696 }
697 state->p++;
698 skip_whitespace(state);
699 // If a body follows, it's a definition.
700 if (*state->p == '{')
701 result_type = parse_aggregate(state, '{', '}', name);
702 else {
703 // Otherwise, it's a forward declaration (a named reference).
704 infix_status status = infix_type_create_named_reference(state->arena, &result_type, name);
705 if (status != INFIX_SUCCESS) {
706 state->last_error = status;
707 result_type = nullptr;
708 }
709 }
710 }
711 else if (consume_keyword(state, "union")) {
712 // NAMED UNION or REFERENCE: union<Name><...> or union<Name>
713 if (*state->p != '<') {
714 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
715 state->depth--;
716 return nullptr;
717 }
718 state->p++;
719 const char * name = parse_identifier(state);
720 if (!name) {
721 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
722 state->depth--;
723 return nullptr;
724 }
725 if (*state->p != '>') {
726 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
727 state->depth--;
728 return nullptr;
729 }
730 state->p++;
731 skip_whitespace(state);
732 // If a body follows, it's a definition.
733 if (*state->p == '<')
734 result_type = parse_aggregate(state, '<', '>', name);
735 else {
736 // Otherwise, it's a forward declaration (a named reference).
737 infix_status status = infix_type_create_named_reference(state->arena, &result_type, name);
738 if (status != INFIX_SUCCESS) {
739 state->last_error = status;
740 result_type = nullptr;
741 }
742 }
743 }
744 else if (*state->p == 'e') {
745 // ENUM TYPE: e<Name>?:<type>
746 state->p++;
747 skip_whitespace(state);
748 const char * name = nullptr;
749 // Check for optional name, e.g., e<MyEnum>:.
750 if (*state->p == '<') {
751 state->p++;
752 name = parse_identifier(state);
753 if (!name) {
754 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
755 state->depth--;
756 return nullptr;
757 }
758 if (*state->p != '>') {
759 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
760 state->depth--;
761 return nullptr;
762 }
763 state->p++;
764 skip_whitespace(state);
765 }
766 if (*state->p != ':') {
767 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
768 state->depth--;
769 return nullptr;
770 }
771 state->p++;
772 skip_whitespace(state);
773 // An enum must have an explicit underlying integer type.
774 infix_type * underlying_type = parse_type(state);
775 if (!underlying_type) {
776 state->depth--;
777 return nullptr;
778 }
779 // The spec requires the underlying type to be an integer.
780 if (underlying_type->category != INFIX_TYPE_PRIMITIVE) {
781 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
782 state->depth--;
783 return nullptr;
784 }
785 if (infix_type_create_enum(state->arena, &result_type, underlying_type) != INFIX_SUCCESS) {
786 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
787 result_type = nullptr;
788 }
789 (void)name; // Name is stored for introspection but not used by the core ABI.
790 }
791 else if (*state->p == 'c' && state->p[1] == '[') { // extra test to make sure we avoid eating `char`
792 // COMPLEX TYPE: c[<type>]
793 state->p++;
794 skip_whitespace(state);
795 if (*state->p != '[') {
796 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
797 state->depth--;
798 return NULL;
799 }
800 state->p++;
801 skip_whitespace(state);
802 infix_type * base_type = parse_type(state);
803 if (!base_type) {
804 state->depth--;
805 return NULL;
806 }
807 skip_whitespace(state);
808 if (*state->p != ']') {
809 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
810 state->depth--;
811 return NULL;
812 }
813 state->p++;
814 if (infix_type_create_complex(state->arena, &result_type, base_type) != INFIX_SUCCESS) {
815 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
816 result_type = NULL;
817 }
818 }
819 else {
820 // PRIMITIVE TYPE
821 // If none of the above constructors match, it must be a primitive type.
822 result_type = parse_primitive(state);
823 if (!result_type) {
824 // If even that fails, this is an unrecognized token. Backtrack and report error.
825 state->p = p_before_type;
826 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
827 }
828 }
829
830 state->depth--;
831 return result_type;
832}

850static infix_type * parse_aggregate(parser_state * state, char start_char, char end_char, const char * name) {
851 if (state->depth >= MAX_RECURSION_DEPTH) {
852 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
853 return nullptr;
854 }
855 state->depth++;
856
857 if (*state->p != start_char) {
858 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
859 state->depth--;
860 return nullptr;
861 }
862 state->p++;
863
864 size_t num_members = 0;
865 infix_struct_member * members = parse_aggregate_members(state, end_char, &num_members);
866 if (state->last_error != INFIX_SUCCESS) {
867 state->depth--;
868 return nullptr;
869 }
870
871 if (*state->p != end_char) {
872 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
873 state->depth--;
874 return nullptr;
875 }
876 state->p++;
877
878 infix_type * agg_type = nullptr;
879 // Dispatch to the correct creation function based on the delimiter.
880 infix_status status = (start_char == '{') ? infix_type_create_struct(state->arena, &agg_type, members, num_members)
881 : infix_type_create_union(state->arena, &agg_type, members, num_members);
882
883 if (status != INFIX_SUCCESS) {
884 state->last_error = status;
885 state->depth--;
886 return nullptr;
887 }
888 (void)name; // Name is stored for introspection.
889
890 state->depth--;
891 return agg_type;
892}

911static infix_status parse_function_signature_details(parser_state * state,
912 infix_type ** out_ret_type,
913 infix_function_argument ** out_args,
914 size_t * out_num_args,
915 size_t * out_num_fixed_args) {
916 if (*state->p != '(')
917 return INFIX_ERROR_INVALID_ARGUMENT;
918 state->p++;
919 skip_whitespace(state);
920
921 typedef struct arg_node {
922 infix_function_argument arg;
923 struct arg_node * next;
924 } arg_node;
925 arg_node *head = nullptr, *tail = nullptr;
926 size_t num_args = 0;
927
928 // Phase 1: Parse Fixed Arguments
929 if (*state->p != ')') {
930 while (1) {
931 skip_whitespace(state);
932 // The list of fixed arguments can be terminated by a ')' or a ';'.
933 if (*state->p == ')' || *state->p == ';')
934 break;
935
936 const char * p_before_arg = state->p;
937 const char * name = parse_identifier(state);
938 skip_whitespace(state);
939 if (name && *state->p == ':')
940 state->p++;
941 else {
942 name = nullptr;
943 state->p = p_before_arg;
944 }
945
946 infix_type * arg_type = parse_type(state);
947 if (!arg_type)
948 return state->last_error;
949
950 arg_node * node = infix_arena_alloc(state->arena, sizeof(arg_node), _Alignof(arg_node));
951 if (!node) {
952 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
953 return INFIX_ERROR_ALLOCATION_FAILED;
954 }
955 node->arg.type = arg_type;
956 node->arg.name = name;
957 node->next = nullptr;
958 if (!head)
959 head = tail = node;
960 else {
961 tail->next = node;
962 tail = node;
963 }
964 num_args++;
965
966 skip_whitespace(state);
967 if (*state->p == ',') {
968 state->p++;
969 skip_whitespace(state);
970 // A comma must be followed by another type, not the end of the list.
971 if (*state->p == ')' || *state->p == ';') {
972 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
973 return INFIX_ERROR_INVALID_ARGUMENT;
974 }
975 }
976 else if (*state->p != ')' && *state->p != ';')
977 return INFIX_ERROR_INVALID_ARGUMENT;
978 else
979 break; // End of the fixed argument list.
980 }
981 }
982 *out_num_fixed_args = num_args;
983
984 // Phase 2: Parse Variadic Arguments (if any)
985 if (*state->p == ';') {
986 state->p++;
987 if (*state->p != ')') { // Check for arguments after the separator.
988 while (1) {
989 skip_whitespace(state);
990 if (*state->p == ')')
991 break; // End of variadic list.
992
993 const char * p_before_arg = state->p;
994 const char * name = parse_identifier(state);
995 skip_whitespace(state);
996 if (name && *state->p == ':')
997 state->p++;
998 else {
999 name = nullptr;
1000 state->p = p_before_arg;
1001 }
1002
1003 infix_type * arg_type = parse_type(state);
1004 if (!arg_type)
1005 return state->last_error;
1006
1007 arg_node * node = infix_arena_alloc(state->arena, sizeof(arg_node), _Alignof(arg_node));
1008 if (!node) {
1009 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
1010 return INFIX_ERROR_ALLOCATION_FAILED;
1011 }
1012 node->arg.type = arg_type;
1013 node->arg.name = name;
1014 node->next = nullptr;
1015 if (!head)
1016 head = tail = node;
1017 else {
1018 tail->next = node;
1019 tail = node;
1020 }
1021 num_args++;
1022
1023 skip_whitespace(state);
1024 if (*state->p == ',') {
1025 state->p++;
1026 skip_whitespace(state);
1027 // A comma must be followed by another type, not the end of the list.
1028 if (*state->p == ')') {
1029 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
1030 return INFIX_ERROR_INVALID_ARGUMENT;
1031 }
1032 }
1033 else if (*state->p != ')')
1034 return INFIX_ERROR_INVALID_ARGUMENT;
1035 else
1036 break;
1037 }
1038 }
1039 }
1040
1041 skip_whitespace(state);
1042 if (*state->p != ')')
1043 return INFIX_ERROR_INVALID_ARGUMENT;
1044 state->p++;
1045 skip_whitespace(state);
1046
1047 if (state->p[0] != '-' || state->p[1] != '>')
1048 return INFIX_ERROR_INVALID_ARGUMENT;
1049 state->p += 2;
1050
1051 *out_ret_type = parse_type(state);
1052 if (!*out_ret_type)
1053 return state->last_error;
1054
1055 // Convert the linked list into a final, contiguous array.
1056 infix_function_argument * args = num_args > 0
1057 ? infix_arena_alloc(state->arena, sizeof(infix_function_argument) * num_args, _Alignof(infix_function_argument))
1058 : nullptr;
1059 if (num_args > 0 && !args) {
1060 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
1061 return INFIX_ERROR_ALLOCATION_FAILED;
1062 }
1063 arg_node * current = head;
1064 for (size_t i = 0; i < num_args; i++) {
1065 args[i] = current->arg;
1066 current = current->next;
1067 }
1068
1069 *out_args = args;
1070 *out_num_args = num_args;
1071
1072 return INFIX_SUCCESS;
1073}

1098c23_nodiscard infix_status infix_type_from_signature(infix_type ** out_type,
1099 infix_arena_t ** out_arena,
1100 const char * signature) {
1101 if (!out_type || !out_arena || !signature || *signature == '\0')
1102 return INFIX_ERROR_INVALID_ARGUMENT;
1103
1104 // Each parse operation creates its own arena for the resulting type graph.
1105 *out_arena = infix_arena_create(4096);
1106 if (!*out_arena)
1107 return INFIX_ERROR_ALLOCATION_FAILED;
1108
1109 // Initialize the parser state.
1110 parser_state state = {.p = signature, .arena = *out_arena, .depth = 0, .last_error = INFIX_SUCCESS};
1111 infix_type * type = parse_type(&state);
1112
1113 if (type) {
1114 skip_whitespace(&state);
1115 // A successful parse of a single type should consume the entire string.
1116 // If there are trailing characters, the signature is considered invalid.
1117 if (state.p[0] != '\0') {
1118 type = nullptr;
1119 state.last_error = INFIX_ERROR_INVALID_ARGUMENT;
1120 }
1121 }
1122
1123 // On any failure, we must destroy the arena to prevent memory leaks.
1124 if (!type) {
1125 infix_arena_destroy(*out_arena);
1126 *out_arena = nullptr;
1127 *out_type = nullptr;
1128 return state.last_error != INFIX_SUCCESS ? state.last_error : INFIX_ERROR_INVALID_ARGUMENT;
1129 }
1130
1131 *out_type = type;
1132 return INFIX_SUCCESS;
1133}

1149c23_nodiscard infix_status infix_signature_parse(const char * signature,
1150 infix_arena_t ** out_arena,
1151 infix_type ** out_ret_type,
1152 infix_function_argument ** out_args,
1153 size_t * out_num_args,
1154 size_t * out_num_fixed_args) {
1155 if (!signature || !out_arena || !out_ret_type || !out_args || !out_num_args || !out_num_fixed_args)
1156 return INFIX_ERROR_INVALID_ARGUMENT;
1157
1158 // Create a new arena for this specific parsing operation.
1159 *out_arena = infix_arena_create(8192); // 8KB default for potentially complex signatures.
1160 if (!*out_arena)
1161 return INFIX_ERROR_ALLOCATION_FAILED;
1162
1163 parser_state state = {.p = signature, .arena = *out_arena, .depth = 0, .last_error = INFIX_SUCCESS};
1164
1165 // Delegate to the internal implementation.
1166 infix_status status =
1167 parse_function_signature_details(&state, out_ret_type, out_args, out_num_args, out_num_fixed_args);
1168
1169 if (status == INFIX_SUCCESS) {
1170 skip_whitespace(&state);
1171 // Ensure the entire string was consumed.
1172 if (state.p[0] != '\0')
1173 status = INFIX_ERROR_INVALID_ARGUMENT;
1174 }
1175
1176 // Clean up the arena on any failure.
1177 if (status != INFIX_SUCCESS) {
1178 infix_arena_destroy(*out_arena);
1179 *out_arena = nullptr;
1180 return state.last_error != INFIX_SUCCESS ? state.last_error : status;
1181 }
1182
1183 return INFIX_SUCCESS;
1184}

1199c23_nodiscard infix_status infix_forward_create(infix_forward_t ** out_trampoline, const char * signature) {
1200 infix_arena_t * arena = nullptr;
1201 infix_type * ret_type = nullptr;
1202 infix_function_argument * args = nullptr;
1203 size_t num_args, num_fixed;
1204
1205 // First, parse the signature string into a type graph.
1206 infix_status status = infix_signature_parse(signature, &arena, &ret_type, &args, &num_args, &num_fixed);
1207 if (status != INFIX_SUCCESS)
1208 return status;
1209
1210 // Bridge: The manual API expects an array of `infix_type*`. We must extract
1211 // the type pointers from our parsed arguments to build this temporary array.
1212 infix_type ** arg_types =
1213 (num_args > 0) ? infix_arena_alloc(arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *)) : NULL;
1214 if (num_args > 0 && !arg_types) {
1215 infix_arena_destroy(arena);
1216 return INFIX_ERROR_ALLOCATION_FAILED;
1217 }
1218 for (size_t i = 0; i < num_args; ++i)
1219 arg_types[i] = args[i].type;
1220
1221 // Then, pass the resulting types to the manual creation function.
1222 status = infix_forward_create_manual(out_trampoline, ret_type, arg_types, num_args, num_fixed);
1223
1224 // Finally, destroy the temporary arena used for parsing.
1225 infix_arena_destroy(arena);
1226 return status;
1227}
