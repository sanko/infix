130void code_buffer_append(code_buffer * buf, const void * data, size_t len) {
131 if (buf->error) // If we're already in an error state, do nothing.
132 return;
133
134 if (len > SIZE_MAX - buf->size) {
135 //~ fprintf(stderr,
136 //~ "Error: Appending %llu bytes to buffer of size %llu would cause overflow.\n",
137 //~ (unsigned long long)len,
138 //~ (unsigned long long)buf->size);
139 buf->error = true;
140 return;
141 }
142
143 if (buf->size + len > buf->capacity) {
144 size_t new_capacity = buf->capacity;
145 while (new_capacity < buf->size + len) {
146 if (new_capacity > SIZE_MAX / 2) {
147 //~ fprintf(stderr, "Error: code_buffer cannot grow further without overflow.\n");
148 buf->error = true;
149 return;
150 }
151 new_capacity *= 2;
152 }
153
154 // Arena allocators don't support realloc. We must allocate a new block and copy.
155 void * new_code = infix_arena_alloc(buf->arena, new_capacity, 16);
156 if (new_code == nullptr) {
157 //~ fprintf(stderr, "Error: infix_arena_alloc failed in code_buffer_append.\n");
158 buf->error = true;
159 return;
160 }
161 // Copy the old code to the new, larger buffer.
162 infix_memcpy(new_code, buf->code, buf->size);
163 buf->code = new_code;
164 buf->capacity = new_capacity;
165 }
166 infix_memcpy(buf->code + buf->size, data, len);
167 buf->size += len;
168}

209static infix_type * _copy_type_graph_to_arena(infix_arena_t * dest_arena, const infix_type * src_type) {
210 if (src_type == nullptr)
211 return nullptr;
212
213 // If the source type is a static primitive (not from an arena), we don't need to copy it.
214 // We can just return the original pointer.
215 if (!src_type->is_arena_allocated)
216 return (infix_type *)src_type;
217
218 // Allocate space for the new type in the destination arena and copy the base struct.
219 infix_type * dest_type = infix_arena_alloc(dest_arena, sizeof(infix_type), _Alignof(infix_type));
220 if (dest_type == nullptr)
221 return nullptr;
222
223 memcpy(dest_type, src_type, sizeof(infix_type));
224
225 // Recursively copy any nested types based on the category.
226 switch (src_type->category) {
227 case INFIX_TYPE_POINTER:
228 dest_type->meta.pointer_info.pointee_type =
229 _copy_type_graph_to_arena(dest_arena, src_type->meta.pointer_info.pointee_type);
230 break;
231 case INFIX_TYPE_ARRAY:
232 dest_type->meta.array_info.element_type =
233 _copy_type_graph_to_arena(dest_arena, src_type->meta.array_info.element_type);
234 break;
235 case INFIX_TYPE_STRUCT:
236 case INFIX_TYPE_UNION:
237 if (src_type->meta.aggregate_info.num_members > 0) {
238 size_t members_size = sizeof(infix_struct_member) * src_type->meta.aggregate_info.num_members;
239 dest_type->meta.aggregate_info.members =
240 infix_arena_alloc(dest_arena, members_size, _Alignof(infix_struct_member));
241 if (dest_type->meta.aggregate_info.members == nullptr)
242 return nullptr; // Allocation failed.
243
244 memcpy(dest_type->meta.aggregate_info.members, src_type->meta.aggregate_info.members, members_size);
245
246 for (size_t i = 0; i < src_type->meta.aggregate_info.num_members; ++i)
247 dest_type->meta.aggregate_info.members[i].type =
248 _copy_type_graph_to_arena(dest_arena, src_type->meta.aggregate_info.members[i].type);
249 }
250 break;
251 // Other cases like ENUM, COMPLEX, VECTOR, etc. would follow the same pattern.
252 case INFIX_TYPE_ENUM:
253 dest_type->meta.enum_info.underlying_type =
254 _copy_type_graph_to_arena(dest_arena, src_type->meta.enum_info.underlying_type);
255 break;
256 case INFIX_TYPE_COMPLEX:
257 dest_type->meta.complex_info.base_type =
258 _copy_type_graph_to_arena(dest_arena, src_type->meta.complex_info.base_type);
259 break;
260 case INFIX_TYPE_VECTOR:
261 dest_type->meta.vector_info.element_type =
262 _copy_type_graph_to_arena(dest_arena, src_type->meta.vector_info.element_type);
263 break;
264 default:
265 // For PRIMITIVE, VOID, NAMED_REFERENCE, etc., the initial shallow copy is sufficient.
266 // For REVERSE_TRAMPOLINE (function pointers), a deep copy of its signature is also needed.
267 // This case would need to be handled if we support introspection of function pointer members.
268 break;
269 }
270 return dest_type;
271}

291static bool _is_type_graph_resolved(infix_type * type) {
292 if (!type)
293 return true;
294 switch (type->category) {
295 case INFIX_TYPE_NAMED_REFERENCE:
296 return false;
297 case INFIX_TYPE_POINTER:
298 return _is_type_graph_resolved(type->meta.pointer_info.pointee_type);
299 case INFIX_TYPE_ARRAY:
300 return _is_type_graph_resolved(type->meta.array_info.element_type);
301 case INFIX_TYPE_STRUCT:
302 case INFIX_TYPE_UNION:
303 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i)
304 if (!_is_type_graph_resolved(type->meta.aggregate_info.members[i].type))
305 return false;
306 return true;
307 case INFIX_TYPE_REVERSE_TRAMPOLINE:
308 if (!_is_type_graph_resolved(type->meta.func_ptr_info.return_type))
309 return false;
310 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i)
311 if (!_is_type_graph_resolved(type->meta.func_ptr_info.args[i].type))
312 return false;
313 return true;
314 default:
315 return true;
316 }
317}

331c23_nodiscard infix_status _infix_forward_create_internal(infix_forward_t ** out_trampoline,
332 infix_type * return_type,
333 infix_type ** arg_types,
334 size_t num_args,
335 size_t num_fixed_args,
336 infix_arena_t * source_arena) {
337 if (out_trampoline == nullptr)
338 return INFIX_ERROR_INVALID_ARGUMENT;
339
340 // Validate the type graphs to ensure they don't contain unresolved placeholders.
341 if (!_is_type_graph_resolved(return_type))
342 return INFIX_ERROR_INVALID_ARGUMENT;
343 for (size_t i = 0; i < num_args; ++i) {
344 if (!_is_type_graph_resolved(arg_types[i]))
345 return INFIX_ERROR_INVALID_ARGUMENT;
346 }
347
348 const infix_forward_abi_spec * spec = get_current_forward_abi_spec();
349 if (spec == nullptr)
350 return INFIX_ERROR_UNSUPPORTED_ABI;
351
352 infix_status status = INFIX_SUCCESS;
353 infix_call_frame_layout * layout = nullptr;
354 infix_forward_t * handle = nullptr;
355 infix_arena_t * arena = nullptr;
356 code_buffer buf;
357
358 // Use a larger arena (64KB) to handle extreme cases (I hope)
359 arena = infix_arena_create(65536);
360 if (!arena)
361 return INFIX_ERROR_ALLOCATION_FAILED;
362
363 code_buffer_init(&buf, arena);
364
365 INFIX_DEBUG_PRINTF("Generating Generic Forward Trampoline");
366
367 status = spec->prepare_forward_call_frame(arena, &layout, return_type, arg_types, num_args, num_fixed_args);
368 if (status != INFIX_SUCCESS)
369 goto cleanup;
370 status = spec->generate_forward_prologue(&buf, layout);
371 if (status != INFIX_SUCCESS)
372 goto cleanup;
373 status = spec->generate_forward_argument_moves(&buf, layout, arg_types, num_args, num_fixed_args);
374 if (status != INFIX_SUCCESS)
375 goto cleanup;
376
377#if defined(INFIX_ARCH_X64)
378 EMIT_BYTES(&buf, 0x4D, 0x85, 0xE4); // test r12, r12 (r12 holds the target function ptr)
379 EMIT_BYTES(&buf, 0x75, 0x02); // jnz +2 (skip next instruction if not null)
380 EMIT_BYTES(&buf, 0x0F, 0x0B); // ud2 (undefined instruction to cause a safe crash)
381 EMIT_BYTES(&buf, 0x41, 0xFF, 0xD4); // call r12
382#elif defined(INFIX_ARCH_AARCH64)
383 EMIT_BYTES(&buf, 0x53, 0x00, 0x00, 0xb5); // cbnz x19, +8 (x19 holds the target)
384 EMIT_BYTES(&buf, 0x00, 0x00, 0x20, 0xd4); // brk #0 (break instruction for safe crash)
385 EMIT_BYTES(&buf, 0x60, 0x02, 0x3f, 0xd6); // blr x19 (branch with link to register)
386#endif
387
388 status = spec->generate_forward_epilogue(&buf, layout, return_type);
389 if (status != INFIX_SUCCESS)
390 goto cleanup;
391
392 // Check for memory allocation errors within the arena or code buffer.
393 if (buf.error || arena->error) {
394 status = INFIX_ERROR_ALLOCATION_FAILED;
395 goto cleanup;
396 }
397
398 handle = infix_calloc(1, sizeof(infix_forward_t));
399 if (handle == nullptr) {
400 status = INFIX_ERROR_ALLOCATION_FAILED;
401 goto cleanup;
402 }
403
404 // --- "Just-Right" Arena Sizing Optimization ---
405 size_t required_size = 0;
406 if (source_arena)
407 required_size = source_arena->current_offset;
408 else
409 required_size = 8192; // Fallback size for manual API calls
410 handle->arena = infix_arena_create(required_size + INFIX_TRAMPOLINE_HEADROOM);
411 if (handle->arena == NULL) {
412 status = INFIX_ERROR_ALLOCATION_FAILED;
413 goto cleanup;
414 }
415
416 // Deep copy all type info into the handle's own arena to make it self-contained.
417 handle->return_type = _copy_type_graph_to_arena(handle->arena, return_type);
418 handle->arg_types = infix_arena_alloc(handle->arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *));
419 if (num_args > 0 && handle->arg_types == nullptr) {
420 status = INFIX_ERROR_ALLOCATION_FAILED;
421 goto cleanup;
422 }
423 for (size_t i = 0; i < num_args; ++i)
424 handle->arg_types[i] = _copy_type_graph_to_arena(handle->arena, arg_types[i]);
425
426 handle->num_args = num_args;
427 handle->num_fixed_args = num_fixed_args;
428
429 handle->exec = infix_executable_alloc(buf.size);
430 if (handle->exec.rw_ptr == nullptr) {
431 status = INFIX_ERROR_ALLOCATION_FAILED;
432 goto cleanup;
433 }
434
435 infix_memcpy(handle->exec.rw_ptr, buf.code, buf.size);
436
437 if (!infix_executable_make_executable(handle->exec)) {
438 status = INFIX_ERROR_PROTECTION_FAILED;
439 goto cleanup;
440 }
441
442 infix_dump_hex(handle->exec.rx_ptr, handle->exec.size, "Forward Trampoline Machine Code");
443 *out_trampoline = handle;
444
445cleanup:
446 // If we failed at any point, ensure the partially created handle is fully destroyed.
447 if (status != INFIX_SUCCESS && handle != nullptr)
448 infix_forward_destroy(handle);
449 infix_arena_destroy(arena);
450 return status;
451}

511c23_nodiscard infix_status infix_reverse_create_manual(infix_reverse_t ** out_context,
512 infix_type * return_type,
513 infix_type ** arg_types,
514 size_t num_args,
515 size_t num_fixed_args,
516 void * user_callback_fn,
517 void * user_data) {
518 if (out_context == nullptr || num_fixed_args > num_args)
519 return INFIX_ERROR_INVALID_ARGUMENT;
520
521 // Validate the user-provided type graphs.
522 if (!_is_type_graph_resolved(return_type))
523 return INFIX_ERROR_INVALID_ARGUMENT;
524 for (size_t i = 0; i < num_args; ++i) {
525 if (!_is_type_graph_resolved(arg_types[i]))
526 return INFIX_ERROR_INVALID_ARGUMENT;
527 }
528
529 const infix_reverse_abi_spec * spec = get_current_reverse_abi_spec();
530 if (spec == nullptr)
531 return INFIX_ERROR_UNSUPPORTED_ABI;
532
533 INFIX_DEBUG_PRINTF("Generating Generic Reverse Trampoline");
534
535 infix_status status = INFIX_SUCCESS;
536 infix_reverse_call_frame_layout * layout = nullptr;
537 infix_reverse_t * context = nullptr;
538 infix_arena_t * arena = nullptr; // Temporary arena for ABI-specific calculations
539 infix_protected_t prot = {.rw_ptr = nullptr, .size = 0};
540 code_buffer buf;
541
542 // Use a temporary arena for layout calculations and code generation.
543 // The final context will have its own persistent arena.
544 arena = infix_arena_create(65536);
545 if (!arena)
546 return INFIX_ERROR_ALLOCATION_FAILED;
547
548 code_buffer_init(&buf, arena);
549
550 size_t page_size = get_page_size();
551 size_t context_alloc_size = (sizeof(infix_reverse_t) + page_size - 1) & ~(page_size - 1);
552 prot = infix_protected_alloc(context_alloc_size);
553 if (prot.rw_ptr == nullptr) {
554 status = INFIX_ERROR_ALLOCATION_FAILED;
555 goto cleanup;
556 }
557 context = (infix_reverse_t *)prot.rw_ptr;
558 infix_memset(context, 0, context_alloc_size);
559
560 // Create a new persistent arena that will be owned by the context.
561 context->arena = infix_arena_create(8192); // 8KB should be enough for most type graphs.
562 if (context->arena == nullptr) {
563 status = INFIX_ERROR_ALLOCATION_FAILED;
564 goto cleanup;
565 }
566
567 context->protected_ctx = prot;
568 context->num_args = num_args;
569 context->num_fixed_args = num_fixed_args;
570 context->is_variadic = (num_fixed_args < num_args);
571 context->user_callback_fn = user_callback_fn;
572 context->user_data = user_data;
573 context->internal_dispatcher = infix_internal_dispatch_callback_fn_impl;
574
575 // Deep copy all type information from the user's arena into the context's own arena.
576 context->return_type = _copy_type_graph_to_arena(context->arena, return_type);
577 context->arg_types = infix_arena_alloc(context->arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *));
578 if (num_args > 0 && context->arg_types == nullptr) {
579 status = INFIX_ERROR_ALLOCATION_FAILED;
580 goto cleanup;
581 }
582 for (size_t i = 0; i < num_args; ++i) {
583 context->arg_types[i] = _copy_type_graph_to_arena(context->arena, arg_types[i]);
584 }
585
586 // The user's callback handler expects the `infix_reverse_t*` context
587 // as its FIRST argument. We must create a new argument type list that reflects this.
588 // The new list will have `1 (for context) + num_args` elements.
589 infix_type ** callback_arg_types =
590 infix_arena_alloc(arena, (1 + num_args) * sizeof(infix_type *), _Alignof(infix_type *));
591 if (callback_arg_types == nullptr) {
592 status = INFIX_ERROR_ALLOCATION_FAILED;
593 goto cleanup;
594 }
595
596 // The first argument is the context pointer.
597 callback_arg_types[0] = infix_type_create_pointer();
598
599 // Now, copy and promote the original user-defined arguments into the rest of the array.
600 if (context->is_variadic) {
601 for (size_t i = 0; i < context->num_args; ++i) {
602 infix_type * current_type = context->arg_types[i];
603 if (i >= num_fixed_args) { // Apply default argument promotions for variadic part.
604 if (is_float(current_type))
605 callback_arg_types[i + 1] = infix_type_create_primitive(INFIX_PRIMITIVE_DOUBLE);
606 else if (current_type->category == INFIX_TYPE_PRIMITIVE && current_type->size < sizeof(int))
607 callback_arg_types[i + 1] = infix_type_create_primitive(INFIX_PRIMITIVE_SINT32);
608 else
609 callback_arg_types[i + 1] = current_type;
610 }
611 else
612 callback_arg_types[i + 1] = current_type;
613 }
614 }
615 else {
616 // For non-variadic functions, just copy the original types.
617 if (context->num_args > 0)
618 infix_memcpy(&callback_arg_types[1], context->arg_types, context->num_args * sizeof(infix_type *));
619 }
620
621 // Generate the cached forward trampoline with the *correct, prepended* argument list.
622 status = infix_forward_create_manual(&context->cached_forward_trampoline,
623 context->return_type,
624 callback_arg_types,
625 context->num_args + 1,
626 context->num_fixed_args + 1);
627
628 if (status != INFIX_SUCCESS)
629 goto cleanup;
630
631 status = spec->prepare_reverse_call_frame(arena, &layout, context);
632 if (status != INFIX_SUCCESS)
633 goto cleanup;
634
635 status = spec->generate_reverse_prologue(&buf, layout);
636 if (status != INFIX_SUCCESS)
637 goto cleanup;
638
639 status = spec->generate_reverse_argument_marshalling(&buf, layout, context);
640 if (status != INFIX_SUCCESS)
641 goto cleanup;
642
643 status = spec->generate_reverse_dispatcher_call(&buf, layout, context);
644 if (status != INFIX_SUCCESS)
645 goto cleanup;
646
647 status = spec->generate_reverse_epilogue(&buf, layout, context);
648 if (status != INFIX_SUCCESS)
649 goto cleanup;
650
651 if (buf.error || arena->error) {
652 status = INFIX_ERROR_ALLOCATION_FAILED;
653 goto cleanup;
654 }
655
656 context->exec = infix_executable_alloc(buf.size);
657 if (context->exec.rw_ptr == nullptr) {
658 status = INFIX_ERROR_ALLOCATION_FAILED;
659 goto cleanup;
660 }
661
662 infix_memcpy(context->exec.rw_ptr, buf.code, buf.size);
663 size_t code_size = buf.size;
664
665 if (!infix_executable_make_executable(context->exec)) {
666 status = INFIX_ERROR_PROTECTION_FAILED;
667 goto cleanup;
668 }
669
670 if (!infix_protected_make_readonly(context->protected_ctx)) {
671 status = INFIX_ERROR_PROTECTION_FAILED;
672 goto cleanup;
673 }
674
675 INFIX_DEBUG_PRINTF("Hardened reverse trampoline context at %p to be read-only.", (void *)context);
676 infix_dump_hex(context->exec.rx_ptr, code_size, "Reverse Trampoline Machine Code");
677
678 *out_context = context;
679
680cleanup:
681 if (status != INFIX_SUCCESS && context != nullptr)
682 // If we failed, we must free the context and its nested resources.
683 // infix_reverse_destroy handles this correctly.
684 infix_reverse_destroy(context);
685
686 infix_arena_destroy(arena);
687 return status;
688}
