116void code_buffer_init(code_buffer * buf, infix_arena_t * arena) {

132void code_buffer_append(code_buffer * buf, const void * data, size_t len) {
133 if (buf->error) // If we're already in an error state, do nothing.
134 return;
135
136 if (len > SIZE_MAX - buf->size) {
137 //~ fprintf(stderr,
138 //~ "Error: Appending %llu bytes to buffer of size %llu would cause overflow.\n",
139 //~ (unsigned long long)len,
140 //~ (unsigned long long)buf->size);
141 buf->error = true;
142 return;
143 }
144
145 if (buf->size + len > buf->capacity) {
146 size_t new_capacity = buf->capacity;
147 while (new_capacity < buf->size + len) {
148 if (new_capacity > SIZE_MAX / 2) {
149 //~ fprintf(stderr, "Error: code_buffer cannot grow further without overflow.\n");
150 buf->error = true;
151 return;
152 }
153 new_capacity *= 2;
154 }
155
156 // Arena allocators don't support realloc. We must allocate a new block and copy.
157 void * new_code = infix_arena_alloc(buf->arena, new_capacity, 16);
158 if (new_code == nullptr) {
159 //~ fprintf(stderr, "Error: infix_arena_alloc failed in code_buffer_append.\n");
160 buf->error = true;
161 return;
162 }
163 // Copy the old code to the new, larger buffer.
164 infix_memcpy(new_code, buf->code, buf->size);
165 buf->code = new_code;
166 buf->capacity = new_capacity;
167 }
168 infix_memcpy(buf->code + buf->size, data, len);
169 buf->size += len;
170}

217static bool _is_type_graph_resolved(infix_type * type) {
218 if (!type)
219 return true; // Null is fine.
220
221 switch (type->category) {
222 case INFIX_TYPE_NAMED_REFERENCE:
223 return false; // Found an unresolved reference!
224
225 case INFIX_TYPE_POINTER:
226 return _is_type_graph_resolved(type->meta.pointer_info.pointee_type);
227
228 case INFIX_TYPE_ARRAY:
229 return _is_type_graph_resolved(type->meta.array_info.element_type);
230
231 case INFIX_TYPE_STRUCT:
232 case INFIX_TYPE_UNION:
233 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
234 if (!_is_type_graph_resolved(type->meta.aggregate_info.members[i].type))
235 return false;
236 }
237 return true;
238
239 case INFIX_TYPE_REVERSE_TRAMPOLINE:
240 if (!_is_type_graph_resolved(type->meta.func_ptr_info.return_type))
241 return false;
242 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i) {
243 if (!_is_type_graph_resolved(type->meta.func_ptr_info.args[i].type))
244 return false;
245 }
246 return true;
247
248 // Primitives, Void, etc., are always resolved.
249 default:
250 return true;
251 }
252}

274c23_nodiscard infix_status infix_forward_create_manual(infix_forward_t ** out_trampoline,
275 infix_type * return_type,
276 infix_type ** arg_types,
277 size_t num_args,
278 size_t num_fixed_args) {
279 if (out_trampoline == nullptr)
280 return INFIX_ERROR_INVALID_ARGUMENT;
281
282 // Validate the type graphs
283 if (!_is_type_graph_resolved(return_type))
284 return INFIX_ERROR_INVALID_ARGUMENT;
285 for (size_t i = 0; i < num_args; ++i) {
286 if (!_is_type_graph_resolved(arg_types[i]))
287 return INFIX_ERROR_INVALID_ARGUMENT;
288 }
289
290 const infix_forward_abi_spec * spec = get_current_forward_abi_spec();
291 if (spec == nullptr)
292 return INFIX_ERROR_UNSUPPORTED_ABI;
293
294 infix_status status = INFIX_SUCCESS;
295 infix_call_frame_layout * layout = nullptr;
296 infix_forward_t * handle = nullptr;
297 infix_arena_t * arena = nullptr;
298 code_buffer buf;
299
300 // Use a larger arena (64KB) to handle extreme cases (I hope)
301 arena = infix_arena_create(65536);
302 if (!arena)
303 return INFIX_ERROR_ALLOCATION_FAILED;
304
305 code_buffer_init(&buf, arena);
306
307 INFIX_DEBUG_PRINTF("Generating Generic Forward Trampoline");
308
309 status = spec->prepare_forward_call_frame(arena, &layout, return_type, arg_types, num_args, num_fixed_args);
310 if (status != INFIX_SUCCESS)
311 goto cleanup;
312
313 status = spec->generate_forward_prologue(&buf, layout);
314 if (status != INFIX_SUCCESS)
315 goto cleanup;
316
317 status = spec->generate_forward_argument_moves(&buf, layout, arg_types, num_args, num_fixed_args);
318 if (status != INFIX_SUCCESS)
319 goto cleanup;
320
321#if defined(INFIX_ARCH_X64)
322 EMIT_BYTES(&buf, 0x4D, 0x85, 0xE4); // test r12, r12 (r12 holds the target function ptr)
323 EMIT_BYTES(&buf, 0x75, 0x02); // jnz +2 (skip next instruction if not null)
324 EMIT_BYTES(&buf, 0x0F, 0x0B); // ud2 (undefined instruction to cause a safe crash)
325 EMIT_BYTES(&buf, 0x41, 0xFF, 0xD4); // call r12
326#elif defined(INFIX_ARCH_AARCH64)
327 EMIT_BYTES(&buf, 0x53, 0x00, 0x00, 0xb5); // cbnz x19, +8 (x19 holds the target)
328 EMIT_BYTES(&buf, 0x00, 0x00, 0x20, 0xd4); // brk #0 (break instruction for safe crash)
329 EMIT_BYTES(&buf, 0x60, 0x02, 0x3f, 0xd6); // blr x19 (branch with link to register)
330#endif
331
332 status = spec->generate_forward_epilogue(&buf, layout, return_type);
333 if (status != INFIX_SUCCESS)
334 goto cleanup;
335
336 // Check for memory allocation errors within the arena or code buffer.
337 if (buf.error || arena->error) {
338 status = INFIX_ERROR_ALLOCATION_FAILED;
339 goto cleanup;
340 }
341
342 handle = infix_calloc(1, sizeof(infix_forward_t));
343 if (handle == nullptr) {
344 status = INFIX_ERROR_ALLOCATION_FAILED;
345 goto cleanup;
346 }
347
348 handle->exec = infix_executable_alloc(buf.size);
349 if (handle->exec.rw_ptr == nullptr) {
350 status = INFIX_ERROR_ALLOCATION_FAILED;
351 infix_free(handle); // Manually free handle as infix_forward_destroy won't be called
352 handle = nullptr;
353 goto cleanup;
354 }
355
356 infix_memcpy(handle->exec.rw_ptr, buf.code, buf.size);
357
358 if (!infix_executable_make_executable(handle->exec)) {
359 infix_executable_free(handle->exec);
360 infix_free(handle);
361 handle = nullptr;
362 status = INFIX_ERROR_PROTECTION_FAILED;
363 goto cleanup;
364 }
365
366 infix_dump_hex(handle->exec.rx_ptr, handle->exec.size, "Forward Trampoline Machine Code");
367 *out_trampoline = handle;
368
369cleanup:
370 // All temporary allocations (layout, code buffer) are cleaned up at once.
371 infix_arena_destroy(arena);
372 return status;
373}

417c23_nodiscard infix_status infix_reverse_create_manual(infix_reverse_t ** out_context,
418 infix_type * return_type,
419 infix_type ** arg_types,
420 size_t num_args,
421 size_t num_fixed_args,
422 void * user_callback_fn,
423 void * user_data) {
424 if (out_context == nullptr || num_fixed_args > num_args)
425 return INFIX_ERROR_INVALID_ARGUMENT;
426
427 // Validate the graph
428 if (!_is_type_graph_resolved(return_type))
429 return INFIX_ERROR_INVALID_ARGUMENT;
430 for (size_t i = 0; i < num_args; ++i) {
431 if (!_is_type_graph_resolved(arg_types[i]))
432 return INFIX_ERROR_INVALID_ARGUMENT;
433 }
434
435 const infix_reverse_abi_spec * spec = get_current_reverse_abi_spec();
436 if (spec == nullptr)
437 return INFIX_ERROR_UNSUPPORTED_ABI;
438
439 INFIX_DEBUG_PRINTF("Generating Generic Reverse Trampoline");
440
441 infix_status status = INFIX_SUCCESS;
442 infix_reverse_call_frame_layout * layout = nullptr;
443 infix_reverse_t * context = nullptr;
444 infix_protected_t prot = {.rw_ptr = nullptr, .size = 0};
445 infix_arena_t * arena = nullptr;
446 code_buffer buf;
447
448 // Use a larger arena (64KB) to handle extreme cases (I hope)
449 arena = infix_arena_create(65536);
450 if (!arena)
451 return INFIX_ERROR_ALLOCATION_FAILED;
452
453 code_buffer_init(&buf, arena);
454
455 size_t page_size = get_page_size();
456 size_t context_alloc_size = (sizeof(infix_reverse_t) + page_size - 1) & ~(page_size - 1);
457 prot = infix_protected_alloc(context_alloc_size);
458 if (prot.rw_ptr == nullptr) {
459 status = INFIX_ERROR_ALLOCATION_FAILED;
460 goto cleanup;
461 }
462
463 context = (infix_reverse_t *)prot.rw_ptr;
464 infix_memset(context, 0, context_alloc_size);
465 context->protected_ctx = prot;
466 context->return_type = return_type;
467 context->arg_types = arg_types;
468 context->num_args = num_args;
469 context->num_fixed_args = num_fixed_args;
470 context->is_variadic = (num_fixed_args < num_args);
471 context->user_callback_fn = user_callback_fn;
472 context->user_data = user_data;
473 context->internal_dispatcher = infix_internal_dispatch_callback_fn_impl;
474
475 // The user's callback handler expects the `infix_reverse_t*` context
476 // as its FIRST argument. We must create a new argument type list that reflects this.
477 // The new list will have `1 (for context) + num_args` elements.
478 infix_type ** callback_arg_types =
479 infix_arena_alloc(arena, (1 + num_args) * sizeof(infix_type *), _Alignof(infix_type *));
480 if (callback_arg_types == nullptr) {
481 status = INFIX_ERROR_ALLOCATION_FAILED;
482 goto cleanup;
483 }
484
485 // The first argument is the context pointer.
486 callback_arg_types[0] = infix_type_create_pointer();
487
488 // Now, copy and promote the original user-defined arguments into the rest of the array.
489 if (context->is_variadic) {
490 for (size_t i = 0; i < num_args; ++i) {
491 infix_type * current_type = arg_types[i];
492 if (i >= num_fixed_args) { // Apply default argument promotions for variadic part.
493 if (is_float(current_type))
494 callback_arg_types[i + 1] = infix_type_create_primitive(INFIX_PRIMITIVE_DOUBLE);
495 else if (current_type->category == INFIX_TYPE_PRIMITIVE && current_type->size < sizeof(int))
496 callback_arg_types[i + 1] = infix_type_create_primitive(INFIX_PRIMITIVE_SINT32);
497 else
498 callback_arg_types[i + 1] = current_type;
499 }
500 else
501 callback_arg_types[i + 1] = current_type;
502 }
503 }
504 else {
505 // For non-variadic functions, just copy the original types.
506 if (num_args > 0)
507 infix_memcpy(&callback_arg_types[1], arg_types, num_args * sizeof(infix_type *));
508 }
509
510 // Generate the cached forward trampoline with the *correct, prepended* argument list.
511 // The total number of arguments and fixed arguments are each increased by one.
512 status = infix_forward_create_manual(
513 &context->cached_forward_trampoline, return_type, callback_arg_types, num_args + 1, num_fixed_args + 1);
514
515 if (status != INFIX_SUCCESS)
516 goto cleanup;
517
518 status = spec->prepare_reverse_call_frame(arena, &layout, context);
519 if (status != INFIX_SUCCESS)
520 goto cleanup;
521
522 status = spec->generate_reverse_prologue(&buf, layout);
523 if (status != INFIX_SUCCESS)
524 goto cleanup;
525
526 status = spec->generate_reverse_argument_marshalling(&buf, layout, context);
527 if (status != INFIX_SUCCESS)
528 goto cleanup;
529
530 status = spec->generate_reverse_dispatcher_call(&buf, layout, context);
531 if (status != INFIX_SUCCESS)
532 goto cleanup;
533
534 status = spec->generate_reverse_epilogue(&buf, layout, context);
535 if (status != INFIX_SUCCESS)
536 goto cleanup;
537
538 if (buf.error || arena->error) {
539 status = INFIX_ERROR_ALLOCATION_FAILED;
540 goto cleanup;
541 }
542
543 context->exec = infix_executable_alloc(buf.size);
544 if (context->exec.rw_ptr == nullptr) {
545 status = INFIX_ERROR_ALLOCATION_FAILED;
546 goto cleanup;
547 }
548
549 infix_memcpy(context->exec.rw_ptr, buf.code, buf.size);
550 size_t code_size = buf.size;
551
552 if (!infix_executable_make_executable(context->exec)) {
553 status = INFIX_ERROR_PROTECTION_FAILED;
554 goto cleanup;
555 }
556
557 if (!infix_protected_make_readonly(context->protected_ctx)) {
558 status = INFIX_ERROR_PROTECTION_FAILED;
559 goto cleanup;
560 }
561
562 INFIX_DEBUG_PRINTF("Hardened reverse trampoline context at %p to be read-only.", (void *)context);
563 infix_dump_hex(context->exec.rx_ptr, code_size, "Reverse Trampoline Machine Code");
564
565 *out_context = context;
566
567cleanup:
568 if (status != INFIX_SUCCESS && context != nullptr)
569 // If we failed, we must free the context and its nested resources.
570 // infix_reverse_destroy handles this correctly.
571 infix_reverse_destroy(context);
572
573 infix_arena_destroy(arena);
574 return status;
575}
