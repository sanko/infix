116void code_buffer_init(code_buffer * buf, infix_arena_t * arena) {

118 buf->arena = arena;
119 buf->code = infix_arena_alloc(arena, buf->capacity, 16); // Align to 16 bytes for safety

132void code_buffer_append(code_buffer * buf, const void * data, size_t len) {
133 if (buf->error) // If we're already in an error state, do nothing.
134 return;
135
136 if (len > SIZE_MAX - buf->size) {
137 //~ fprintf(stderr,
138 //~ "Error: Appending %llu bytes to buffer of size %llu would cause overflow.\n",
139 //~ (unsigned long long)len,
140 //~ (unsigned long long)buf->size);
141 buf->error = true;
142 return;
143 }
144
145 if (buf->size + len > buf->capacity) {
146 size_t new_capacity = buf->capacity;
147 while (new_capacity < buf->size + len) {
148 if (new_capacity > SIZE_MAX / 2) {
149 //~ fprintf(stderr, "Error: code_buffer cannot grow further without overflow.\n");
150 buf->error = true;
151 return;
152 }
153 new_capacity *= 2;
154 }
155
156 // Arena allocators don't support realloc. We must allocate a new block and copy.
157 void * new_code = infix_arena_alloc(buf->arena, new_capacity, 16);
158 if (new_code == NULL) {
159 //~ fprintf(stderr, "Error: infix_arena_alloc failed in code_buffer_append.\n");
160 buf->error = true;
161 return;
162 }
163 // Copy the old code to the new, larger buffer.
164 infix_memcpy(new_code, buf->code, buf->size);
165 buf->code = new_code;
166 buf->capacity = new_capacity;
167 }
168 infix_memcpy(buf->code + buf->size, data, len);
169 buf->size += len;
170}

230c23_nodiscard infix_status infix_forward_create_manual(infix_forward_t ** out_trampoline,
231 infix_type * return_type,
232 infix_type ** arg_types,
233 size_t num_args,
234 size_t num_fixed_args) {
235 if (out_trampoline == nullptr)
236 return INFIX_ERROR_INVALID_ARGUMENT;
237
238 const infix_forward_abi_spec * spec = get_current_forward_abi_spec();
239 if (spec == nullptr)
240 return INFIX_ERROR_UNSUPPORTED_ABI;
241
242 infix_status status = INFIX_SUCCESS;
243 infix_call_frame_layout * layout = nullptr;
244 infix_forward_t * handle = nullptr;
245 infix_arena_t * arena = nullptr;
246 code_buffer buf;
247
248 // Use a larger arena (64KB) to handle extreme cases (I hope)
249 arena = infix_arena_create(65536);
250 if (!arena)
251 return INFIX_ERROR_ALLOCATION_FAILED;
252
253 code_buffer_init(&buf, arena);
254
255 INFIX_DEBUG_PRINTF("Generating Generic Forward Trampoline");
256
257 status = spec->prepare_forward_call_frame(arena, &layout, return_type, arg_types, num_args, num_fixed_args);
258 if (status != INFIX_SUCCESS)
259 goto cleanup;
260
261 status = spec->generate_forward_prologue(&buf, layout);
262 if (status != INFIX_SUCCESS)
263 goto cleanup;
264
265 status = spec->generate_forward_argument_moves(&buf, layout, arg_types, num_args, num_fixed_args);
266 if (status != INFIX_SUCCESS)
267 goto cleanup;
268
269#if defined(INFIX_ARCH_X64)
270 EMIT_BYTES(&buf, 0x4D, 0x85, 0xE4); // test r12, r12 (r12 holds the target function ptr)
271 EMIT_BYTES(&buf, 0x75, 0x02); // jnz +2 (skip next instruction if not null)
272 EMIT_BYTES(&buf, 0x0F, 0x0B); // ud2 (undefined instruction to cause a safe crash)
273 EMIT_BYTES(&buf, 0x41, 0xFF, 0xD4); // call r12
274#elif defined(INFIX_ARCH_AARCH64)
275 EMIT_BYTES(&buf, 0x53, 0x00, 0x00, 0xb5); // cbnz x19, +8 (x19 holds the target)
276 EMIT_BYTES(&buf, 0x00, 0x00, 0x20, 0xd4); // brk #0 (break instruction for safe crash)
277 EMIT_BYTES(&buf, 0x60, 0x02, 0x3f, 0xd6); // blr x19 (branch with link to register)
278#endif
279
280 status = spec->generate_forward_epilogue(&buf, layout, return_type);
281 if (status != INFIX_SUCCESS)
282 goto cleanup;
283
284 // Check for memory allocation errors within the arena or code buffer.
285 if (buf.error || arena->error) {
286 status = INFIX_ERROR_ALLOCATION_FAILED;
287 goto cleanup;
288 }
289
290 handle = infix_calloc(1, sizeof(infix_forward_t));
291 if (handle == nullptr) {
292 status = INFIX_ERROR_ALLOCATION_FAILED;
293 goto cleanup;
294 }
295
296 handle->exec = infix_executable_alloc(buf.size);
297 if (handle->exec.rw_ptr == nullptr) {
298 status = INFIX_ERROR_ALLOCATION_FAILED;
299 infix_free(handle); // Manually free handle as infix_forward_destroy won't be called
300 handle = nullptr;
301 goto cleanup;
302 }
303
304 infix_memcpy(handle->exec.rw_ptr, buf.code, buf.size);
305
306 if (!infix_executable_make_executable(handle->exec)) {
307 infix_executable_free(handle->exec);
308 infix_free(handle);
309 handle = nullptr;
310 status = INFIX_ERROR_PROTECTION_FAILED;
311 goto cleanup;
312 }
313
314 infix_dump_hex(handle->exec.rx_ptr, handle->exec.size, "Forward Trampoline Machine Code");
315 *out_trampoline = handle;
316
317cleanup:
318 // All temporary allocations (layout, code buffer) are cleaned up at once.
319 infix_arena_destroy(arena);
320 return status;
321}

365c23_nodiscard infix_status infix_reverse_create_manual(infix_reverse_t ** out_context,
366 infix_type * return_type,
367 infix_type ** arg_types,
368 size_t num_args,
369 size_t num_fixed_args,
370 void * user_callback_fn,
371 void * user_data) {
372 if (out_context == nullptr || num_fixed_args > num_args)
373 return INFIX_ERROR_INVALID_ARGUMENT;
374
375 const infix_reverse_abi_spec * spec = get_current_reverse_abi_spec();
376 if (spec == nullptr)
377 return INFIX_ERROR_UNSUPPORTED_ABI;
378
379 INFIX_DEBUG_PRINTF("Generating Generic Reverse Trampoline");
380
381 infix_status status = INFIX_SUCCESS;
382 infix_reverse_call_frame_layout * layout = nullptr;
383 infix_reverse_t * context = nullptr;
384 infix_protected_t prot = {.rw_ptr = nullptr, .size = 0};
385 infix_arena_t * arena = nullptr;
386 code_buffer buf;
387
388 // Use a larger arena (64KB) to handle extreme cases (I hope)
389 arena = infix_arena_create(65536);
390 if (!arena)
391 return INFIX_ERROR_ALLOCATION_FAILED;
392
393 code_buffer_init(&buf, arena);
394
395 size_t page_size = get_page_size();
396 size_t context_alloc_size = (sizeof(infix_reverse_t) + page_size - 1) & ~(page_size - 1);
397 prot = infix_protected_alloc(context_alloc_size);
398 if (prot.rw_ptr == nullptr) {
399 status = INFIX_ERROR_ALLOCATION_FAILED;
400 goto cleanup;
401 }
402
403 context = (infix_reverse_t *)prot.rw_ptr;
404 infix_memset(context, 0, context_alloc_size);
405 context->protected_ctx = prot;
406 context->return_type = return_type;
407 context->arg_types = arg_types;
408 context->num_args = num_args;
409 context->num_fixed_args = num_fixed_args;
410 context->is_variadic = (num_fixed_args < num_args);
411 context->user_callback_fn = user_callback_fn;
412 context->user_data = user_data;
413 context->internal_dispatcher = infix_internal_dispatch_callback_fn_impl;
414
415 // The user's callback handler expects the `infix_reverse_t*` context
416 // as its FIRST argument. We must create a new argument type list that reflects this.
417 // The new list will have `1 (for context) + num_args` elements.
418 infix_type ** callback_arg_types =
419 infix_arena_alloc(arena, (1 + num_args) * sizeof(infix_type *), _Alignof(infix_type *));
420 if (callback_arg_types == NULL) {
421 status = INFIX_ERROR_ALLOCATION_FAILED;
422 goto cleanup;
423 }
424
425 // The first argument is the context pointer.
426 callback_arg_types[0] = infix_type_create_pointer();
427
428 // Now, copy and promote the original user-defined arguments into the rest of the array.
429 if (context->is_variadic) {
430 for (size_t i = 0; i < num_args; ++i) {
431 infix_type * current_type = arg_types[i];
432 if (i >= num_fixed_args) { // Apply default argument promotions for variadic part.
433 if (is_float(current_type))
434 callback_arg_types[i + 1] = infix_type_create_primitive(INFIX_PRIMITIVE_DOUBLE);
435 else if (current_type->category == INFIX_TYPE_PRIMITIVE && current_type->size < sizeof(int))
436 callback_arg_types[i + 1] = infix_type_create_primitive(INFIX_PRIMITIVE_SINT32);
437 else
438 callback_arg_types[i + 1] = current_type;
439 }
440 else
441 callback_arg_types[i + 1] = current_type;
442 }
443 }
444 else {
445 // For non-variadic functions, just copy the original types.
446 if (num_args > 0)
447 infix_memcpy(&callback_arg_types[1], arg_types, num_args * sizeof(infix_type *));
448 }
449
450 // Generate the cached forward trampoline with the *correct, prepended* argument list.
451 // The total number of arguments and fixed arguments are each increased by one.
452 status = infix_forward_create_manual(
453 &context->cached_forward_trampoline, return_type, callback_arg_types, num_args + 1, num_fixed_args + 1);
454
455 if (status != INFIX_SUCCESS)
456 goto cleanup;
457
458 status = spec->prepare_reverse_call_frame(arena, &layout, context);
459 if (status != INFIX_SUCCESS)
460 goto cleanup;
461
462 status = spec->generate_reverse_prologue(&buf, layout);
463 if (status != INFIX_SUCCESS)
464 goto cleanup;
465
466 status = spec->generate_reverse_argument_marshalling(&buf, layout, context);
467 if (status != INFIX_SUCCESS)
468 goto cleanup;
469
470 status = spec->generate_reverse_dispatcher_call(&buf, layout, context);
471 if (status != INFIX_SUCCESS)
472 goto cleanup;
473
474 status = spec->generate_reverse_epilogue(&buf, layout, context);
475 if (status != INFIX_SUCCESS)
476 goto cleanup;
477
478 if (buf.error || arena->error) {
479 status = INFIX_ERROR_ALLOCATION_FAILED;
480 goto cleanup;
481 }
482
483 context->exec = infix_executable_alloc(buf.size);
484 if (context->exec.rw_ptr == nullptr) {
485 status = INFIX_ERROR_ALLOCATION_FAILED;
486 goto cleanup;
487 }
488
489 infix_memcpy(context->exec.rw_ptr, buf.code, buf.size);
490 size_t code_size = buf.size;
491
492 if (!infix_executable_make_executable(context->exec)) {
493 status = INFIX_ERROR_PROTECTION_FAILED;
494 goto cleanup;
495 }
496
497 if (!infix_protected_make_readonly(context->protected_ctx)) {
498 status = INFIX_ERROR_PROTECTION_FAILED;
499 goto cleanup;
500 }
501
502 INFIX_DEBUG_PRINTF("Hardened reverse trampoline context at %p to be read-only.", (void *)context);
503 infix_dump_hex(context->exec.rx_ptr, code_size, "Reverse Trampoline Machine Code");
504
505 *out_context = context;
506
507cleanup:
508 if (status != INFIX_SUCCESS && context != nullptr)
509 // If we failed, we must free the context and its nested resources.
510 // infix_reverse_destroy handles this correctly.
511 infix_reverse_destroy(context);
512
513 infix_arena_destroy(arena);
514 return status;
515}
