204typedef struct {

206 size_t count;
207 size_t failed;

218// Thread-Safe State Management
219
220// Each thread gets its own private state stack.
221static TAP_THREAD_LOCAL tap_state_t state_stack[MAX_DEPTH];
222static TAP_THREAD_LOCAL tap_state_t * current_state = NULL;
223
224// A global counter for the final exit code, updated atomically.
225static TAP_ATOMIC_SIZE_T g_total_failed = 0;

227// One-time initialization for global setup (e.g., printing TAP header).
228#if defined(_WIN32) || defined(__CYGWIN__)
229static INIT_ONCE g_tap_init_once = INIT_ONCE_STATIC_INIT;
230static BOOL CALLBACK _tap_init_routine(PINIT_ONCE initOnce, PVOID param, PVOID * context) {
231 (void)initOnce;
232 (void)param;
233 (void)context;
234 printf("TAP version %d\n", TAP_VERSION);
235 fflush(stdout);
236 return TRUE;
237}
238#elif defined(__unix__) || defined(__APPLE__)
239static pthread_once_t g_tap_init_once = PTHREAD_ONCE_INIT;
240static void _tap_init_routine(void) {
241 printf("TAP version %d\n", TAP_VERSION);

246// This function must be called at the start of every public API function.
247static void _tap_ensure_initialized(void) {
248#if defined(_WIN32) || defined(__CYGWIN__)
249 InitOnceExecuteOnce(&g_tap_init_once, _tap_init_routine, NULL, NULL);
250#elif defined(__unix__) || defined(__APPLE__)
251 pthread_once(&g_tap_init_once, _tap_init_routine);
252#endif
253 if (!current_state) {
254 current_state = &state_stack[0];
255 memset(current_state, 0, sizeof(tap_state_t));
256 current_state->plan = NO_PLAN;
257 }
258}

269static void push_state(void) {
270 if (current_state >= &state_stack[MAX_DEPTH - 1]) {
271 tap_bail_out("Exceeded maximum subtest depth of %d", MAX_DEPTH);
272 }
273 tap_state_t * parent = current_state;
274 current_state++;
275 memset(current_state, 0, sizeof(tap_state_t));
276 current_state->plan = NO_PLAN;
277 current_state->indent_level = parent->indent_level + 1;
278 current_state->todo = parent->todo;
279 if (parent->todo) {
280 snprintf(current_state->todo_reason, sizeof(current_state->todo_reason), "%s", parent->todo_reason);
281 }
282}

284static void pop_state(void) {
285 if (current_state <= &state_stack[0]) {
286 tap_bail_out("Attempted to pop state from base level");
287 }
288 current_state--;
289}

309bool tap_ok(bool condition, const char * file, int line, const char * func, const char * expr, const char * name, ...) {
310 _tap_ensure_initialized();
311 if (current_state->skipping) {
312 return true;
313 }
314
315 char name_buffer[256] = {0};
316 if (name && name[0] != '\0') {
317 va_list args;
318 va_start(args, name);
319 vsnprintf(name_buffer, sizeof(name_buffer), name, args);
320 va_end(args);
321 }
322
323 current_state->count++;
324
325 if (!condition) {
326 if (current_state->todo) {
327 current_state->failed_todo++;
328 }
329 else {
330 current_state->failed++;
331 TAP_ATOMIC_FETCH_ADD(&g_total_failed, 1);
332 }
333 }
334
335 print_indent(stdout);
336 printf("%s %llu", condition ? "ok" : "not ok", (unsigned long long)current_state->count);
337 if (name_buffer[0] != '\0') {
338 printf(" - %s", name_buffer);
339 }
340 if (current_state->todo) {
341 printf(" # TODO %s", current_state->todo_reason);
342 }
343 printf("\n");
344
345 if (!condition && !current_state->todo) {
346 print_indent(stdout);
347 fprintf(stdout, "#\n");
348 print_indent(stdout);
349 fprintf(stdout, "# message: 'Test failed'\n");
350 print_indent(stdout);
351 fprintf(stdout, "# severity: fail\n");
352 print_indent(stdout);
353 fprintf(stdout, "# data:\n");
354 print_indent(stdout);
355 fprintf(stdout, "# file: %s\n", file);
356 print_indent(stdout);
357 fprintf(stdout, "# line: %d\n", line);
358 print_indent(stdout);
359 fprintf(stdout, "# function: %s\n", func);
360 print_indent(stdout);
361 fprintf(stdout, "# expression: '%s'\n", expr);
362 print_indent(stdout);
363 fprintf(stdout, "# ...\n");
364 }
365 fflush(stdout);
366 return condition;
367}

447bool tap_subtest_start(const char * name) {
448 _tap_ensure_initialized();
449 print_indent(stdout);
450 fprintf(stdout, "# Subtest: %s\n", name);
451 fflush(stdout);

457bool tap_subtest_end(void) {
458 _tap_ensure_initialized();
459
460 if (!current_state->has_plan) {
461 current_state->plan = current_state->count;
462 print_indent(stdout);
463 printf("1..%llu\n", (unsigned long long)current_state->plan);
464 }
465 bool plan_ok = (current_state->plan == current_state->count);
466 bool subtest_ok = (current_state->failed == 0) && plan_ok;
467
468 char name_buffer[256];
469 snprintf(name_buffer, sizeof(name_buffer), "%s", current_state->subtest_name);
470
471 pop_state(); // Return to parent's context.
472
473 ok(subtest_ok, "%s", name_buffer);
474
475 return false; // Ensures the subtest for-loop only runs once.
476}

478int tap_done(void) {
479 _tap_ensure_initialized();
480 if (current_state != &state_stack[0]) {
481 return (int)TAP_ATOMIC_FETCH_ADD(&g_total_failed, 0);
482 }
483
484 if (!current_state->has_plan) {
485 current_state->plan = current_state->count;
486 print_indent(stdout);
487 printf("1..%llu\n", (unsigned long long)current_state->plan);
488 fflush(stdout);
489 }
490
491 if (current_state->skipping) {
492 print_indent(stdout);
493 printf("1..%llu # SKIP %s\n", (unsigned long long)current_state->plan, current_state->skip_reason);
494 fflush(stdout);
495 return 0;
496 }
497
498 if (current_state->plan != current_state->count) {
499 fail("Test plan adherence (planned %llu, but ran %llu)",
500 (unsigned long long)current_state->plan,
501 (unsigned long long)current_state->count);
502 }
503
504 size_t final_failed_count = (size_t)TAP_ATOMIC_FETCH_ADD(&g_total_failed, 0);
505 diag("Tests finished. Failed %llu of %llu planned tests.",
506 (unsigned long long)final_failed_count,
507 (unsigned long long)current_state->plan);
508
509 return (int)final_failed_count;
510}
