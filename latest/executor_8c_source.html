88static int shm_open_anonymous() {
89 char shm_name[64];
90 uint64_t random_val = 0;
91
92 // Open /dev/urandom to get cryptographically secure random bytes.
93 int rand_fd = open("/dev/urandom", O_RDONLY);
94 if (rand_fd < 0)
95 return -1;
96
97 // Read 8 bytes (64 bits) of random data.
98 ssize_t bytes_read = read(rand_fd, &random_val, sizeof(random_val));
99 close(rand_fd);
100 if (bytes_read != sizeof(random_val))
101 return -1;
102 // Create a highly unpredictable name to minimize collision risk.
103 snprintf(shm_name, sizeof(shm_name), "/infix-jit-%d-%llx", getpid(), (unsigned long long)random_val);
104 // O_EXCL is a critical security flag. It ensures that shm_open will fail if
105 // a file with this name already exists, preventing symlink attacks or races.
106 int fd = shm_open(shm_name, O_RDWR | O_CREAT | O_EXCL, 0600);
107 if (fd >= 0) {
108 // We've successfully created the object. Now, unlink the name immediately.
109 // The file descriptor remains valid, but the name is gone from the filesystem,
110 // making it truly anonymous and ensuring automatic cleanup.
111 shm_unlink(shm_name);
112 return fd;
113 }
114
115 // If shm_open failed, it's likely a more serious system issue, as a
116 // name collision is now astronomically unlikely.
117 return -1;
118}
119#endif

143c23_nodiscard infix_executable_t infix_executable_alloc(size_t size) {
144 infix_executable_t exec = {.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0};
145 if (size == 0) {
146 return exec;
147 }
148
149#if defined(INFIX_OS_WINDOWS)
150 // Windows: Single Mapping (VirtualAlloc).
151 void * code = VirtualAlloc(nullptr, size, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
152 if (code == nullptr)
153 return exec;
154 // In a single-mapping model, both pointers initially point to the same RW address.
155 exec.rw_ptr = code;
156 exec.rx_ptr = code;
157
158#elif defined(INFIX_OS_MACOS) || defined(INFIX_OS_TERMUX) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
159 // Single Mapping (mmap). This simpler approach is more reliable on these platforms.
160 // - macOS/OpenBSD: Avoids issues with shm + fork.
161 // - termux: https://github.com/termux/libandroid-shmem/issues/10
162 void * code = MAP_FAILED;
163#if defined(MAP_ANON)
164 int flags = MAP_PRIVATE | MAP_ANON;
165#if defined(INFIX_OS_MACOS)
166 flags |= MAP_JIT;
167#endif
168 code = mmap(nullptr, size, PROT_READ | PROT_WRITE, flags, -1, 0);
169#else
170 // Fallback for systems like DragonFly BSD that may not define MAP_ANON.
171 int fd = open("/dev/zero", O_RDWR);
172 if (fd != -1) {
173 code = mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_PRIVATE, fd, 0);
174 close(fd);
175 }
176#endif
177 if (code == MAP_FAILED)
178 return exec;
179
180 // Both pointers point to the same memory region.
181 exec.rw_ptr = code;
182 exec.rx_ptr = code;
183
184#else
185 // Other POSIX (Linux, BSD): Dual Mapping (shm_open) for strict W^X.
186 exec.shm_fd = shm_open_anonymous();
187 if (exec.shm_fd < 0)
188 return exec;
189
190 if (ftruncate(exec.shm_fd, size) != 0) {
191 close(exec.shm_fd);
192 return exec;
193 }
194 // Create two separate virtual memory mappings to the same underlying shared memory object.
195 exec.rw_ptr = mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, exec.shm_fd, 0);
196 exec.rx_ptr = mmap(nullptr, size, PROT_READ | PROT_EXEC, MAP_SHARED, exec.shm_fd, 0);
197 if (exec.rw_ptr == MAP_FAILED || exec.rx_ptr == MAP_FAILED) {
198 if (exec.rw_ptr != MAP_FAILED)
199 munmap(exec.rw_ptr, size);
200 if (exec.rx_ptr != MAP_FAILED)
201 munmap(exec.rx_ptr, size);
202 close(exec.shm_fd);
203 // Reset all fields to indicate failure.
204 return (infix_executable_t){.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0, .shm_fd = -1};
205 }
206#endif
207
208 exec.size = size;
209 INFIX_DEBUG_PRINTF("Allocated JIT memory. RW at %p, RX at %p", exec.rw_ptr, exec.rx_ptr);
210 return exec;
211}

224void infix_executable_free(infix_executable_t exec) {
225 if (exec.size == 0)
226 return;
227
228#if defined(INFIX_OS_WINDOWS)
229 if (exec.rw_ptr) {
230 // Turn the page into a guard page by revoking all access.
231 // In a free function, we can't return an error, but in a debug
232 // build, we should at least be aware of the failure.
233 if (!VirtualProtect(exec.rw_ptr, exec.size, PAGE_NOACCESS, &(DWORD){0}))
234 INFIX_DEBUG_PRINTF("WARNING: VirtualProtect failed to set PAGE_NOACCESS on freed trampoline.");
235 // Now, release the memory reservation.
236 VirtualFree(exec.rw_ptr, 0, MEM_RELEASE);
237 }
238#elif defined(INFIX_OS_MACOS) || defined(INFIX_OS_TERMUX) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
239 if (exec.rw_ptr) {
240 // On single-map platforms, rw_ptr == rx_ptr
241 // Turn the page into a guard page.
242 mprotect(exec.rw_ptr, exec.size, PROT_NONE);
243 // Now, unmap the memory.
244 munmap(exec.rw_ptr, exec.size);
245 }
246#else
247 // Other POSIX (Linux, BSD) with dual-mapping
248 // For dual-mapped systems, we only need to make the *executable* view inaccessible.
249 if (exec.rx_ptr)
250 mprotect(exec.rx_ptr, exec.size, PROT_NONE);
251 // Now unmap both views and close the shared memory file descriptor.
252 if (exec.rw_ptr)
253 munmap(exec.rw_ptr, exec.size);
254 if (exec.rx_ptr && exec.rx_ptr != exec.rw_ptr)
255 munmap(exec.rx_ptr, exec.size);
256 if (exec.shm_fd >= 0)
257 close(exec.shm_fd);
258#endif
259}

282c23_nodiscard bool infix_executable_make_executable(infix_executable_t exec) {
283 if (exec.rw_ptr == nullptr || exec.size == 0)
284 return false;
285
286#if defined(INFIX_ARCH_AARCH64)
287 // On all ARM64 platforms, the instruction cache must be flushed before execution.
288// This is done *before* changing memory permissions to ensure the writes are visible.
289#if defined(_MSC_VER)
290 FlushInstructionCache(GetCurrentProcess(), exec.rw_ptr, exec.size);
291#else
292 // A GCC/Clang built-in that emits the necessary cache maintenance instructions.
293 __builtin___clear_cache((char *)exec.rw_ptr, (char *)exec.rw_ptr + exec.size);
294#endif
295#endif
296
297 bool result = false;
298#if defined(INFIX_OS_WINDOWS)
299 // For single-map on Windows, we change the protection from RW to RX.
300 result = VirtualProtect(exec.rw_ptr, exec.size, PAGE_EXECUTE_READ, &(DWORD){0});
301#elif defined(INFIX_OS_MACOS) || defined(INFIX_OS_TERMUX) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
302 // For single-map on these POSIX systems, we also change protection from RW to RX.
303 result = (mprotect(exec.rw_ptr, exec.size, PROT_READ | PROT_EXEC) == 0);
304#else
305 result = true; // Dual-map is already executable
306#endif
307
308 if (result)
309 INFIX_DEBUG_PRINTF("Memory at %p is ready for execution.", exec.rx_ptr);
310 return result;
311}

374c23_nodiscard bool infix_protected_make_readonly(infix_protected_t prot) {
375 if (prot.size == 0)
376 return false;
377 bool result = false;
378#if defined(INFIX_OS_WINDOWS)
379 // On Linux and BSDs, this works as expected.
380 result = VirtualProtect(prot.rw_ptr, prot.size, PAGE_READONLY, &(DWORD){0});
381#elif !defined(INFIX_OS_MACOS)
382 // On Linux and BSDs, this works as expected.
383 result = (mprotect(prot.rw_ptr, prot.size, PROT_READ) == 0);
384#else
385 // On macOS, mprotect on mmap'd data pages can be unreliable.
386 // We skip this hardening step for now to ensure stability.
387 (void)prot;
388 result = true;
389#endif
390 return result;
391}

417void infix_internal_dispatch_callback_fn_impl(infix_reverse_t * context, void * return_value_ptr, void ** args_array) {
418 INFIX_DEBUG_PRINTF("In infix_internal_dispatch_callback_fn_impl");
419 INFIX_DEBUG_PRINTF(" Context: %p, User Callback: %p, NumArgs: %llu",
420 (void *)context,
421 context->user_callback_fn,
422 (unsigned long long)context->num_args);
423 infix_forward_t * trampoline = context->cached_forward_trampoline;
424 if (trampoline == nullptr) {
425 // This is a fatal internal error. We can't propagate an error from here,
426 // but we can prevent a crash by zeroing the return buffer if it exists.
427 if (return_value_ptr && context->return_type->size > 0)
428 infix_memset(return_value_ptr, 0, context->return_type->size);
429 return;
430 }
431
432 infix_cif_func cif_func = (infix_cif_func)infix_forward_get_code(trampoline);
433
435 // to call the user's C callback handler with the correct native ABI.
436
437 // The cached_forward_trampoline was generated to expect `num_args + 1` arguments,
438 // with the first one being the `infix_reverse_t*` context.
439 // We must construct a new argument array that reflects this.
440#if defined(INFIX_COMPILER_MSVC)
441 // MSVC does not support VLAs, so we use its intrinsic `_alloca` for stack allocation.
442 void ** callback_args = (void **)_alloca(sizeof(void *) * (context->num_args + 1));
443#else
444 void * callback_args[context->num_args + 1];
445#endif
446
447 // The first argument to the user's handler is the context pointer itself.
448 // The args array needs a pointer to this value.
449 callback_args[0] = &context;
450
451 // Copy the original argument pointers into the rest of the new array.
452 if (context->num_args > 0)
453 infix_memcpy(&callback_args[1], args_array, context->num_args * sizeof(void *));
454
455 // Call the user's handler through the trampoline with the complete, prepended argument list.
456 cif_func(context->user_callback_fn, return_value_ptr, callback_args);
457
458 INFIX_DEBUG_PRINTF("Exiting infix_internal_dispatch_callback_fn_impl");
459}
