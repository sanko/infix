66/*
67 * @internal
68 * Creates a temporary, anonymous shared memory file descriptor on POSIX systems
69 * that support the dual-mapping strategy (e.g., Linux).
70 *
71 * This function uses a "create-then-unlink" pattern for security and automatic cleanup:
72 * 1. It generates a unique, unpredictable name for the shared memory object.
73 * 2. It creates the object with `shm_open` using `O_CREAT | O_EXCL` to prevent
74 * race conditions and symlink attacks.
75 * 3. It immediately calls `shm_unlink` to remove the name from the filesystem.
76 *
77 * The open file descriptor remains valid, but the object is now truly anonymous
78 * and will be automatically cleaned up by the kernel when the process terminates
79 * or the descriptor is closed.
80 *
81 * @return A valid file descriptor on success, or -1 on failure.
82 */
83static int shm_open_anonymous() {
84 char shm_name[64];
85 uint64_t random_val = 0;
86
87 int rand_fd = open("/dev/urandom", O_RDONLY);
88 if (rand_fd < 0)
89 return -1;
90
91 ssize_t bytes_read = read(rand_fd, &random_val, sizeof(random_val));
92 close(rand_fd);
93 if (bytes_read != sizeof(random_val))
94 return -1;
95
96 snprintf(shm_name, sizeof(shm_name), "/infix-jit-%d-%llx", getpid(), (unsigned long long)random_val);
97
98 int fd = shm_open(shm_name, O_RDWR | O_CREAT | O_EXCL, 0600);
99 if (fd >= 0) {
100 // Unlink the name immediately. The fd remains valid, and the object
101 // becomes anonymous, guaranteeing kernel cleanup.
102 shm_unlink(shm_name);
103 return fd;
104 }
105
106 return -1;
107}

114/*
115 * Implementation for infix_executable_alloc.
116 * This is a cross-platform wrapper that allocates page-aligned memory suitable for JIT code.
117 *
118 * It employs different strategies based on the OS to achieve W^X:
119 * - Windows: Uses `VirtualAlloc` for a single region, initially RW.
120 * - macOS, OpenBSD, Termux: Uses a simple, single `mmap` with `MAP_PRIVATE | MAP_ANON`. This
121 * approach is more reliable and avoids `SIGBUS` errors common with more complex JIT
122 * setups on these specific platforms.
123 * - Linux & other POSIX: Uses a "dual-mapping" technique with an anonymous shared memory
124 * object to create separate writable (`rw_ptr`) and executable (`rx_ptr`) virtual
125 * mappings to the same physical memory, satisfying strict security policies.
126 */
127c23_nodiscard infix_executable_t infix_executable_alloc(size_t size) {
128 infix_executable_t exec = {.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0};
129 if (size == 0) {
130 return exec;
131 }
132
133#if defined(INFIX_OS_WINDOWS)
134 // Windows: Single Mapping (VirtualAlloc).
135 void * code = VirtualAlloc(nullptr, size, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
136 if (code == nullptr)
137 return exec;
138 exec.rw_ptr = code;
139 exec.rx_ptr = code;
140
141#elif defined(INFIX_OS_MACOS) || defined(INFIX_OS_TERMUX) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
142 // Single Mapping (mmap). This simpler approach is more reliable on these platforms.
143 void * code = MAP_FAILED;
144#if defined(MAP_ANON)
145 int flags = MAP_PRIVATE | MAP_ANON;
146#if defined(INFIX_OS_MACOS)
147 flags |= MAP_JIT;
148#endif
149 code = mmap(nullptr, size, PROT_READ | PROT_WRITE, flags, -1, 0);
150#else
151 // Fallback for systems that may not define MAP_ANON.
152 int fd = open("/dev/zero", O_RDWR);
153 if (fd != -1) {
154 code = mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_PRIVATE, fd, 0);
155 close(fd);
156 }
157#endif
158 if (code == MAP_FAILED)
159 return exec;
160
161 exec.rw_ptr = code;
162 exec.rx_ptr = code;
163
164#else
165 // Other POSIX (Linux, etc.): Dual Mapping (shm_open) for strict W^X.
166 exec.shm_fd = shm_open_anonymous();
167 if (exec.shm_fd < 0)
168 return exec;
169
170 if (ftruncate(exec.shm_fd, size) != 0) {
171 close(exec.shm_fd);
172 return exec;
173 }
174
175 // Create two separate virtual mappings to the same physical memory.
176 exec.rw_ptr = mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, exec.shm_fd, 0);
177 exec.rx_ptr = mmap(nullptr, size, PROT_READ | PROT_EXEC, MAP_SHARED, exec.shm_fd, 0);
178
179 if (exec.rw_ptr == MAP_FAILED || exec.rx_ptr == MAP_FAILED) {
180 if (exec.rw_ptr != MAP_FAILED)
181 munmap(exec.rw_ptr, size);
182 if (exec.rx_ptr != MAP_FAILED)
183 munmap(exec.rx_ptr, size);
184 close(exec.shm_fd);
185 return (infix_executable_t){.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0, .shm_fd = -1};
186 }
187#endif
188
189 exec.size = size;
190 INFIX_DEBUG_PRINTF("Allocated JIT memory. RW at %p, RX at %p", exec.rw_ptr, exec.rx_ptr);
191 return exec;
192}

194/*
195 * Implementation for infix_executable_free.
196 * This function first changes the memory's protection to be inaccessible
197 * (`PAGE_NOACCESS` or `PROT_NONE`), turning it into a "guard page". Any
198 * attempt to execute a dangling function pointer that points to this freed memory
199 * will cause an immediate and safe access violation, rather than executing
200 * stale or unrelated data. This is a critical security hardening feature.
201 */
202void infix_executable_free(infix_executable_t exec) {
203 if (exec.size == 0)
204 return;
205
206#if defined(INFIX_OS_WINDOWS)
207 if (exec.rw_ptr) {
208 // Arm the guard page by revoking all access.
209 if (!VirtualProtect(exec.rw_ptr, exec.size, PAGE_NOACCESS, &(DWORD){0}))
210 INFIX_DEBUG_PRINTF("WARNING: VirtualProtect failed to set PAGE_NOACCESS.");
211 // Now, release the memory.
212 VirtualFree(exec.rw_ptr, 0, MEM_RELEASE);
213 }
214#elif defined(INFIX_OS_MACOS) || defined(INFIX_OS_TERMUX) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
215 if (exec.rw_ptr) {
216 // On single-map platforms, rw_ptr == rx_ptr.
217 mprotect(exec.rw_ptr, exec.size, PROT_NONE);
218 munmap(exec.rw_ptr, exec.size);
219 }
220#else
221 // On dual-map platforms, we unmap both views and close the shared memory fd.
222 if (exec.rx_ptr)
223 mprotect(exec.rx_ptr, exec.size, PROT_NONE);
224 if (exec.rw_ptr)
225 munmap(exec.rw_ptr, exec.size);
226 if (exec.rx_ptr && exec.rx_ptr != exec.rw_ptr)
227 munmap(exec.rx_ptr, exec.size);
228 if (exec.shm_fd >= 0)
229 close(exec.shm_fd);
230#endif
231}

233/*
234 * Implementation for infix_executable_make_executable.
235 * This is the final step in the JIT process. After machine code has been written
236 * to `rw_ptr`, this function makes `rx_ptr` executable.
237 *
238 * It performs two critical, platform-specific actions:
239 * 1. On AArch64, it flushes the instruction cache to ensure the CPU sees the newly
240 * written code bytes. This is done *before* changing permissions.
241 * 2. On single-map platforms (Windows, macOS), it changes the memory protection
242 * from Read/Write to Read/Execute, thereby enforcing W^X. On dual-map platforms
243 * (Linux), this is a no-op as `rx_ptr` was already created with PROT_EXEC.
244 */
245c23_nodiscard bool infix_executable_make_executable(infix_executable_t exec) {
246 if (exec.rw_ptr == nullptr || exec.size == 0)
247 return false;
248
249#if defined(INFIX_ARCH_AARCH64)
250 // On all ARM64 platforms, the instruction cache must be flushed *before* execution
251 // to ensure the CPU's instruction pipeline sees the newly written data.
252#if defined(_MSC_VER)
253 FlushInstructionCache(GetCurrentProcess(), exec.rw_ptr, exec.size);
254#else
255 __builtin___clear_cache((char *)exec.rw_ptr, (char *)exec.rw_ptr + exec.size);
256#endif
257#endif
258
259 bool result = false;
260#if defined(INFIX_OS_WINDOWS)
261 // On single-map Windows, change protection from RW to RX.
262 result = VirtualProtect(exec.rw_ptr, exec.size, PAGE_EXECUTE_READ, &(DWORD){0});
263#elif defined(INFIX_OS_MACOS) || defined(INFIX_OS_TERMUX) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
264 // On single-map POSIX, change protection from RW to RX.
265 result = (mprotect(exec.rw_ptr, exec.size, PROT_READ | PROT_EXEC) == 0);
266#else
267 // On dual-map platforms, the rx_ptr is already executable.
268 result = true;
269#endif
270
271 if (result)
272 INFIX_DEBUG_PRINTF("Memory at %p is ready for execution.", exec.rx_ptr);
273 return result;
274}

285c23_nodiscard infix_protected_t infix_protected_alloc(size_t size) {

287 if (size == 0)
288 return prot;

292#else // POSIX platforms

302 }
303#endif
304 if (prot.rw_ptr == MAP_FAILED)
305 prot.rw_ptr = nullptr;
306#endif
307
308 if (prot.rw_ptr)
309 prot.size = size;
310 return prot;
311}

327/*
328 * Implementation for infix_protected_make_readonly.
329 * This is a security hardening step. After the reverse trampoline context is fully
330 * initialized, it is made read-only to prevent it from being modified at runtime,
331 * which could otherwise lead to security vulnerabilities.
332 * Note: This feature is currently disabled on macOS because `mprotect` on
333 * general-purpose `mmap`'d memory has been observed to be unreliable.
334 */
335c23_nodiscard bool infix_protected_make_readonly(infix_protected_t prot) {
336 if (prot.size == 0)
337 return false;
338 bool result = false;
339#if defined(INFIX_OS_WINDOWS)
340 result = VirtualProtect(prot.rw_ptr, prot.size, PAGE_READONLY, &(DWORD){0});
341#elif !defined(INFIX_OS_MACOS)
342 // On Linux and other BSDs, this works as expected.
343 result = (mprotect(prot.rw_ptr, prot.size, PROT_READ) == 0);
344#else
345 // On macOS, mprotect on mmap'd data pages can be unreliable.
346 // We skip this hardening step for now to ensure stability.
347 (void)prot;
348 result = true;
349#endif
350 return result;
351}

357/*
358 * Implementation for infix_internal_dispatch_callback_fn_impl.
359 * This is the high-level C bridge called by the low-level assembly of a reverse trampoline.
360 * It receives arguments in a normalized format (`void** args_array`).
361 *
362 * Its primary job is to prepare and call the user's C handler via a pre-compiled
363 * forward trampoline (`cached_forward_trampoline`). This involves:
364 * 1. Constructing a new, augmented argument array on the stack.
365 * 2. Setting the *first* element of this new array to be a pointer to the `context`.
366 * 3. Copying the original arguments into the rest of the new array.
367 * 4. Calling the cached trampoline, which was generated to expect this exact
368 * `(context, ...)` argument signature.
369 */
370void infix_internal_dispatch_callback_fn_impl(infix_reverse_t * context, void * return_value_ptr, void ** args_array) {
371 INFIX_DEBUG_PRINTF("Dispatching callback. Context: %p, User Fn: %p", (void *)context, context->user_callback_fn);
372
373 infix_forward_t * trampoline = context->cached_forward_trampoline;
374 if (trampoline == nullptr) {
375 // This is a fatal internal error. We cannot propagate it, but we can prevent
376 // a crash by zeroing the return buffer if it exists.
377 if (return_value_ptr && context->return_type->size > 0)
378 infix_memset(return_value_ptr, 0, context->return_type->size);
379 return;
380 }
381
382 infix_cif_func cif_func = (infix_cif_func)infix_forward_get_code(trampoline);
383
384 // The cached forward trampoline was generated to expect `num_args + 1` arguments,
385 // with the first one being the `infix_reverse_t*` context. We must construct a
386 // new argument array that reflects this.
387#if defined(INFIX_COMPILER_MSVC)
388 // MSVC does not support VLAs, so we use its intrinsic `_alloca` for stack allocation.
389 void ** callback_args = (void **)_alloca(sizeof(void *) * (context->num_args + 1));
390#else
391 void * callback_args[context->num_args + 1];
392#endif
393
394 // The first argument to the user's handler is the context pointer itself.
395 callback_args[0] = &context;
396
397 // Copy the original argument pointers into the rest of the new array.
398 if (context->num_args > 0)
399 infix_memcpy(&callback_args[1], args_array, context->num_args * sizeof(void *));
400
401 // Call the user's handler through the trampoline with the complete, prepended argument list.
402 cif_func(context->user_callback_fn, return_value_ptr, callback_args);
403
404 INFIX_DEBUG_PRINTF("Exiting callback dispatcher.");
405}
