65c23_nodiscard infix_library_t * infix_library_open(const char * path) {
66 if (path == nullptr)
67 return nullptr;
68
69 // Allocate memory for our opaque wrapper struct.
70 infix_library_t * lib = infix_malloc(sizeof(infix_library_t));
71 if (lib == nullptr)
72 return nullptr;
73
74#if defined(INFIX_OS_WINDOWS)
75 // On Windows, use LoadLibraryA to load the DLL.
76 // 'A' specifies the ANSI version, which matches the `const char*` input.
77 lib->handle = LoadLibraryA(path);
78#else
79 // On POSIX systems, use dlopen.
80 // - RTLD_LAZY: Resolves symbols only when they are first used (lazy binding),
81 // which can improve startup performance.
82 // - RTLD_GLOBAL: Makes symbols from this library available for resolution
83 // by subsequently loaded libraries. This is important for handling
84 // complex dependencies between shared objects.
85 lib->handle = dlopen(path, RTLD_LAZY | RTLD_GLOBAL);
86#endif
87
88 // Both LoadLibraryA and dlopen return NULL on failure.
89 if (lib->handle == nullptr) {
90 // If the OS call failed, we must free the wrapper struct we allocated
91 // to prevent a memory leak.
92 infix_free(lib);
93 return nullptr;
94 }
95 return lib;
96}

106void infix_library_close(infix_library_t * lib) {
107 if (lib == nullptr)
108 return;
109
110 // Only attempt to unload if the native handle is valid.
111 if (lib->handle) {
112#if defined(INFIX_OS_WINDOWS)
113 // On Windows, FreeLibrary decrements the library's reference count.
114 // The DLL is unloaded when its reference count reaches zero.
115 FreeLibrary((HMODULE)lib->handle);
116#else
117 // On POSIX, dlclose does the same.
118 dlclose(lib->handle);
119#endif
120 }
121 // Finally, free our wrapper struct.
122 infix_free(lib);
123}

137c23_nodiscard void * infix_library_get_symbol(infix_library_t * lib, const char * symbol_name) {
138 if (lib == nullptr || lib->handle == nullptr || symbol_name == nullptr)
139 return nullptr;
140
141#if defined(INFIX_OS_WINDOWS)
142 // On Windows, use GetProcAddress. The return type is technically FARPROC,
143 // which must be cast to a generic `void*` for our cross-platform API.
144 return (void *)GetProcAddress((HMODULE)lib->handle, symbol_name);
145#else
146 // On POSIX, use dlsym. It directly returns a `void*`.
147 return dlsym(lib->handle, symbol_name);
148#endif
149}

163c23_nodiscard infix_status infix_read_global(infix_library_t * lib,
164 const char * symbol_name,
165 const char * type_signature,
166 void * buffer) {
167 if (buffer == nullptr)
168 return INFIX_ERROR_INVALID_ARGUMENT;
169
170 // Find the address of the global variable in the library.
171 void * symbol_addr = infix_library_get_symbol(lib, symbol_name);
172 if (symbol_addr == nullptr)
173 return INFIX_ERROR_INVALID_ARGUMENT;
174
175 // Parse the type signature to determine the size of the variable.
176 // This creates a temporary arena to hold the type information.
177 infix_type * type = nullptr;
178 infix_arena_t * arena = nullptr;
179 infix_status status = infix_type_from_signature(&type, &arena, type_signature);
180
181 if (status != INFIX_SUCCESS)
182 return status;
183
184 // Perform a memory copy of the correct size.
185 memcpy(buffer, symbol_addr, type->size);
186
187 // Clean up the temporary arena used for parsing.
188 infix_arena_destroy(arena);
189 return INFIX_SUCCESS;
190}
