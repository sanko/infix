42/*
43 * @internal
44 * A security limit on the nesting depth of type definitions.
45 * This constant prevents the recursive-descent parser from causing a
46 * stack overflow when parsing maliciously crafted input with excessive
47 * nesting, such as `[[[[...[int]...]]]]`.
48 */

52 * @internal
53 * @struct parser_state
54 * @brief Holds the complete state of the parser during a run.
55 * @details This struct is passed by pointer to all parsing functions, allowing

78static void skip_whitespace(parser_state * state) {
79 while (true) {
80 // First, consume all standard whitespace characters.
81 while (isspace((unsigned char)*state->p))
82 state->p++;
83 // Next, check for the start of a line comment.
84 if (*state->p == '#') {
85 // If found, consume all characters until the next newline or the end of the string.
86 while (*state->p != '\n' && *state->p != '\0')
87 state->p++;
88 }
89 else
90 // If neither whitespace nor a comment is found, we're done.
91 break;
92 }
93}

103static bool parse_size_t(parser_state * state, size_t * out_val) {
104 const char * start = state->p;
105 char * end;
106 // Use strtoull for battle tested parsing of 64-bit unsigned integers.
107 unsigned long long val = strtoull(start, &end, 10);
108 // If the end pointer hasn't moved, no digits were consumed.
109 if (end == start) {
110 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
111 return false;
112 }
113 *out_val = (size_t)val;
114 state->p = end; // Advance the parser's pointer.
115 return true;
116}

126static const char * parse_identifier(parser_state * state) {
127 skip_whitespace(state);
128 const char * start = state->p;
129 // An identifier must start with an alphabetic character or an underscore.
130 if (!isalpha((unsigned char)*start) && *start != '_')
131 return nullptr;
132 // Consume subsequent alphanumeric characters or underscores.
133 while (isalnum((unsigned char)*state->p) || *state->p == '_')
134 state->p++;
135 size_t len = state->p - start;
136 if (len == 0)
137 return nullptr;
138
139 // Allocate space for the identifier in the arena and copy it.
140 char * name = infix_arena_alloc(state->arena, len + 1, 1);
141 if (!name) {
142 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
143 return nullptr;
144 }
145 memcpy(name, start, len);
146 name[len] = '\0';
147 return name;
148}

159static bool consume_keyword(parser_state * state, const char * keyword) {
160 skip_whitespace(state);
161 size_t len = strlen(keyword);
162 if (strncmp(state->p, keyword, len) == 0) {
163 // This is the "whole word" check. The character after the keyword
164 // cannot be alphanumeric, which would imply it's part of a larger word.
165 if (isalnum((unsigned char)state->p[len]) || state->p[len] == '_')
166 return false;
167
168 // The keyword matched. Advance the pointer and return success.
169 state->p += len;
170 skip_whitespace(state);
171 return true;
172 }
173 return false;
174}

183static const char * parse_optional_name_prefix(parser_state * state) {
184 skip_whitespace(state);
185 const char * p_before = state->p;
186
187 const char * name = parse_identifier(state);
188 if (name) {
189 skip_whitespace(state);
190 if (*state->p == ':') {
191 state->p++; // Consume colon
192 return name;
193 }
194 }
195
196 // If it wasn't a "name:" pattern, backtrack.
197 state->p = p_before;
198 return nullptr;
199}

213 if (consume_keyword(state, "int8"))

215 if (consume_keyword(state, "uint8"))
216 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT8);
217 if (consume_keyword(state, "int16"))
218 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT16);
219 if (consume_keyword(state, "uint16"))
220 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT16);
221 if (consume_keyword(state, "int32"))
222 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT32);
223 if (consume_keyword(state, "uint32"))
224 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT32);
225 if (consume_keyword(state, "int64"))
226 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT64);
227 if (consume_keyword(state, "uint64"))
228 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT64);
229 if (consume_keyword(state, "int128"))
230 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT128);
231 if (consume_keyword(state, "uint128"))
232 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT128);

250 if (consume_keyword(state, "short"))

257 if (consume_keyword(state, "ulonglong"))
258 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT64);
259 if (consume_keyword(state, "longlong"))

263 : INFIX_PRIMITIVE_UINT32);
264 if (consume_keyword(state, "long"))
265 return infix_type_create_primitive(sizeof(long) == 8 ? INFIX_PRIMITIVE_SINT64 : INFIX_PRIMITIVE_SINT32);
266 if (consume_keyword(state, "double"))
267 return infix_type_create_primitive(INFIX_PRIMITIVE_DOUBLE);
268 if (consume_keyword(state, "float"))
269 return infix_type_create_primitive(INFIX_PRIMITIVE_FLOAT);
270 // If no keyword matches, it's not a primitive type.
271 return nullptr;

283static bool is_function_signature_ahead(const parser_state * state) {
284 const char * p = state->p;
285 // The ambiguity only exists if we start with '('.
286 if (*p != '(')
287 return false;
288 p++;
289
290 // Find the matching closing parenthesis, respecting nested parentheses.
291 int depth = 1;
292 while (*p != '\0' && depth > 0) {
293 if (*p == '(')
294 depth++;
295 else if (*p == ')')
296 depth--;
297 p++;
298 }
299 // If we didn't find a matching parenthesis, it's malformed.
300 if (depth != 0)
301 return false;
302
303 // Skip any whitespace or comments between the ')' and the next token.
304 while (isspace((unsigned char)*p) || *p == '#') {
305 if (*p == '#') {
306 while (*p != '\n' && *p != '\0')
307 p++;
308 }
309 else
310 p++;
311 }
312
313 // The upcoming construct is a function signature if and only if the next
314 // non-whitespace token is the arrow '->'.
315 return (p[0] == '-' && p[1] == '>');
316}

331static infix_struct_member * parse_aggregate_members(parser_state * state, char end_char, size_t * out_num_members) {
332 // A temporary node for building the linked list of members.
333 typedef struct member_node {
334 infix_struct_member m;
335 struct member_node * next;
336 } member_node;
337 member_node *head = nullptr, *tail = nullptr;
338 size_t num_members = 0;
339
340 skip_whitespace(state);
341 // Handle empty aggregates like `{}` or `<>` by checking for the end character immediately.
342 if (*state->p != end_char) {
343 while (1) {
344 const char * name = parse_optional_name_prefix(state);
345 infix_type * member_type = parse_type(state);
346
347 if (!member_type) {
348 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
349 return nullptr;
350 }
351
352 // The C standard forbids members of type void.
353 if (member_type->category == INFIX_TYPE_VOID) {
354 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
355 return nullptr;
356 }
357
358 // Add the parsed member to our linked list.
359 member_node * node = infix_arena_alloc(state->arena, sizeof(member_node), _Alignof(member_node));
360 if (!node) {
361 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
362 return nullptr;
363 }
364 node->m = infix_type_create_member(name, member_type, 0); // Offset is calculated later.
365 node->next = nullptr;
366 if (!head)
367 head = tail = node;
368 else {
369 tail->next = node;
370 tail = node;
371 }
372 num_members++;
373
374 skip_whitespace(state);
375 // After a member, we expect either a comma or the end character.
376 if (*state->p == ',') {
377 state->p++;
378 skip_whitespace(state);
379 // A trailing comma like `{int,}` is a syntax error.
380 if (*state->p == end_char) {
381 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
382 return nullptr;
383 }
384 }
385 else if (*state->p == end_char)
386 break; // End of the list.
387 else {
388 // Anything else is a syntax error.
389 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
390 return nullptr;
391 }
392 }
393 }
394
395 *out_num_members = num_members;
396 if (num_members == 0)
397 return nullptr;
398
399 // Pass 2: Convert the linked list to a contiguous array.
400 infix_struct_member * members =
401 infix_arena_alloc(state->arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
402 if (!members) {
403 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
404 return nullptr;
405 }
406 member_node * current = head;
407 for (size_t i = 0; i < num_members; i++) {
408 members[i] = current->m;
409 current = current->next;
410 }
411 return members;
412}

421static infix_type * parse_packed_struct(parser_state * state) {
422 size_t alignment = 1; // The default alignment for a packed struct is 1.
423 if (*state->p == '!') {
424 state->p++;
425 // Check for an explicit alignment override, e.g., `!4:{...}`.
426 if (isdigit((unsigned char)*state->p)) {
427 if (!parse_size_t(state, &alignment))
428 return nullptr;
429 if (*state->p != ':') {
430 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
431 return nullptr;
432 }
433 state->p++;
434 }
435 }
436
437 skip_whitespace(state);
438 if (*state->p != '{') {
439 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
440 return nullptr;
441 }
442 state->p++;
443
444 size_t num_members = 0;
445 infix_struct_member * members = parse_aggregate_members(state, '}', &num_members);
446 if (state->last_error != INFIX_SUCCESS && num_members > 0)
447 return nullptr;
448
449 if (*state->p != '}') {
450 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
451 return nullptr;
452 }
453 state->p++;
454
455 infix_type * packed_type = nullptr;
456 // For packed structs, the size is simply the sum of member sizes, as there's
457 // no internal padding. The final layout calculation is handled by the `_create` function.
458 size_t total_size = 0;
459 for (size_t i = 0; i < num_members; ++i)
460 total_size += members[i].type->size;
461
462 infix_status status =
463 infix_type_create_packed_struct(state->arena, &packed_type, total_size, alignment, members, num_members);
464 if (status != INFIX_SUCCESS) {
465 state->last_error = status;
466 return nullptr;
467 }
468 return packed_type;
469}

477static infix_type * parse_vector_type(parser_state * state) {
478 state->p++; // Consume 'v'
479 skip_whitespace(state);
480
481 if (*state->p != '[') {
482 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
483 return nullptr;
484 }
485 state->p++; // Consume '['
486 skip_whitespace(state);
487
488 size_t num_elements;
489 if (!parse_size_t(state, &num_elements))
490 return nullptr;
491
492 skip_whitespace(state);
493
494 if (*state->p != ':') {
495 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
496 return nullptr;
497 }
498 state->p++; // Consume ':'
499 skip_whitespace(state);
500
501 infix_type * element_type = parse_type(state);
502 if (!element_type) {
503 return nullptr;
504 }
505 if (element_type->category != INFIX_TYPE_PRIMITIVE) {
506 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
507 return nullptr;
508 }
509
510 skip_whitespace(state);
511 if (*state->p != ']') {
512 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
513 return nullptr;
514 }
515 state->p++; // Consume ']'
516
517 infix_type * vector_type = nullptr;
518 if (infix_type_create_vector(state->arena, &vector_type, element_type, num_elements) != INFIX_SUCCESS) {
519 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
520 return nullptr;
521 }
522 return vector_type;
523}

533static infix_type * parse_function_type(parser_state * state) {
534 infix_type * ret_type = nullptr;
535 infix_function_argument * args = nullptr;
536 size_t num_args = 0, num_fixed = 0;
537
538 // Delegate the heavy lifting of parsing the argument list, arrow, and return type.
539 if (parse_function_signature_details(state, &ret_type, &args, &num_args, &num_fixed) != INFIX_SUCCESS)
540 return nullptr;
541
542 // If parsing was successful, allocate a new type descriptor to represent the function signature.
543 infix_type * func_type = infix_arena_alloc(state->arena, sizeof(infix_type), _Alignof(infix_type));
544 if (!func_type) {
545 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
546 return nullptr;
547 }
548
549 // A function signature type itself has the size and alignment of a pointer.
550 *func_type = *infix_type_create_pointer();
551 func_type->is_arena_allocated = true;
552 func_type->category = INFIX_TYPE_REVERSE_TRAMPOLINE;
553
554 // Populate the metadata with the parsed components.
555 func_type->meta.func_ptr_info.return_type = ret_type;
556 func_type->meta.func_ptr_info.args = args;
557 func_type->meta.func_ptr_info.num_args = num_args;
558 func_type->meta.func_ptr_info.num_fixed_args = num_fixed;
559
560 return func_type;
561}

572static infix_type * parse_aggregate(parser_state * state, char start_char, char end_char, const char * name) {
573 if (state->depth >= MAX_RECURSION_DEPTH) {
574 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
575 return nullptr;
576 }
577 state->depth++;
578
579 if (*state->p != start_char) {
580 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
581 state->depth--;
582 return nullptr;
583 }
584 state->p++;
585
586 size_t num_members = 0;
587 infix_struct_member * members = parse_aggregate_members(state, end_char, &num_members);
588 if (state->last_error != INFIX_SUCCESS) {
589 state->depth--;
590 return nullptr;
591 }
592
593 if (*state->p != end_char) {
594 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
595 state->depth--;
596 return nullptr;
597 }
598 state->p++;
599
600 infix_type * agg_type = nullptr;
601 // Dispatch to the correct creation function based on the delimiter.
602 infix_status status = (start_char == '{') ? infix_type_create_struct(state->arena, &agg_type, members, num_members)
603 : infix_type_create_union(state->arena, &agg_type, members, num_members);
604
605 if (status != INFIX_SUCCESS) {
606 state->last_error = status;
607 state->depth--;
608 return nullptr;
609 }
610 (void)name; // Name is stored for introspection but not used by the core ABI.
611
612 state->depth--;
613 return agg_type;
614}

625static infix_type * parse_type(parser_state * state) {
626 // Security: Prevent stack overflow from deeply nested type definitions.
627 if (state->depth >= MAX_RECURSION_DEPTH) {
628 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
629 return nullptr;
630 }
631 state->depth++;
632
633 skip_whitespace(state);
634 infix_type * result_type = nullptr;
635 const char * p_before_type = state->p; // Save position for backtracking on error.
636
637 // The dispatch logic begins here, checking one token at a time.
638 if (*state->p == '*') {
639 // POINTER TYPE: *<type>
640 state->p++; // Consume the '*'.
641 skip_whitespace(state);
642 // Recursively call parse_type to get the type the pointer points to.
643 infix_type * pointee_type = parse_type(state);
644 if (!pointee_type) {
645 state->depth--;
646 return nullptr;
647 } // Propagate failure.
648 // Create the final pointer type.
649 if (infix_type_create_pointer_to(state->arena, &result_type, pointee_type) != INFIX_SUCCESS) {
650 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
651 result_type = nullptr;
652 }
653 }
654 else if (*state->p == '(') {
655 // GROUPED TYPE or FUNCTION TYPE
656 // This is the ambiguous case. We must look ahead to see if it's a function.
657 if (is_function_signature_ahead(state))
658 result_type = parse_function_type(state);
659 else {
660 // It's a simple grouped type, e.g., *( (int)->void ).
661 state->p++; // Consume '('.
662 skip_whitespace(state);
663 result_type = parse_type(state); // Parse the inner type.
664 if (!result_type) {
665 state->depth--;
666 return nullptr;
667 }
668 skip_whitespace(state);
669 if (*state->p != ')') {
670 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
671 result_type = nullptr;
672 }
673 else
674 state->p++; // Consume ')'.
675 }
676 }
677 else if (*state->p == '[') {
678 // ARRAY TYPE: [<size>:<type>]
679 state->p++;
680 skip_whitespace(state);
681 size_t num_elements;
682 if (!parse_size_t(state, &num_elements)) {
683 state->depth--;
684 return nullptr;
685 }
686 skip_whitespace(state);
687 if (*state->p != ':') {
688 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
689 state->depth--;
690 return nullptr;
691 }
692 state->p++;
693 skip_whitespace(state);
694 infix_type * element_type = parse_type(state);
695 if (!element_type) {
696 state->depth--;
697 return nullptr;
698 }
699 // Arrays of void are illegal in C.
700 if (element_type->category == INFIX_TYPE_VOID) {
701 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
702 state->depth--;
703 return nullptr;
704 }
705 skip_whitespace(state);
706 if (*state->p != ']') {
707 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
708 state->depth--;
709 return nullptr;
710 }
711 state->p++;
712 if (infix_type_create_array(state->arena, &result_type, element_type, num_elements) != INFIX_SUCCESS) {
713 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
714 result_type = nullptr;
715 }
716 }
717 else if (*state->p == '!')
718 // PACKED STRUCT TYPE: !<align>?:{...}
719 result_type = parse_packed_struct(state);
720 else if (*state->p == '{')
721 // ANONYMOUS STRUCT TYPE: {...}
722 result_type = parse_aggregate(state, '{', '}', nullptr);
723 else if (*state->p == '<')
724 // ANONYMOUS UNION TYPE: <...>
725 result_type = parse_aggregate(state, '<', '>', nullptr);
726 else if (consume_keyword(state, "struct")) {
727 // NAMED STRUCT or REFERENCE: struct<Name>{...} or struct<Name>
728 if (*state->p != '<') {
729 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
730 state->depth--;
731 return nullptr;
732 }
733 state->p++;
734 const char * name = parse_identifier(state);
735 if (!name) {
736 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
737 state->depth--;
738 return nullptr;
739 }
740 if (*state->p != '>') {
741 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
742 state->depth--;
743 return nullptr;
744 }
745 state->p++;
746 skip_whitespace(state);
747 // If a body follows, it's a definition.
748 if (*state->p == '{')
749 result_type = parse_aggregate(state, '{', '}', name);
750 else {
751 // Otherwise, it's a forward declaration (a named reference).
752 infix_status status = infix_type_create_named_reference(state->arena, &result_type, name);
753 if (status != INFIX_SUCCESS) {
754 state->last_error = status;
755 result_type = nullptr;
756 }
757 }
758 }
759 else if (consume_keyword(state, "union")) {
760 // NAMED UNION or REFERENCE: union<Name><...> or union<Name>
761 if (*state->p != '<') {
762 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
763 state->depth--;
764 return nullptr;
765 }
766 state->p++;
767 const char * name = parse_identifier(state);
768 if (!name) {
769 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
770 state->depth--;
771 return nullptr;
772 }
773 if (*state->p != '>') {
774 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
775 state->depth--;
776 return nullptr;
777 }
778 state->p++;
779 skip_whitespace(state);
780 // If a body follows, it's a definition.
781 if (*state->p == '<')
782 result_type = parse_aggregate(state, '<', '>', name);
783 else {
784 // Otherwise, it's a forward declaration (a named reference).
785 infix_status status = infix_type_create_named_reference(state->arena, &result_type, name);
786 if (status != INFIX_SUCCESS) {
787 state->last_error = status;
788 result_type = nullptr;
789 }
790 }
791 }
792 else if (*state->p == 'e') {
793 // ENUM TYPE: e<Name>?:<type>
794 state->p++;
795 skip_whitespace(state);
796 const char * name = nullptr;
797 // Check for optional name, e.g., e<MyEnum>:.
798 if (*state->p == '<') {
799 state->p++;
800 name = parse_identifier(state);
801 if (!name) {
802 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
803 state->depth--;
804 return nullptr;
805 }
806 if (*state->p != '>') {
807 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
808 state->depth--;
809 return nullptr;
810 }
811 state->p++;
812 skip_whitespace(state);
813 }
814 if (*state->p != ':') {
815 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
816 state->depth--;
817 return nullptr;
818 }
819 state->p++;
820 skip_whitespace(state);
821 // An enum must have an explicit underlying integer type.
822 infix_type * underlying_type = parse_type(state);
823 if (!underlying_type) {
824 state->depth--;
825 return nullptr;
826 }
827 // The spec requires the underlying type to be an integer.
828 if (underlying_type->category != INFIX_TYPE_PRIMITIVE) {
829 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
830 state->depth--;
831 return nullptr;
832 }
833 if (infix_type_create_enum(state->arena, &result_type, underlying_type) != INFIX_SUCCESS) {
834 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
835 result_type = nullptr;
836 }
837 (void)name; // Name is stored for introspection but not used by the core ABI.
838 }
839 else if (*state->p == 'c' && state->p[1] == '[') { // extra test to make sure we avoid eating `char`
840 // COMPLEX TYPE: c[<type>]
841 state->p++;
842 skip_whitespace(state);
843 if (*state->p != '[') {
844 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
845 state->depth--;
846 return nullptr;
847 }
848 state->p++;
849 skip_whitespace(state);
850 infix_type * base_type = parse_type(state);
851 if (!base_type) {
852 state->depth--;
853 return nullptr;
854 }
855 skip_whitespace(state);
856 if (*state->p != ']') {
857 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
858 state->depth--;
859 return nullptr;
860 }
861 state->p++;
862 if (infix_type_create_complex(state->arena, &result_type, base_type) != INFIX_SUCCESS) {
863 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
864 result_type = nullptr;
865 }
866 }
867 else if (*state->p == 'v' && state->p[1] == '[')
868 // VECTOR TYPE: v[<N>:<type>]
869 result_type = parse_vector_type(state);
870 else {
871 // PRIMITIVE TYPE
872 // If none of the above constructors match, it must be a primitive type.
873 result_type = parse_primitive(state);
874 if (!result_type) {
875 // If even that fails, this is an unrecognized token. Backtrack and report error.
876 state->p = p_before_type;
877 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
878 }
879 }
880
881 state->depth--;
882 return result_type;
883}

893static infix_status parse_function_signature_details(parser_state * state,
894 infix_type ** out_ret_type,
895 infix_function_argument ** out_args,
896 size_t * out_num_args,
897 size_t * out_num_fixed_args) {
898 if (*state->p != '(')
899 return INFIX_ERROR_INVALID_ARGUMENT;
900 state->p++;
901 skip_whitespace(state);
902
903 typedef struct arg_node {
904 infix_function_argument arg;
905 struct arg_node * next;
906 } arg_node;
907 arg_node *head = nullptr, *tail = nullptr;
908 size_t num_args = 0;
909
910 // Phase 1: Parse Fixed Arguments
911 if (*state->p != ')') {
912 while (1) {
913 skip_whitespace(state);
914 // The list of fixed arguments can be terminated by a ')' or a ';'.
915 if (*state->p == ')' || *state->p == ';')
916 break;
917
918 const char * name = parse_optional_name_prefix(state);
919 infix_type * arg_type = parse_type(state);
920 if (!arg_type)
921 return state->last_error;
922
923 arg_node * node = infix_arena_alloc(state->arena, sizeof(arg_node), _Alignof(arg_node));
924 if (!node) {
925 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
926 return INFIX_ERROR_ALLOCATION_FAILED;
927 }
928 node->arg.type = arg_type;
929 node->arg.name = name;
930 node->next = nullptr;
931 if (!head)
932 head = tail = node;
933 else {
934 tail->next = node;
935 tail = node;
936 }
937 num_args++;
938
939 skip_whitespace(state);
940 if (*state->p == ',') {
941 state->p++;
942 skip_whitespace(state);
943 // A comma must be followed by another type, not the end of the list.
944 if (*state->p == ')' || *state->p == ';') {
945 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
946 return INFIX_ERROR_INVALID_ARGUMENT;
947 }
948 }
949 else if (*state->p != ')' && *state->p != ';')
950 return INFIX_ERROR_INVALID_ARGUMENT;
951 else
952 break; // End of the fixed argument list.
953 }
954 }
955 *out_num_fixed_args = num_args;
956
957 // Phase 2: Parse Variadic Arguments (if any)
958 if (*state->p == ';') {
959 state->p++;
960 if (*state->p != ')') { // Check for arguments after the separator.
961 while (1) {
962 skip_whitespace(state);
963 if (*state->p == ')')
964 break; // End of variadic list.
965
966 const char * name = parse_optional_name_prefix(state);
967 infix_type * arg_type = parse_type(state);
968 if (!arg_type)
969 return state->last_error;
970
971 arg_node * node = infix_arena_alloc(state->arena, sizeof(arg_node), _Alignof(arg_node));
972 if (!node) {
973 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
974 return INFIX_ERROR_ALLOCATION_FAILED;
975 }
976 node->arg.type = arg_type;
977 node->arg.name = name;
978 node->next = nullptr;
979 if (!head)
980 head = tail = node;
981 else {
982 tail->next = node;
983 tail = node;
984 }
985 num_args++;
986
987 skip_whitespace(state);
988 if (*state->p == ',') {
989 state->p++;
990 skip_whitespace(state);
991 // A comma must be followed by another type, not the end of the list.
992 if (*state->p == ')') {
993 state->last_error = INFIX_ERROR_INVALID_ARGUMENT;
994 return INFIX_ERROR_INVALID_ARGUMENT;
995 }
996 }
997 else if (*state->p != ')')
998 return INFIX_ERROR_INVALID_ARGUMENT;
999 else
1000 break;
1001 }
1002 }
1003 }
1004
1005 skip_whitespace(state);
1006 if (*state->p != ')')
1007 return INFIX_ERROR_INVALID_ARGUMENT;
1008 state->p++;
1009 skip_whitespace(state);
1010
1011 if (state->p[0] != '-' || state->p[1] != '>')
1012 return INFIX_ERROR_INVALID_ARGUMENT;
1013 state->p += 2;
1014
1015 *out_ret_type = parse_type(state);
1016 if (!*out_ret_type)
1017 return state->last_error;
1018
1019 // Convert the linked list into a final, contiguous array.
1020 infix_function_argument * args = num_args > 0
1021 ? infix_arena_alloc(state->arena, sizeof(infix_function_argument) * num_args, _Alignof(infix_function_argument))
1022 : nullptr;
1023 if (num_args > 0 && !args) {
1024 state->last_error = INFIX_ERROR_ALLOCATION_FAILED;
1025 return INFIX_ERROR_ALLOCATION_FAILED;
1026 }
1027 arg_node * current = head;
1028 for (size_t i = 0; i < num_args; i++) {
1029 args[i] = current->arg;
1030 current = current->next;
1031 }
1032
1033 *out_args = args;
1034 *out_num_args = num_args;
1035
1036 return INFIX_SUCCESS;
1037}

1039/*
1040 * Implementation for infix_type_from_signature.
1041 * This sets up a dedicated arena and parser state, invokes the core `parse_type`
1042 * function, and ensures the entire signature string was consumed.
1043 */
1044c23_nodiscard infix_status infix_type_from_signature(infix_type ** out_type,
1045 infix_arena_t ** out_arena,
1046 const char * signature) {
1047 if (!out_type || !out_arena || !signature || *signature == '\0')
1048 return INFIX_ERROR_INVALID_ARGUMENT;
1049
1050 // Each parse operation creates its own arena for the resulting type graph.
1051 *out_arena = infix_arena_create(4096);
1052 if (!*out_arena)
1053 return INFIX_ERROR_ALLOCATION_FAILED;
1054
1055 // Initialize the parser state.
1056 parser_state state = {.p = signature, .arena = *out_arena, .depth = 0, .last_error = INFIX_SUCCESS};
1057 infix_type * type = parse_type(&state);
1058
1059 if (type) {
1060 skip_whitespace(&state);
1061 // A successful parse of a single type should consume the entire string.
1062 // If there are trailing characters, the signature is considered invalid.
1063 if (state.p[0] != '\0') {
1064 type = nullptr;
1065 state.last_error = INFIX_ERROR_INVALID_ARGUMENT;
1066 }
1067 }
1068
1069 // On any failure, we must destroy the arena to prevent memory leaks.
1070 if (!type) {
1071 infix_arena_destroy(*out_arena);
1072 *out_arena = nullptr;
1073 *out_type = nullptr;
1074 return state.last_error != INFIX_SUCCESS ? state.last_error : INFIX_ERROR_INVALID_ARGUMENT;
1075 }
1076
1077 *out_type = type;
1078 return INFIX_SUCCESS;
1079}

1123/*
1124 * Implementation for infix_forward_create.
1125 * This is the primary high-level API. It orchestrates the process:
1126 * 1. Parse the signature string into a temporary type graph.
1127 * 2. Bridge the parsed `infix_function_argument` array to an `infix_type*` array.
1128 * 3. Call the internal trampoline creation logic, which deep-copies the types.
1129 * 4. Clean up the temporary arena used for parsing.
1130 */
1131c23_nodiscard infix_status infix_forward_create(infix_forward_t ** out_trampoline, const char * signature) {
1132 infix_arena_t * arena = nullptr;
1133 infix_type * ret_type = nullptr;
1134 infix_function_argument * args = nullptr;
1135 size_t num_args, num_fixed;
1136
1137 // First, parse the signature string into a type graph.
1138 infix_status status = infix_signature_parse(signature, &arena, &ret_type, &args, &num_args, &num_fixed);
1139 if (status != INFIX_SUCCESS)
1140 return status;
1141
1142 // Bridge: The manual API expects an array of `infix_type*`. We must extract
1143 // the type pointers from our parsed arguments to build this temporary array.
1144 infix_type ** arg_types =
1145 (num_args > 0) ? infix_arena_alloc(arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *)) : nullptr;
1146 if (num_args > 0 && !arg_types) {
1147 infix_arena_destroy(arena);
1148 return INFIX_ERROR_ALLOCATION_FAILED;
1149 }
1150 for (size_t i = 0; i < num_args; ++i)
1151 arg_types[i] = args[i].type;
1152
1153 // Instead of calling the public manual API, we now need
1154 // a way to tell the creation function about our temporary arena so it can
1155 // calculate the exact memory needed. Let's assume an internal function
1156 // _infix_forward_create_from_parsed_arena exists for this.
1157 status = _infix_forward_create_internal(out_trampoline, ret_type, arg_types, num_args, num_fixed, arena);
1158
1159 // The temporary arena has now been measured and its types copied. It can be safely destroyed.
1160 infix_arena_destroy(arena);
1161 return status;
1162}
