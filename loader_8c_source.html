52c23_nodiscard infix_library_t * infix_library_open(const char * path) {
53 if (path == nullptr)
54 return nullptr;
55
56 // Allocate memory for our opaque wrapper struct.
57 infix_library_t * lib = infix_malloc(sizeof(infix_library_t));
58 if (lib == nullptr) {
59 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
60 return nullptr;
61 }
62
63#if defined(INFIX_OS_WINDOWS)
64 // On Windows, use LoadLibraryA to load the DLL.
65 // 'A' specifies the ANSI version, which matches the `const char*` input.
66 lib->handle = LoadLibraryA(path);
67#else
68 // On POSIX systems, use dlopen.
69 // - RTLD_LAZY: Resolves symbols only when they are first used (lazy binding),
70 // which can improve startup performance.
71 // - RTLD_GLOBAL: Makes symbols from this library available for resolution
72 // by subsequently loaded libraries. This is important for handling
73 // complex dependencies between shared objects.
74 lib->handle = dlopen(path, RTLD_LAZY | RTLD_GLOBAL);
75#endif
76
77 // Both LoadLibraryA and dlopen return NULL on failure.
78 if (lib->handle == nullptr) {
79#if defined(INFIX_OS_WINDOWS)
80 _infix_set_system_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_LIBRARY_LOAD_FAILED, GetLastError(), nullptr);
81#else
82 _infix_set_system_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_LIBRARY_LOAD_FAILED, 0, dlerror());
83#endif
84 // If the OS call failed, we must free the wrapper struct we allocated
85 // to prevent a memory leak.
86 infix_free(lib);
87 return nullptr;
88 }
89 return lib;
90}

100void infix_library_close(infix_library_t * lib) {
101 if (lib == nullptr)
102 return;
103
104 // Only attempt to unload if the native handle is valid.
105 if (lib->handle) {
106#if defined(INFIX_OS_WINDOWS)
107 // On Windows, FreeLibrary decrements the library's reference count.
108 // The DLL is unloaded when its reference count reaches zero.
109 FreeLibrary((HMODULE)lib->handle);
110#else
111 // On POSIX, dlclose does the same.
112 dlclose(lib->handle);
113#endif
114 }
115 // Finally, free our wrapper struct.
116 infix_free(lib);
117}

131c23_nodiscard void * infix_library_get_symbol(infix_library_t * lib, const char * symbol_name) {
132 if (lib == nullptr || lib->handle == nullptr || symbol_name == nullptr)
133 return nullptr;
134
135#if defined(INFIX_OS_WINDOWS)
136 // On Windows, use GetProcAddress. The return type is technically FARPROC,
137 // which must be cast to a generic `void*` for our cross-platform API.
138 return (void *)GetProcAddress((HMODULE)lib->handle, symbol_name);
139#else
140 // On POSIX, use dlsym. It directly returns a `void*`.
141 return dlsym(lib->handle, symbol_name);
142#endif
143}

157c23_nodiscard infix_status infix_read_global(infix_library_t * lib,
158 const char * symbol_name,
159 const char * type_signature,
160 void * buffer) {
161 if (buffer == nullptr)
162 return INFIX_ERROR_INVALID_ARGUMENT;
163
164 // Find the address of the global variable in the library.
165 void * symbol_addr = infix_library_get_symbol(lib, symbol_name);
166 if (symbol_addr == nullptr) {
167 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_SYMBOL_NOT_FOUND, 0);
168 return INFIX_ERROR_INVALID_ARGUMENT;
169 }
170
171 // Parse the type signature to determine the size of the variable.
172 // This creates a temporary arena to hold the type information.
173 infix_type * type = nullptr;
174 infix_arena_t * arena = nullptr;
175 infix_status status = infix_type_from_signature(&type, &arena, type_signature, nullptr);
176
177 if (status != INFIX_SUCCESS)
178 return status;
179
180 // Perform a memory copy of the correct size.
181 infix_memcpy(buffer, symbol_addr, type->size);
182
183 // Clean up the temporary arena used for parsing.
184 infix_arena_destroy(arena);
185 return INFIX_SUCCESS;
186}
