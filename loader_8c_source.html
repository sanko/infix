50c23_nodiscard infix_library_t * infix_library_open(const char * path) {
51 if (path == nullptr)
52 return nullptr;
53
54 infix_library_t * lib = infix_calloc(1, sizeof(infix_library_t));
55 if (lib == nullptr) {
56 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
57 return nullptr;
58 }
59
60#if defined(INFIX_OS_WINDOWS)
61 lib->handle = LoadLibraryA(path);
62#else
63 // Use RTLD_LAZY for performance (resolve symbols only when they are first used).
64 // Use RTLD_GLOBAL to make symbols from this library available for resolution
65 // by other libraries that might be loaded later. This is important for complex
66 // dependency chains.
67 lib->handle = dlopen(path, RTLD_LAZY | RTLD_GLOBAL);
68#endif
69
70 if (lib->handle == nullptr) {
71#if defined(INFIX_OS_WINDOWS)
72 // On Windows, GetLastError() provides the specific error code.
73 _infix_set_system_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_LIBRARY_LOAD_FAILED, GetLastError(), nullptr);
74#else
75 // On POSIX, dlerror() returns a human-readable string.
76 _infix_set_system_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_LIBRARY_LOAD_FAILED, 0, dlerror());
77#endif
78 infix_free(lib);
79 return nullptr;
80 }
81 return lib;
82}

148c23_nodiscard infix_status infix_read_global(infix_library_t * lib,
149 const char * symbol_name,
150 const char * type_signature,
151 void * buffer) {
152 if (buffer == nullptr)
153 return INFIX_ERROR_INVALID_ARGUMENT;
154
155 void * symbol_addr = infix_library_get_symbol(lib, symbol_name);
156 if (symbol_addr == nullptr) {
157 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_SYMBOL_NOT_FOUND, 0);
158 return INFIX_ERROR_INVALID_ARGUMENT;
159 }
160
161 // Parse the signature to get the type's size.
162 infix_type * type = nullptr;
163 infix_arena_t * arena = nullptr;
164 infix_status status = infix_type_from_signature(&type, &arena, type_signature, nullptr);
165 if (status != INFIX_SUCCESS)
166 return status;
167
168 // Safely copy the data using the parsed size.
169 infix_memcpy(buffer, symbol_addr, type->size);
170
171 infix_arena_destroy(arena);
172 return INFIX_SUCCESS;
173}
