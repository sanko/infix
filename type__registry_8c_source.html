79static uint64_t _registry_hash_string(const char * str) {
80 uint64_t hash = 5381;
81 int c;
82 while ((c = *str++))
83 hash = ((hash << 5) + hash) + c; // hash * 33 + c
84 return hash;
85}

94static _infix_registry_entry_t * _registry_lookup(infix_registry_t * registry, const char * name) {
95 if (!registry || !name)
96 return nullptr;
97 size_t index = _registry_hash_string(name) % registry->num_buckets;
98 // Traverse the linked list (chain) at the computed bucket index.
99 for (_infix_registry_entry_t * current = registry->buckets[index]; current; current = current->next)
100 if (strcmp(current->name, name) == 0)
101 return current;
102 return nullptr;
103}

118static _infix_registry_entry_t * _registry_insert(infix_registry_t * registry, const char * name) {
119 size_t index = _registry_hash_string(name) % registry->num_buckets;
120 _infix_registry_entry_t * new_entry =
121 infix_arena_alloc(registry->arena, sizeof(_infix_registry_entry_t), _Alignof(_infix_registry_entry_t));
122 if (!new_entry) {
123 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
124 return nullptr;
125 }
126 size_t name_len = strlen(name) + 1;
127 char * name_copy = infix_arena_alloc(registry->arena, name_len, 1);
128 if (!name_copy) {
129 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
130 return nullptr;
131 }
132 infix_memcpy(name_copy, name, name_len);
133 new_entry->name = name_copy;
134 new_entry->type = nullptr;
135 new_entry->is_forward_declaration = false;
136 // Prepend to the linked list in the bucket.
137 new_entry->next = registry->buckets[index];
138 registry->buckets[index] = new_entry;
139 registry->num_items++;
140 return new_entry;
141}

156c23_nodiscard infix_registry_t * infix_registry_create(void) {
157 _infix_clear_error();
158 infix_registry_t * registry = infix_malloc(sizeof(infix_registry_t));
159 if (!registry) {
160 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
161 return nullptr;
162 }
163 registry->arena = infix_arena_create(16384);
164 if (!registry->arena) {
165 infix_free(registry);
166 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
167 return nullptr;
168 }
169 registry->num_buckets = INITIAL_REGISTRY_BUCKETS;
170 registry->buckets = infix_arena_calloc(
171 registry->arena, registry->num_buckets, sizeof(_infix_registry_entry_t *), _Alignof(_infix_registry_entry_t *));
172 if (!registry->buckets) {
173 infix_arena_destroy(registry->arena);
174 infix_free(registry);
175 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
176 return nullptr;
177 }
178 registry->num_items = 0;
179 return registry;
180}

218static infix_status _resolve_type_graph_inplace_recursive(infix_arena_t * temp_arena,
219 infix_type ** type_ptr,
220 infix_registry_t * registry,
221 resolve_memo_node_t ** memo_head) {
222 if (!type_ptr || !*type_ptr || !(*type_ptr)->is_arena_allocated)
223 return INFIX_SUCCESS;
224
225 infix_type * type = *type_ptr;
226
227 // Cycle detection: If we've seen this node before, we're in a cycle.
228 // Return success to break the loop.
229 for (resolve_memo_node_t * node = *memo_head; node != nullptr; node = node->next)
230 if (node->src == type)
231 return INFIX_SUCCESS;
232
233 // Allocate the memoization node from the stable temporary arena.
234 resolve_memo_node_t * memo_node =
235 infix_arena_alloc(temp_arena, sizeof(resolve_memo_node_t), _Alignof(resolve_memo_node_t));
236 if (!memo_node) {
237 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
238 return INFIX_ERROR_ALLOCATION_FAILED;
239 }
240 memo_node->src = type;
241 memo_node->next = *memo_head;
242 *memo_head = memo_node;
243
244 if (type->category == INFIX_TYPE_NAMED_REFERENCE) {
245 if (!registry) {
246 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
247 return INFIX_ERROR_INVALID_ARGUMENT;
248 }
249 const char * name = type->meta.named_reference.name;
250 _infix_registry_entry_t * entry = _registry_lookup(registry, name);
251 if (!entry || !entry->type) {
252 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
253 return INFIX_ERROR_INVALID_ARGUMENT;
254 }
255 *type_ptr = entry->type;
256 return INFIX_SUCCESS;
257 }
258
259 infix_status status = INFIX_SUCCESS;
260 switch (type->category) {
261 case INFIX_TYPE_POINTER:
262 status = _resolve_type_graph_inplace_recursive(
263 temp_arena, &type->meta.pointer_info.pointee_type, registry, memo_head);
264 break;
265 case INFIX_TYPE_ARRAY:
266 status =
267 _resolve_type_graph_inplace_recursive(temp_arena, &type->meta.array_info.element_type, registry, memo_head);
268 break;
269 case INFIX_TYPE_STRUCT:
270 case INFIX_TYPE_UNION:
271 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
272 status = _resolve_type_graph_inplace_recursive(
273 temp_arena, &type->meta.aggregate_info.members[i].type, registry, memo_head);
274 if (status != INFIX_SUCCESS)
275 break;
276 }
277 break;
278 case INFIX_TYPE_REVERSE_TRAMPOLINE:
279 status = _resolve_type_graph_inplace_recursive(
280 temp_arena, &type->meta.func_ptr_info.return_type, registry, memo_head);
281 if (status != INFIX_SUCCESS)
282 break;
283 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i) {
284 status = _resolve_type_graph_inplace_recursive(
285 temp_arena, &type->meta.func_ptr_info.args[i].type, registry, memo_head);
286 if (status != INFIX_SUCCESS)
287 break;
288 }
289 break;
290 case INFIX_TYPE_ENUM:
291 status = _resolve_type_graph_inplace_recursive(
292 temp_arena, &type->meta.enum_info.underlying_type, registry, memo_head);
293 break;
294 case INFIX_TYPE_COMPLEX:
295 status =
296 _resolve_type_graph_inplace_recursive(temp_arena, &type->meta.complex_info.base_type, registry, memo_head);
297 break;
298 case INFIX_TYPE_VECTOR:
299 status = _resolve_type_graph_inplace_recursive(
300 temp_arena, &type->meta.vector_info.element_type, registry, memo_head);
301 break;
302 default:
303 break;
304 }
305
306 return status;
307}

351static void _registry_parser_skip_whitespace(_registry_parser_state_t * state) {
352 while (1) {
353 while (isspace((unsigned char)*state->p))
354 state->p++;
355 if (*state->p == '#') // Skip comments
356 while (*state->p != '\n' && *state->p != '\0')
357 state->p++;
358 else
359 break;
360 }
361}

371static char * _registry_parser_parse_name(_registry_parser_state_t * state, char * buffer, size_t buf_size) {
372 _registry_parser_skip_whitespace(state);
373 const char * name_start = state->p;
374 while (isalnum((unsigned char)*state->p) || *state->p == '_' || *state->p == ':') {
375 if (*state->p == ':' && state->p[1] != ':')
376 break; // Handle single colon as non-identifier char.
377 if (*state->p == ':')
378 state->p++; // Skip the first ':' of '::'
379 state->p++;
380 }
381 size_t len = state->p - name_start;
382 if (len == 0 || len >= buf_size)
383 return nullptr;
384 infix_memcpy(buffer, name_start, len);
385 buffer[len] = '\0';
386 return buffer;
387}

418c23_nodiscard infix_status infix_register_types(infix_registry_t * registry, const char * definitions) {
419 _infix_clear_error();
420 if (!registry || !definitions) {
421 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_UNKNOWN, 0);
422 return INFIX_ERROR_INVALID_ARGUMENT;
423 }
424
425 _registry_parser_state_t state = {.p = definitions, .start = definitions};
426 g_infix_last_signature_context = definitions;
427
428 // A temporary structure to hold information about each definition found in Pass 1.
429 struct def_info {
430 _infix_registry_entry_t * entry;
431 const char * def_body_start;
432 size_t def_body_len;
433 };
434 struct def_info defs_found[256]; // Max 256 definitions in a single call.
435 size_t num_defs_found = 0;
436
437 // Pass 1: Scan & Index all names and their definition bodies.
438 while (*state.p != '\0') {
439 _registry_parser_skip_whitespace(&state);
440 if (*state.p == '\0')
441 break;
442
443 if (*state.p != '@') {
444 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
445 return INFIX_ERROR_INVALID_ARGUMENT;
446 }
447 state.p++;
448
449 char name_buffer[256];
450 if (!_registry_parser_parse_name(&state, name_buffer, sizeof(name_buffer))) {
451 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
452 return INFIX_ERROR_INVALID_ARGUMENT;
453 }
454 _infix_registry_entry_t * entry = _registry_lookup(registry, name_buffer);
455 _registry_parser_skip_whitespace(&state);
456
457 if (*state.p == '=') { // This is a full definition.
458 state.p++;
459 _registry_parser_skip_whitespace(&state);
460 // It's an error to redefine a type that wasn't a forward declaration.
461 if (entry && !entry->is_forward_declaration) {
462 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
463 return INFIX_ERROR_INVALID_ARGUMENT;
464 }
465 if (!entry) { // If it doesn't exist, create it.
466 entry = _registry_insert(registry, name_buffer);
467 if (!entry)
468 return INFIX_ERROR_ALLOCATION_FAILED;
469 }
470 if (num_defs_found >= 256) {
471 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_RECURSION_DEPTH_EXCEEDED, state.p - state.start);
472 return INFIX_ERROR_INVALID_ARGUMENT;
473 }
474
475 // Find the end of the type definition body (the next top-level ';').
476 defs_found[num_defs_found].entry = entry;
477 defs_found[num_defs_found].def_body_start = state.p;
478 int nest_level = 0;
479 const char * body_end = state.p;
480 while (*body_end != '\0' && !(*body_end == ';' && nest_level == 0)) {
481 if (*body_end == '{' || *body_end == '<' || *body_end == '(' || *body_end == '[')
482 nest_level++;
483 if (*body_end == '}' || *body_end == '>' || *body_end == ')' || *body_end == ']')
484 nest_level--;
485 body_end++;
486 }
487 defs_found[num_defs_found].def_body_len = body_end - state.p;
488 state.p = body_end;
489 num_defs_found++;
490 }
491 else if (*state.p == ';') { // This is a forward declaration.
492 if (entry) { // Cannot forward-declare an already-defined type.
493 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
494 return INFIX_ERROR_INVALID_ARGUMENT;
495 }
496 entry = _registry_insert(registry, name_buffer);
497 if (!entry)
498 return INFIX_ERROR_ALLOCATION_FAILED;
499 entry->is_forward_declaration = true;
500 }
501 else {
502 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
503 return INFIX_ERROR_INVALID_ARGUMENT;
504 }
505
506 if (*state.p == ';')
507 state.p++;
508 }
509
510 // Pass 2: Parse the bodies of all found definitions into the registry.
511 for (size_t i = 0; i < num_defs_found; ++i) {
512 _infix_registry_entry_t * entry = defs_found[i].entry;
513 // Make a temporary, null-terminated copy of the definition body substring.
514 char * body_copy = infix_malloc(defs_found[i].def_body_len + 1);
515 if (!body_copy) {
516 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
517 return INFIX_ERROR_ALLOCATION_FAILED;
518 }
519 infix_memcpy(body_copy, defs_found[i].def_body_start, defs_found[i].def_body_len);
520 body_copy[defs_found[i].def_body_len] = '\0';
521
522 // "Parse" step: parse into a temporary arena.
523 infix_type * raw_type = nullptr;
524 infix_arena_t * parser_arena = nullptr;
525 infix_status status = _infix_parse_type_internal(&raw_type, &parser_arena, body_copy);
526 infix_free(body_copy);
527 if (status != INFIX_SUCCESS) {
528 // Adjust the error position to be relative to the full definition string.
529 infix_error_details_t err = infix_get_last_error();
530 _infix_set_error(err.category, err.code, (defs_found[i].def_body_start - definitions) + err.position);
531 return INFIX_ERROR_INVALID_ARGUMENT;
532 }
533
534 // "Copy" step: copy from temp arena to the registry's permanent arena.
535 entry->type = _copy_type_graph_to_arena(registry->arena, raw_type);
536 infix_arena_destroy(parser_arena);
537 if (!entry->type) {
538 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
539 return INFIX_ERROR_ALLOCATION_FAILED;
540 }
541
542 // Annotate the type with its own name for printing and introspection.
543 if (entry->type->category == INFIX_TYPE_STRUCT || entry->type->category == INFIX_TYPE_UNION)
544 entry->type->meta.aggregate_info.name = entry->name;
545 entry->is_forward_declaration = false;
546 }
547
548 // Pass 3: Resolve and layout all the newly defined types.
549 for (size_t i = 0; i < num_defs_found; ++i) {
550 _infix_registry_entry_t * entry = defs_found[i].entry;
551 if (entry->type) {
552 // "Resolve" and "Layout" steps.
553 if (_infix_resolve_type_graph_inplace(&entry->type, registry) != INFIX_SUCCESS)
554 return INFIX_ERROR_INVALID_ARGUMENT;
555 _infix_type_recalculate_layout(entry->type);
556 }
557 }
558 return INFIX_SUCCESS;
559}

569c23_nodiscard bool infix_registry_iterator_next(infix_registry_iterator_t * iter) {
570 if (!iter || !iter->registry)
571 return false;
572
573
574 const _infix_registry_entry_t * entry = iter->current_entry;
575
576 // If we have a current entry, start from the next one in the chain.
577 if (iter->current_entry)
578 entry = entry->next;
579
580 // Otherwise, if we are starting, begin with the head of the current bucket.
581 else if (iter->current_bucket < iter->registry->num_buckets)
582 entry = iter->registry->buckets[iter->current_bucket];
583
584 while (true) {
585 // Traverse the current chain looking for a valid entry.
586 while (entry) {
587 if (entry->type && !entry->is_forward_declaration) {
588 // Found one. Update the iterator and return successfully.
589 iter->current_entry = entry;
590 return true;
591 }
592 entry = entry->next;
593 }
594
595 // If we're here, the current chain is exhausted. Move to the next bucket.
596 iter->current_bucket++;
597
598 // If there are no more buckets, we're done.
599 if (iter->current_bucket >= iter->registry->num_buckets) {
600 iter->current_entry = nullptr;
601 return false;
602 }
603
604 // Start the search from the head of the new bucket.
605 entry = iter->registry->buckets[iter->current_bucket];
606 }
