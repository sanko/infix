79static uint64_t _registry_hash_string(const char * str) {
80 uint64_t hash = 5381;
81 int c;
82 while ((c = *str++))
83 hash = ((hash << 5) + hash) + c; // hash * 33 + c
84 return hash;
85}

94static _infix_registry_entry_t * _registry_lookup(infix_registry_t * registry, const char * name) {
95 if (!registry || !name)
96 return nullptr;
97 size_t index = _registry_hash_string(name) % registry->num_buckets;
98 // Traverse the linked list (chain) at the computed bucket index.
99 for (_infix_registry_entry_t * current = registry->buckets[index]; current; current = current->next) {
100 if (strcmp(current->name, name) == 0)
101 return current;
102 }
103 return nullptr;
104}

119static _infix_registry_entry_t * _registry_insert(infix_registry_t * registry, const char * name) {
120 size_t index = _registry_hash_string(name) % registry->num_buckets;
121 _infix_registry_entry_t * new_entry =
122 infix_arena_alloc(registry->arena, sizeof(_infix_registry_entry_t), _Alignof(_infix_registry_entry_t));
123 if (!new_entry)
124 return nullptr;
125 // The name string must be copied into the registry's arena to ensure its lifetime.
126 size_t name_len = strlen(name) + 1;
127 char * name_copy = infix_arena_alloc(registry->arena, name_len, 1);
128 if (!name_copy)
129 return nullptr;
130 infix_memcpy(name_copy, name, name_len);
131 new_entry->name = name_copy;
132 new_entry->type = nullptr;
133 new_entry->is_forward_declaration = false;
134 // Prepend to the linked list in the bucket.
135 new_entry->next = registry->buckets[index];
136 registry->buckets[index] = new_entry;
137 registry->num_items++;
138 return new_entry;
139}

154c23_nodiscard infix_registry_t * infix_registry_create(void) {
155 _infix_clear_error();
156 infix_registry_t * registry = infix_malloc(sizeof(infix_registry_t));
157 if (!registry) {
158 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
159 return nullptr;
160 }
161 registry->arena = infix_arena_create(16384);
162 if (!registry->arena) {
163 infix_free(registry);
164 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
165 return nullptr;
166 }
167 registry->num_buckets = INITIAL_REGISTRY_BUCKETS;
168 registry->buckets = infix_arena_calloc(
169 registry->arena, registry->num_buckets, sizeof(_infix_registry_entry_t *), _Alignof(_infix_registry_entry_t *));
170 if (!registry->buckets) {
171 infix_arena_destroy(registry->arena);
172 infix_free(registry);
173 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
174 return nullptr;
175 }
176 registry->num_items = 0;
177 return registry;
178}

216static infix_status _resolve_type_graph_inplace_recursive(infix_type ** type_ptr,
217 infix_registry_t * registry,
218 resolve_memo_node_t ** memo_head) {
219 if (!type_ptr || !*type_ptr || !(*type_ptr)->is_arena_allocated)
220 return INFIX_SUCCESS; // Base case: Don't touch static or null types.
221
222 infix_type * type = *type_ptr;
223
224 // Cycle detection: If we've seen this node before, we're in a cycle.
225 // Return success to break the loop.
226 for (resolve_memo_node_t * node = *memo_head; node != NULL; node = node->next) {
227 if (node->src == type)
228 return INFIX_SUCCESS;
229 }
230 // Add this node to the visited list before recursing.
231 resolve_memo_node_t memo_node = {.src = type, .next = *memo_head};
232 *memo_head = &memo_node;
233
234 // Base case: Resolve a named reference.
235 if (type->category == INFIX_TYPE_NAMED_REFERENCE) {
236 if (!registry) {
237 *memo_head = memo_node.next; // Pop from list before returning.
238 return INFIX_ERROR_INVALID_ARGUMENT; // Cannot resolve without a registry.
239 }
240 const char * name = type->meta.named_reference.name;
241 _infix_registry_entry_t * entry = _registry_lookup(registry, name);
242 // Fail if the name is not in the registry or if it's an unresolved forward declaration.
243 if (!entry || !entry->type) {
244 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
245 *memo_head = memo_node.next;
246 return INFIX_ERROR_INVALID_ARGUMENT;
247 }
248 // This is the critical step: replace the pointer to the named reference
249 // with a direct pointer to the canonical type from the registry.
250 *type_ptr = entry->type;
251 *memo_head = memo_node.next;
252 return INFIX_SUCCESS;
253 }
254
255 // Recursive step: Recurse into all child types.
256 infix_status status = INFIX_SUCCESS;
257 switch (type->category) {
258 case INFIX_TYPE_POINTER:
259 status = _resolve_type_graph_inplace_recursive(&type->meta.pointer_info.pointee_type, registry, memo_head);
260 break;
261 case INFIX_TYPE_ARRAY:
262 status = _resolve_type_graph_inplace_recursive(&type->meta.array_info.element_type, registry, memo_head);
263 break;
264 case INFIX_TYPE_STRUCT:
265 case INFIX_TYPE_UNION:
266 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
267 status =
268 _resolve_type_graph_inplace_recursive(&type->meta.aggregate_info.members[i].type, registry, memo_head);
269 if (status != INFIX_SUCCESS)
270 break;
271 }
272 break;
273 case INFIX_TYPE_REVERSE_TRAMPOLINE:
274 status = _resolve_type_graph_inplace_recursive(&type->meta.func_ptr_info.return_type, registry, memo_head);
275 if (status != INFIX_SUCCESS)
276 break;
277 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i) {
278 status = _resolve_type_graph_inplace_recursive(&type->meta.func_ptr_info.args[i].type, registry, memo_head);
279 if (status != INFIX_SUCCESS)
280 break;
281 }
282 break;
283 case INFIX_TYPE_ENUM:
284 status = _resolve_type_graph_inplace_recursive(&type->meta.enum_info.underlying_type, registry, memo_head);
285 break;
286 case INFIX_TYPE_COMPLEX:
287 status = _resolve_type_graph_inplace_recursive(&type->meta.complex_info.base_type, registry, memo_head);
288 break;
289 case INFIX_TYPE_VECTOR:
290 status = _resolve_type_graph_inplace_recursive(&type->meta.vector_info.element_type, registry, memo_head);
291 break;
292 default:
293 break;
294 }
295
296 *memo_head = memo_node.next; // Pop from visited list before returning.
297 return status;
298}

328static void _registry_parser_skip_whitespace(_registry_parser_state_t * state) {
329 while (1) {
330 while (isspace((unsigned char)*state->p))
331 state->p++;
332 if (*state->p == '#') { // Skip comments
333 while (*state->p != '\n' && *state->p != '\0')
334 state->p++;
335 }
336 else
337 break;
338 }
339}

342static char * _registry_parser_parse_name(_registry_parser_state_t * state, char * buffer, size_t buf_size) {
343 _registry_parser_skip_whitespace(state);
344 const char * name_start = state->p;
345 while (isalnum((unsigned char)*state->p) || *state->p == '_' || *state->p == ':') {
346 if (*state->p == ':' && state->p[1] != ':')
347 break; // Handle single colon as non-identifier char.
348 if (*state->p == ':')
349 state->p++; // Skip the first ':' of '::'
350 state->p++;
351 }
352 size_t len = state->p - name_start;
353 if (len == 0 || len >= buf_size)
354 return nullptr;
355 infix_memcpy(buffer, name_start, len);
356 buffer[len] = '\0';
357 return buffer;
358}

389c23_nodiscard infix_status infix_register_types(infix_registry_t * registry, const char * definitions) {
390 _infix_clear_error();
391 if (!registry || !definitions)
392 return INFIX_ERROR_INVALID_ARGUMENT;
393
394 _registry_parser_state_t state = {.p = definitions, .start = definitions};
395 g_infix_last_signature_context = definitions;
396
397 // A temporary structure to hold information about each definition found in Pass 1.
398 struct def_info {
399 _infix_registry_entry_t * entry;
400 const char * def_body_start;
401 size_t def_body_len;
402 };
403 struct def_info defs_found[256]; // Max 256 definitions in a single call.
404 size_t num_defs_found = 0;
405
406 // Pass 1: Scan & Index all names and their definition bodies.
407 while (*state.p != '\0') {
408 _registry_parser_skip_whitespace(&state);
409 if (*state.p == '\0')
410 break;
411
412 if (*state.p != '@') {
413 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
414 return INFIX_ERROR_INVALID_ARGUMENT;
415 }
416 state.p++;
417
418 char name_buffer[256];
419 if (!_registry_parser_parse_name(&state, name_buffer, sizeof(name_buffer))) {
420 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
421 return INFIX_ERROR_INVALID_ARGUMENT;
422 }
423 _infix_registry_entry_t * entry = _registry_lookup(registry, name_buffer);
424 _registry_parser_skip_whitespace(&state);
425
426 if (*state.p == '=') { // This is a full definition.
427 state.p++;
428 _registry_parser_skip_whitespace(&state);
429 // It's an error to redefine a type that wasn't a forward declaration.
430 if (entry && !entry->is_forward_declaration) {
431 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
432 return INFIX_ERROR_INVALID_ARGUMENT;
433 }
434 if (!entry) { // If it doesn't exist, create it.
435 entry = _registry_insert(registry, name_buffer);
436 if (!entry)
437 return INFIX_ERROR_ALLOCATION_FAILED;
438 }
439 if (num_defs_found >= 256)
440 return INFIX_ERROR_INVALID_ARGUMENT; // Too many definitions.
441
442 // Find the end of the type definition body (the next top-level ';').
443 defs_found[num_defs_found].entry = entry;
444 defs_found[num_defs_found].def_body_start = state.p;
445 int nest_level = 0;
446 const char * body_end = state.p;
447 while (*body_end != '\0' && !(*body_end == ';' && nest_level == 0)) {
448 if (*body_end == '{' || *body_end == '<' || *body_end == '(' || *body_end == '[')
449 nest_level++;
450 if (*body_end == '}' || *body_end == '>' || *body_end == ')' || *body_end == ']')
451 nest_level--;
452 body_end++;
453 }
454 defs_found[num_defs_found].def_body_len = body_end - state.p;
455 state.p = body_end;
456 num_defs_found++;
457 }
458 else if (*state.p == ';') { // This is a forward declaration.
459 if (entry) { // Cannot forward-declare an already-defined type.
460 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
461 return INFIX_ERROR_INVALID_ARGUMENT;
462 }
463 entry = _registry_insert(registry, name_buffer);
464 if (!entry)
465 return INFIX_ERROR_ALLOCATION_FAILED;
466 entry->is_forward_declaration = true;
467 }
468 else {
469 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, state.p - state.start);
470 return INFIX_ERROR_INVALID_ARGUMENT;
471 }
472
473 if (*state.p == ';')
474 state.p++;
475 }
476
477 // Pass 2: Parse the bodies of all found definitions into the registry.
478 for (size_t i = 0; i < num_defs_found; ++i) {
479 _infix_registry_entry_t * entry = defs_found[i].entry;
480 // Make a temporary, null-terminated copy of the definition body substring.
481 char * body_copy = infix_malloc(defs_found[i].def_body_len + 1);
482 if (!body_copy)
483 return INFIX_ERROR_ALLOCATION_FAILED;
484 infix_memcpy(body_copy, defs_found[i].def_body_start, defs_found[i].def_body_len);
485 body_copy[defs_found[i].def_body_len] = '\0';
486
487 // "Parse" step: parse into a temporary arena.
488 infix_type * raw_type = nullptr;
489 infix_arena_t * parser_arena = nullptr;
490 infix_status status = _infix_parse_type_internal(&raw_type, &parser_arena, body_copy);
491 infix_free(body_copy);
492 if (status != INFIX_SUCCESS) {
493 infix_arena_destroy(parser_arena);
494 // Adjust the error position to be relative to the full definition string.
495 infix_error_details_t err = infix_get_last_error();
496 _infix_set_error(err.category, err.code, (defs_found[i].def_body_start - definitions) + err.position);
497 return INFIX_ERROR_INVALID_ARGUMENT;
498 }
499
500 // "Copy" step: copy from temp arena to the registry's permanent arena.
501 entry->type = _copy_type_graph_to_arena(registry->arena, raw_type);
502 infix_arena_destroy(parser_arena); // Destroy temp arena.
503 if (!entry->type)
504 return INFIX_ERROR_ALLOCATION_FAILED;
505
506 // Annotate the type with its own name for printing and introspection.
507 if (entry->type->category == INFIX_TYPE_STRUCT || entry->type->category == INFIX_TYPE_UNION) {
508 entry->type->meta.aggregate_info.name = entry->name;
509 }
510 entry->is_forward_declaration = false;
511 }
512
513 // Pass 3: Resolve and layout all the newly defined types.
514 for (size_t i = 0; i < num_defs_found; ++i) {
515 _infix_registry_entry_t * entry = defs_found[i].entry;
516 if (entry->type) {
517 // "Resolve" and "Layout" steps.
518 if (_infix_resolve_type_graph_inplace(&entry->type, registry) != INFIX_SUCCESS)
519 return INFIX_ERROR_INVALID_ARGUMENT;
520 _infix_type_recalculate_layout(entry->type);
521 }
522 }
523 return INFIX_SUCCESS;
524}
