130 // Use pthread_once to ensure the dynamic loading happens exactly once, thread-safely.
131 static pthread_once_t init_once = PTHREAD_ONCE_INIT;
132 pthread_once(&init_once, initialize_macos_apis);

141 CFTypeRef value = nullptr;
142 if (key) {

145 g_macos_apis.CFRelease(key);
146 }

151 result = true;
152 g_macos_apis.CFRelease(value);
153 }
154 return result;
155}

173static int shm_open_anonymous() {
174 char shm_name[64];
175 uint64_t random_val = 0;
176 // Generate a sufficiently random name to avoid collisions if multiple processes
177 // are running this code simultaneously. Using /dev/urandom is a robust way to do this.
178 int rand_fd = open("/dev/urandom", O_RDONLY);
179 if (rand_fd < 0)
180 return -1;
181 ssize_t bytes_read = read(rand_fd, &random_val, sizeof(random_val));
182 close(rand_fd);
183 if (bytes_read != sizeof(random_val))
184 return -1;
185 snprintf(shm_name, sizeof(shm_name), "/infix-jit-%d-%llx", getpid(), (unsigned long long)random_val);
186 // Create the shared memory object exclusively.
187 int fd = shm_open(shm_name, O_RDWR | O_CREAT | O_EXCL, 0600);
188 if (fd >= 0) {
189 // Unlink immediately. The file descriptor remains valid, but the name is removed.
190 // This ensures the kernel will clean up the memory object when the last fd is closed.
191 shm_unlink(shm_name);
192 return fd;
193 }
194 return -1;
195}

205c23_nodiscard infix_executable_t infix_executable_alloc(size_t size) {
206#if defined(INFIX_OS_WINDOWS)
207 infix_executable_t exec = {.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0, .handle = nullptr};
208#else
209 infix_executable_t exec = {.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0, .shm_fd = -1};
210#endif
211 if (size == 0)
212 return exec;
213#if defined(INFIX_OS_WINDOWS)
214 // Windows: Single-mapping W^X. Allocate as RW, later change to RX via VirtualProtect.
215 void * code = VirtualAlloc(nullptr, size, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
216 if (code == nullptr) {
217 _infix_set_system_error(
218 INFIX_CATEGORY_ALLOCATION, INFIX_CODE_EXECUTABLE_MEMORY_FAILURE, GetLastError(), nullptr);
219 return exec;
220 }
221 exec.rw_ptr = code;
222 exec.rx_ptr = code;
223#elif defined(INFIX_OS_MACOS) || defined(INFIX_OS_ANDROID) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
224 // Single-mapping POSIX platforms. Allocate as RW, later change to RX via mprotect.
225 void * code = MAP_FAILED;
226#if defined(MAP_ANON)
227 int flags = MAP_PRIVATE | MAP_ANON;
228#if defined(INFIX_OS_MACOS)
229 // On macOS, we perform a one-time check for JIT support.
230 static bool g_use_secure_jit_path = false;
231 static bool g_checked_jit_support = false;
232 if (!g_checked_jit_support) {
233 g_use_secure_jit_path = has_jit_entitlement();
234 INFIX_DEBUG_PRINTF("macOS JIT check: Entitlement found = %s. Using %s API.",
235 g_use_secure_jit_path ? "yes" : "no",
236 g_use_secure_jit_path ? "secure (MAP_JIT)" : "legacy (mprotect)");
237 g_checked_jit_support = true;
238 }
239 // If entitled, use the modern, more secure MAP_JIT flag.
240 if (g_use_secure_jit_path)
241 flags |= MAP_JIT;
242#endif // INFIX_OS_MACOS
243 code = mmap(nullptr, size, PROT_READ | PROT_WRITE, flags, -1, 0);
244#endif // MAP_ANON
245 if (code == MAP_FAILED) { // Fallback for older systems without MAP_ANON
246 int fd = open("/dev/zero", O_RDWR);
247 if (fd != -1) {
248 code = mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_PRIVATE, fd, 0);
249 close(fd);
250 }
251 }
252 if (code == MAP_FAILED) {
253 _infix_set_system_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_EXECUTABLE_MEMORY_FAILURE, errno, nullptr);
254 return exec;
255 }
256 exec.rw_ptr = code;
257 exec.rx_ptr = code;
258#else
259 // Dual-mapping POSIX platforms (e.g., Linux, FreeBSD). Create two separate views of the same memory.
260 exec.shm_fd = shm_open_anonymous();
261 if (exec.shm_fd < 0) {
262 _infix_set_system_error(
263 INFIX_CATEGORY_ALLOCATION, INFIX_CODE_EXECUTABLE_MEMORY_FAILURE, errno, "shm_open failed");
264 return exec;
265 }
266 if (ftruncate(exec.shm_fd, size) != 0) {
267 _infix_set_system_error(
268 INFIX_CATEGORY_ALLOCATION, INFIX_CODE_EXECUTABLE_MEMORY_FAILURE, errno, "ftruncate failed");
269 close(exec.shm_fd);
270 exec.shm_fd = -1; // Ensure clean state
271 return exec;
272 }
273 // The RW mapping.
274 exec.rw_ptr = mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, exec.shm_fd, 0);
275 // The RX mapping of the exact same physical memory.
276 exec.rx_ptr = mmap(nullptr, size, PROT_READ | PROT_EXEC, MAP_SHARED, exec.shm_fd, 0);
277 // If either mapping fails, clean up both and return an error.
278 if (exec.rw_ptr == MAP_FAILED || exec.rx_ptr == MAP_FAILED) {
279 int err = errno; // Capture errno before cleanup
280 if (exec.rw_ptr != MAP_FAILED)
281 munmap(exec.rw_ptr, size);
282 if (exec.rx_ptr != MAP_FAILED)
283 munmap(exec.rx_ptr, size);
284 close(exec.shm_fd);
285 _infix_set_system_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_EXECUTABLE_MEMORY_FAILURE, err, "mmap failed");
286 return (infix_executable_t){.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0, .shm_fd = -1};
287 }
288
289 // The mmap mappings hold a reference to the shared memory object, so we don't
290 // need the FD anymore. Keeping it open consumes a file descriptor per trampoline,
291 // causing "shm_open failed" after ~1024 trampolines.
292 close(exec.shm_fd);
293 exec.shm_fd = -1;
294#endif
295 exec.size = size;
296 INFIX_DEBUG_PRINTF("Allocated JIT memory. RW at %p, RX at %p", exec.rw_ptr, exec.rx_ptr);
297 return exec;
298}
311void infix_executable_free(infix_executable_t exec) {
312 if (exec.size == 0)
313 return;
314#if defined(INFIX_OS_WINDOWS)
315 if (exec.rw_ptr) {
316 // Change protection to NOACCESS to catch use-after-free bugs immediately.
317 if (!VirtualProtect(exec.rw_ptr, exec.size, PAGE_NOACCESS, &(DWORD){0}))
318 INFIX_DEBUG_PRINTF("WARNING: VirtualProtect failed to set PAGE_NOACCESS guard page.");
319 VirtualFree(exec.rw_ptr, 0, MEM_RELEASE);
320 }
321#elif defined(INFIX_OS_MACOS)
322 // On macOS with MAP_JIT, the memory is managed with special thread-local permissions.
323 // We only need to unmap the single mapping.
324 if (exec.rw_ptr) {
325#if INFIX_MACOS_SECURE_JIT_AVAILABLE // This macro is not yet defined, placeholder for future
326 // If using the secure path, we should toggle write protection back on.
327 static bool g_use_secure_jit_path = false; // Re-check or use a shared flag
328 if (g_use_secure_jit_path)
329 pthread_jit_write_protect_np(true);
330#endif
331 // Creating a guard page before unmapping is good practice.
332 mprotect(exec.rw_ptr, exec.size, PROT_NONE);
333 munmap(exec.rw_ptr, exec.size);
334 }
335#elif defined(INFIX_OS_ANDROID) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
336 // Other single-mapping POSIX systems.
337 if (exec.rw_ptr) {
338 mprotect(exec.rw_ptr, exec.size, PROT_NONE);
339 munmap(exec.rw_ptr, exec.size);
340 }
341#else
342 // Dual-mapping POSIX: protect and unmap both views.
343 if (exec.rx_ptr)
344 mprotect(exec.rx_ptr, exec.size, PROT_NONE);
345 if (exec.rw_ptr)
346 munmap(exec.rw_ptr, exec.size);
347 if (exec.rx_ptr && exec.rx_ptr != exec.rw_ptr) // rw_ptr might be same as rx_ptr on some platforms
348 munmap(exec.rx_ptr, exec.size);
349 if (exec.shm_fd >= 0)
350 close(exec.shm_fd);
351#endif
352}
369c23_nodiscard bool infix_executable_make_executable(infix_executable_t exec) {
370 if (exec.rw_ptr == nullptr || exec.size == 0)
371 return false;
372 // On AArch64 (and other RISC architectures), the instruction and data caches can be
373 // separate. We must explicitly flush the D-cache (where the JIT wrote the code)
374 // and invalidate the I-cache so the CPU fetches the new instructions.
375#if defined(INFIX_ARCH_AARCH64)
376#if defined(_MSC_VER)
377 // Use the Windows-specific API.
378 FlushInstructionCache(GetCurrentProcess(), exec.rw_ptr, exec.size);
379#else
380 // Use the GCC/Clang built-in for other platforms.
381 __builtin___clear_cache((char *)exec.rw_ptr, (char *)exec.rw_ptr + exec.size);
382#endif
383#endif
384 bool result = false;
385#if defined(INFIX_OS_WINDOWS)
386 // Finalize permissions to Read+Execute.
387 result = VirtualProtect(exec.rw_ptr, exec.size, PAGE_EXECUTE_READ, &(DWORD){0});
388 if (!result)
389 _infix_set_system_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_PROTECTION_FAILURE, GetLastError(), nullptr);
390#elif defined(INFIX_OS_MACOS)
391#if INFIX_MACOS_SECURE_JIT_AVAILABLE // Placeholder
392 static bool g_use_secure_jit_path = false;
393 if (g_use_secure_jit_path) {
394 pthread_jit_write_protect_np(false); // Make writable region executable.
395 result = true;
396 }
397 else
398#endif
399 // On macOS with the JIT entitlement, we don't use mprotect. Instead, we toggle
400 // a thread-local "write permission" state for all JIT memory. The memory is
401 // RX by default, and we temporarily make it RW for writing.
402 // However, the current logic does this change via `pthread_jit_write_protect_np`
403 // within the allocator itself. For now, this is a placeholder for that logic.
404 result = (mprotect(exec.rw_ptr, exec.size, PROT_READ | PROT_EXEC) == 0);
405 if (!result)
406 _infix_set_system_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_PROTECTION_FAILURE, errno, nullptr);
407#elif defined(INFIX_OS_ANDROID) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
408 // Other single-mapping POSIX platforms use mprotect.
409 result = (mprotect(exec.rw_ptr, exec.size, PROT_READ | PROT_EXEC) == 0);
410 if (!result)
411 _infix_set_system_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_PROTECTION_FAILURE, errno, nullptr);
412#else
413 // On dual-mapping platforms, the RX mapping is already executable. This is a no-op.
414 result = true;
415#endif
416 if (result)
417 INFIX_DEBUG_PRINTF("Memory at %p is now executable.", exec.rx_ptr);
418 return result;
419}

525void infix_internal_dispatch_callback_fn_impl(infix_reverse_t * context, void * return_value_ptr, void ** args_array) {
526 INFIX_DEBUG_PRINTF(
527 "Dispatching reverse call. Context: %p, User Fn: %p", (void *)context, context->user_callback_fn);
528 if (context->user_callback_fn == nullptr) {
529 // If no handler is set, do nothing. If the function has a return value,
530 // it's good practice to zero it out to avoid returning garbage.
531 if (return_value_ptr && context->return_type->size > 0)
532 infix_memset(return_value_ptr, 0, context->return_type->size);
533 return;
534 }
535 if (context->cached_forward_trampoline != nullptr) {
536 // Path 1: Type-safe "callback". Use the pre-generated forward trampoline to
537 // call the user's C function with the correct signature. This is efficient
538 // and provides a clean interface for the C developer.
539 infix_cif_func cif_func = infix_forward_get_code(context->cached_forward_trampoline);
540 cif_func(return_value_ptr, args_array);
541 }
542 else {
543 // Path 2: Generic "closure". Directly call the user's generic handler.
544 // This path is more flexible and is intended for language bindings where the
545 // handler needs access to the context and raw argument pointers.
546 infix_closure_handler_fn handler = (infix_closure_handler_fn)context->user_callback_fn;
547 handler(context, return_value_ptr, args_array);
548 }
549 INFIX_DEBUG_PRINTF("Exiting reverse call dispatcher.");
550}
