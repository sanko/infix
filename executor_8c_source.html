145static bool has_jit_entitlement(void) {
146 // Use pthread_once to ensure the dynamic loading happens exactly once, thread-safely.
147 static pthread_once_t init_once = PTHREAD_ONCE_INIT;
148 pthread_once(&init_once, initialize_macos_apis);
149
150 if (!g_macos_apis.SecTaskCopyValueForEntitlement || !g_macos_apis.CFStringCreateWithCString)
151 return false;
152
153 bool result = false;
154
155 SecTaskRef task = g_macos_apis.SecTaskCreateFromSelf(g_macos_apis.kCFAllocatorDefault);
156 if (!task)
157 return false;
158
159 CFStringRef key = g_macos_apis.CFStringCreateWithCString(
160 g_macos_apis.kCFAllocatorDefault, "com.apple.security.cs.allow-jit", kCFStringEncodingUTF8);
161 CFTypeRef value = nullptr;
162 if (key) {
163 // This is the core check: ask the system for the value of the entitlement.
164 value = g_macos_apis.SecTaskCopyValueForEntitlement(task, key, nullptr);
165 g_macos_apis.CFRelease(key);
166 }
167 g_macos_apis.CFRelease(task);
168
169 if (value) {
170 // The value of the entitlement is a CFBoolean, so we must extract its value.
171 if (g_macos_apis.CFBooleanGetValue && g_macos_apis.CFBooleanGetValue(value))
172 result = true;
173 g_macos_apis.CFRelease(value);
174 }
175 return result;
176}

196static int shm_open_anonymous() {
197 char shm_name[64];
198 uint64_t random_val = 0;
199
200 // Generate a sufficiently random name to avoid collisions if multiple processes
201 // are running this code simultaneously. Using /dev/urandom is a robust way to do this.
202 int rand_fd = open("/dev/urandom", O_RDONLY);
203 if (rand_fd < 0)
204 return -1;
205
206 ssize_t bytes_read = read(rand_fd, &random_val, sizeof(random_val));
207 close(rand_fd);
208 if (bytes_read != sizeof(random_val))
209 return -1;
210
211 snprintf(shm_name, sizeof(shm_name), "/infix-jit-%d-%llx", getpid(), (unsigned long long)random_val);
212
213 // Create the shared memory object exclusively.
214 int fd = shm_open(shm_name, O_RDWR | O_CREAT | O_EXCL, 0600);
215 if (fd >= 0) {
216 // Unlink immediately. The file descriptor remains valid, but the name is removed.
217 // This ensures the kernel will clean up the memory object when the last fd is closed.
218 shm_unlink(shm_name);
219 return fd;
220 }
221
222 return -1;
223}
224#endif

235c23_nodiscard infix_executable_t infix_executable_alloc(size_t size) {
236#if defined(INFIX_OS_WINDOWS)
237 infix_executable_t exec = {.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0, .handle = nullptr};
238#else
239 infix_executable_t exec = {.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0, .shm_fd = -1};
240#endif
241
242 if (size == 0)
243 return exec;
244
245#if defined(INFIX_OS_WINDOWS)
246 // Windows: Single-mapping W^X. Allocate as RW, later change to RX via VirtualProtect.
247 void * code = VirtualAlloc(nullptr, size, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
248 if (code == nullptr)
249 return exec;
250 exec.rw_ptr = code;
251 exec.rx_ptr = code;
252
253#elif defined(INFIX_OS_MACOS) || defined(INFIX_OS_ANDROID) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
254 // Single-mapping POSIX platforms. Allocate as RW, later change to RX via mprotect.
255 void * code = MAP_FAILED;
256#if defined(MAP_ANON)
257 int flags = MAP_PRIVATE | MAP_ANON;
258#if defined(INFIX_OS_MACOS)
259 // On macOS, we perform a one-time check for JIT support.
260 static bool g_use_secure_jit_path = false;
261 static bool g_checked_jit_support = false;
262 if (!g_checked_jit_support) {
263 g_use_secure_jit_path = has_jit_entitlement();
264 INFIX_DEBUG_PRINTF("macOS JIT check: Entitlement found = %s. Using %s API.",
265 g_use_secure_jit_path ? "yes" : "no",
266 g_use_secure_jit_path ? "secure (MAP_JIT)" : "legacy (mprotect)");
267 g_checked_jit_support = true;
268 }
269 // If entitled, use the modern, more secure MAP_JIT flag.
270 if (g_use_secure_jit_path)
271 flags |= MAP_JIT;
272#endif // INFIX_OS_MACOS
273 code = mmap(nullptr, size, PROT_READ | PROT_WRITE, flags, -1, 0);
274#endif // MAP_ANON
275
276 if (code == MAP_FAILED) { // Fallback for older systems without MAP_ANON
277 int fd = open("/dev/zero", O_RDWR);
278 if (fd != -1) {
279 code = mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_PRIVATE, fd, 0);
280 close(fd);
281 }
282 }
283 if (code == MAP_FAILED)
284 return exec;
285 exec.rw_ptr = code;
286 exec.rx_ptr = code;
287
288#else
289 // Dual-mapping POSIX platforms (e.g., Linux, FreeBSD). Create two separate views of the same memory.
290 exec.shm_fd = shm_open_anonymous();
291 if (exec.shm_fd < 0)
292 return exec;
293 if (ftruncate(exec.shm_fd, size) != 0) {
294 close(exec.shm_fd);
295 return exec;
296 }
297 // The RW mapping.
298 exec.rw_ptr = mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, exec.shm_fd, 0);
299 // The RX mapping of the exact same physical memory.
300 exec.rx_ptr = mmap(nullptr, size, PROT_READ | PROT_EXEC, MAP_SHARED, exec.shm_fd, 0);
301
302 // If either mapping fails, clean up both and return an error.
303 if (exec.rw_ptr == MAP_FAILED || exec.rx_ptr == MAP_FAILED) {
304 if (exec.rw_ptr != MAP_FAILED)
305 munmap(exec.rw_ptr, size);
306 if (exec.rx_ptr != MAP_FAILED)
307 munmap(exec.rx_ptr, size);
308 close(exec.shm_fd);
309 return (infix_executable_t){.rx_ptr = nullptr, .rw_ptr = nullptr, .size = 0, .shm_fd = -1};
310 }
311#endif
312
313 exec.size = size;
314 INFIX_DEBUG_PRINTF("Allocated JIT memory. RW at %p, RX at %p", exec.rw_ptr, exec.rx_ptr);
315 return exec;
316}

330void infix_executable_free(infix_executable_t exec) {
331 if (exec.size == 0)
332 return;
333
334#if defined(INFIX_OS_WINDOWS)
335 if (exec.rw_ptr) {
336 // Change protection to NOACCESS to catch use-after-free bugs immediately.
337 if (!VirtualProtect(exec.rw_ptr, exec.size, PAGE_NOACCESS, &(DWORD){0}))
338 INFIX_DEBUG_PRINTF("WARNING: VirtualProtect failed to set PAGE_NOACCESS guard page.");
339 VirtualFree(exec.rw_ptr, 0, MEM_RELEASE);
340 }
341#elif defined(INFIX_OS_MACOS)
342 // On macOS with MAP_JIT, the memory is managed with special thread-local permissions.
343 // We only need to unmap the single mapping.
344 if (exec.rw_ptr) {
345#if INFIX_MACOS_SECURE_JIT_AVAILABLE // This macro is not yet defined, placeholder for future
346 // If using the secure path, we should toggle write protection back on.
347 static bool g_use_secure_jit_path = false; // Re-check or use a shared flag
348 if (g_use_secure_jit_path)
349 pthread_jit_write_protect_np(true);
350#endif
351 // Creating a guard page before unmapping is good practice.
352 mprotect(exec.rw_ptr, exec.size, PROT_NONE);
353 munmap(exec.rw_ptr, exec.size);
354 }
355#elif defined(INFIX_OS_ANDROID) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
356 // Other single-mapping POSIX systems.
357 if (exec.rw_ptr) {
358 mprotect(exec.rw_ptr, exec.size, PROT_NONE);
359 munmap(exec.rw_ptr, exec.size);
360 }
361#else
362 // Dual-mapping POSIX: protect and unmap both views.
363 if (exec.rx_ptr)
364 mprotect(exec.rx_ptr, exec.size, PROT_NONE);
365 if (exec.rw_ptr)
366 munmap(exec.rw_ptr, exec.size);
367 if (exec.rx_ptr && exec.rx_ptr != exec.rw_ptr) // rw_ptr might be same as rx_ptr on some platforms
368 munmap(exec.rx_ptr, exec.size);
369 if (exec.shm_fd >= 0)
370 close(exec.shm_fd);
371#endif
372}

390c23_nodiscard bool infix_executable_make_executable(infix_executable_t exec) {
391 if (exec.rw_ptr == nullptr || exec.size == 0)
392 return false;
393
394 // On AArch64 (and other RISC architectures), the instruction and data caches can be
395 // separate. We must explicitly flush the D-cache (where the JIT wrote the code)
396 // and invalidate the I-cache so the CPU fetches the new instructions.
397#if defined(INFIX_ARCH_AARCH64)
398#if defined(_MSC_VER)
399 // Use the Windows-specific API.
400 FlushInstructionCache(GetCurrentProcess(), exec.rw_ptr, exec.size);
401#else
402 // Use the GCC/Clang built-in for other platforms.
403 __builtin___clear_cache((char *)exec.rw_ptr, (char *)exec.rw_ptr + exec.size);
404#endif
405#endif
406
407 bool result = false;
408#if defined(INFIX_OS_WINDOWS)
409 // Finalize permissions to Read+Execute.
410 result = VirtualProtect(exec.rw_ptr, exec.size, PAGE_EXECUTE_READ, &(DWORD){0});
411#elif defined(INFIX_OS_MACOS)
412#if INFIX_MACOS_SECURE_JIT_AVAILABLE // Placeholder
413 static bool g_use_secure_jit_path = false;
414 if (g_use_secure_jit_path) {
415 pthread_jit_write_protect_np(false); // Make writable region executable.
416 result = true;
417 }
418 else
419#endif
420 // On macOS with the JIT entitlement, we don't use mprotect. Instead, we toggle
421 // a thread-local "write permission" state for all JIT memory. The memory is
422 // RX by default, and we temporarily make it RW for writing.
423 // However, the current logic does this change via `pthread_jit_write_protect_np`
424 // within the allocator itself. For now, this is a placeholder for that logic.
425 result = (mprotect(exec.rw_ptr, exec.size, PROT_READ | PROT_EXEC) == 0);
426#elif defined(INFIX_OS_ANDROID) || defined(INFIX_OS_OPENBSD) || defined(INFIX_OS_DRAGONFLY)
427 // Other single-mapping POSIX platforms use mprotect.
428 result = (mprotect(exec.rw_ptr, exec.size, PROT_READ | PROT_EXEC) == 0);
429#else
430 // On dual-mapping platforms, the RX mapping is already executable. This is a no-op.
431 result = true;
432#endif
433
434 if (result)
435 INFIX_DEBUG_PRINTF("Memory at %p is now executable.", exec.rx_ptr);
436 return result;
437}

541void infix_internal_dispatch_callback_fn_impl(infix_reverse_t * context, void * return_value_ptr, void ** args_array) {
542 INFIX_DEBUG_PRINTF(
543 "Dispatching reverse call. Context: %p, User Fn: %p", (void *)context, context->user_callback_fn);
544
545 if (context->user_callback_fn == nullptr) {
546 // If no handler is set, do nothing. If the function has a return value,
547 // it's good practice to zero it out to avoid returning garbage.
548 if (return_value_ptr && context->return_type->size > 0)
549 infix_memset(return_value_ptr, 0, context->return_type->size);
550 return;
551 }
552
553 if (context->cached_forward_trampoline != nullptr) {
554 // Path 1: Type-safe "callback". Use the pre-generated forward trampoline to
555 // call the user's C function with the correct signature. This is efficient
556 // and provides a clean interface for the C developer.
557 infix_cif_func cif_func = infix_forward_get_code(context->cached_forward_trampoline);
558 cif_func(return_value_ptr, args_array);
559 }
560 else {
561 // Path 2: Generic "closure". Directly call the user's generic handler.
562 // This path is more flexible and is intended for language bindings where the
563 // handler needs access to the context and raw argument pointers.
564 infix_closure_handler_fn handler = (infix_closure_handler_fn)context->user_callback_fn;
565 handler(context, return_value_ptr, args_array);
566 }
567
568 INFIX_DEBUG_PRINTF("Exiting reverse call dispatcher.");
569}
