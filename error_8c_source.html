110void _infix_set_error(infix_error_category_t category, infix_error_code_t code, size_t position) {
111 g_infix_last_error.category = category;
112 g_infix_last_error.code = code;
113 g_infix_last_error.position = position;
114 g_infix_last_error.system_error_code = 0;
115
116 if (category == INFIX_CATEGORY_PARSER && g_infix_last_signature_context != NULL) {
117 // Generate a rich, GCC-style error message for parser failures.
118 const char * signature = g_infix_last_signature_context;
119 size_t sig_len = strlen(signature);
120 size_t radius = 20;
121
122 size_t start = (position > radius) ? (position - radius) : 0;
123 size_t end = (position + radius < sig_len) ? (position + radius) : sig_len;
124
125 const char * start_indicator = (start > 0) ? "... " : "";
126 const char * end_indicator = (end < sig_len) ? " ..." : "";
127 size_t start_indicator_len = (start > 0) ? 4 : 0;
128
129 char snippet[128];
130 snprintf(snippet,
131 sizeof(snippet),
132 "%s%.*s%s",
133 start_indicator,
134 (int)(end - start),
135 signature + start,
136 end_indicator);
137
138 char pointer[128];
139 size_t caret_pos = position - start + start_indicator_len;
140 snprintf(pointer, sizeof(pointer), "%*s^", (int)caret_pos, "");
141
142 // Build the message piece by piece to avoid buffer overflows.
143 char * p = g_infix_last_error.message;
144 size_t remaining = sizeof(g_infix_last_error.message);
145 int written;
146
147 written = snprintf(p, remaining, "\n\n %s\n %s", snippet, pointer);
148 if (written < 0 || (size_t)written >= remaining) {
149 const char * msg = _get_error_message_for_code(code);
150 _INFIX_SAFE_STRNCPY(g_infix_last_error.message, msg, sizeof(g_infix_last_error.message) - 1);
151 return;
152 }
153 p += written;
154 remaining -= written;
155
156 written = snprintf(p, remaining, "\n\nError: %s", _get_error_message_for_code(code));
157 // If this last part gets truncated, that's acceptable. snprintf will null-terminate.
158 }
159 else {
160 // For non-parser errors, just copy the standard message.
161 const char * msg = _get_error_message_for_code(code);
162 _INFIX_SAFE_STRNCPY(g_infix_last_error.message, msg, sizeof(g_infix_last_error.message) - 1);
163 }
164}
