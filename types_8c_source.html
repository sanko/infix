215static infix_status _create_aggregate_setup(infix_arena_t * arena,
216 infix_type ** out_type,
217 infix_struct_member ** out_arena_members,
218 infix_struct_member * members,
219 size_t num_members) {
220 if (out_type == nullptr)
221 return INFIX_ERROR_INVALID_ARGUMENT;
222 // Pre-flight check: ensure all provided member types are valid.
223 for (size_t i = 0; i < num_members; ++i) {
224 if (members[i].type == nullptr) {
225 *out_type = nullptr;
226 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INVALID_MEMBER_TYPE, 0);
227 return INFIX_ERROR_INVALID_ARGUMENT;
228 }
229 }
230 infix_type * type = infix_arena_calloc(arena, 1, sizeof(infix_type), _Alignof(infix_type));
231 if (type == nullptr) {
232 *out_type = nullptr;
233 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
234 return INFIX_ERROR_ALLOCATION_FAILED;
235 }
236
237 infix_struct_member * arena_members = nullptr;
238 if (num_members > 0) {
239 arena_members =
240 infix_arena_alloc(arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
241 if (arena_members == nullptr) {
242 *out_type = nullptr;
243 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
244 return INFIX_ERROR_ALLOCATION_FAILED;
245 }
246 infix_memcpy(arena_members, members, sizeof(infix_struct_member) * num_members);
247 }
248 *out_type = type;
249 *out_arena_members = arena_members;
250 return INFIX_SUCCESS;
251}

434c23_nodiscard infix_status infix_type_create_union(infix_arena_t * arena,
435 infix_type ** out_type,
436 infix_struct_member * members,
437 size_t num_members) {
438 infix_type * type = nullptr;
439 infix_struct_member * arena_members = nullptr;
440 infix_status status = _create_aggregate_setup(arena, &type, &arena_members, members, num_members);
441 if (status != INFIX_SUCCESS) {
442 *out_type = nullptr;
443 return status;
444 }
445 type->is_arena_allocated = true;
446 type->category = INFIX_TYPE_UNION;
447 type->meta.aggregate_info.members = arena_members;
448 type->meta.aggregate_info.num_members = num_members;
449
450 // A union's size is the size of its largest member, and its alignment is the
451 // alignment of its most-aligned member.
452 size_t max_size = 0;
453 size_t max_alignment = 1;
454 for (size_t i = 0; i < num_members; ++i) {
455 arena_members[i].offset = 0; // All union members have an offset of 0.
456 if (arena_members[i].type->size > max_size)
457 max_size = arena_members[i].type->size;
458 if (arena_members[i].type->alignment > max_alignment)
459 max_alignment = arena_members[i].type->alignment;
460 }
461 type->alignment = max_alignment;
462 // The total size is the size of the largest member, padded up to the required alignment.
463 type->size = _infix_align_up(max_size, max_alignment);
464 // Overflow check
465 if (type->size < max_size) {
466 *out_type = nullptr;
467 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
468 return INFIX_ERROR_INVALID_ARGUMENT;
469 }
470 *out_type = type;
471 return INFIX_SUCCESS;
472}

482c23_nodiscard infix_status infix_type_create_struct(infix_arena_t * arena,
483 infix_type ** out_type,
484 infix_struct_member * members,
485 size_t num_members) {
486 _infix_clear_error();
487 infix_type * type = nullptr;
488 infix_struct_member * arena_members = nullptr;
489 infix_status status = _create_aggregate_setup(arena, &type, &arena_members, members, num_members);
490 if (status != INFIX_SUCCESS) {
491 *out_type = nullptr;
492 return status;
493 }
494 type->is_arena_allocated = true;
495 type->category = INFIX_TYPE_STRUCT;
496 type->meta.aggregate_info.members = arena_members;
497 type->meta.aggregate_info.num_members = num_members;
498
499 // This performs a preliminary layout calculation based on standard C packing rules.
500 // Note: This layout may be incomplete if it contains unresolved named references.
501 // The final, correct layout will be computed by `_infix_type_recalculate_layout`.
502 size_t current_offset = 0;
503 size_t max_alignment = 1;
504 for (size_t i = 0; i < num_members; ++i) {
505 infix_struct_member * member = &arena_members[i];
506 size_t member_align = member->type->alignment;
507
508 // An unresolved named reference will have alignment 0 or 1. If it's 0, we treat it as 1 for
509 // this preliminary pass to avoid division by zero. The real alignment will be fixed later.
510 if (member_align == 0) {
511 if (member->type->category != INFIX_TYPE_NAMED_REFERENCE) {
512 // A zero-alignment type that isn't a named reference is invalid (e.g., a struct with a void member).
513 *out_type = nullptr;
514 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INVALID_MEMBER_TYPE, 0);
515 return INFIX_ERROR_INVALID_ARGUMENT;
516 }
517 member_align = 1;
518 }
519
520 // Align the current offset up to the boundary required by the current member.
521 size_t aligned_offset = _infix_align_up(current_offset, member_align);
522 if (aligned_offset < current_offset) { // Overflow check
523 *out_type = nullptr;
524 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
525 return INFIX_ERROR_INVALID_ARGUMENT;
526 }
527 current_offset = aligned_offset;
528 member->offset = current_offset;
529
530 // Add the member's size to the current offset.
531 if (current_offset > SIZE_MAX - member->type->size) { // Overflow check
532 *out_type = nullptr;
533 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
534 return INFIX_ERROR_INVALID_ARGUMENT;
535 }
536 current_offset += member->type->size;
537 // Keep track of the largest alignment requirement.
538 if (member_align > max_alignment)
539 max_alignment = member_align;
540 }
541 // A struct's alignment is the alignment of its most-aligned member.
542 type->alignment = max_alignment;
543 // The struct's total size is its occupied space, padded up to its alignment.
544 type->size = _infix_align_up(current_offset, max_alignment);
545 if (type->size < current_offset) { // Overflow check
546 *out_type = nullptr;
547 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
548 return INFIX_ERROR_INVALID_ARGUMENT;
549 }
550 *out_type = type;
551 return INFIX_SUCCESS;
552}

686static void _infix_type_recalculate_layout_recursive(infix_arena_t * temp_arena,
687 infix_type * type,
688 recalc_visited_node_t ** visited_head) {
689 if (!type || !type->is_arena_allocated)
690 return; // Base case: Don't modify static singleton types.
691
692 // Cycle detection: If we have already visited this node in the current recursion
693 // path, we are in a cycle. Return immediately to break the loop. The layout of
694 // this node will be calculated when the recursion unwinds to its first visit.
695 for (recalc_visited_node_t * v = *visited_head; v != nullptr; v = v->next)
696 if (v->type == type)
697 return;
698
699 // Allocate the memoization node from a stable temporary arena.
700 recalc_visited_node_t * visited_node =
701 infix_arena_alloc(temp_arena, sizeof(recalc_visited_node_t), _Alignof(recalc_visited_node_t));
702 if (!visited_node)
703 return; // Cannot proceed without memory.
704 visited_node->type = type;
705 visited_node->next = *visited_head;
706 *visited_head = visited_node;
707
708 // Recurse into child types first (post-order traversal).
709 switch (type->category) {
710 case INFIX_TYPE_POINTER:
711 _infix_type_recalculate_layout_recursive(temp_arena, type->meta.pointer_info.pointee_type, visited_head);
712 break;
713 case INFIX_TYPE_ARRAY:
714 _infix_type_recalculate_layout_recursive(temp_arena, type->meta.array_info.element_type, visited_head);
715 break;
716 case INFIX_TYPE_STRUCT:
717 case INFIX_TYPE_UNION:
718 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
719 _infix_type_recalculate_layout_recursive(
720 temp_arena, type->meta.aggregate_info.members[i].type, visited_head);
721 }
722 break;
723 default:
724 break; // Other types have no child types to recurse into.
725 }
726
727 // After children are updated, recalculate this type's layout.
728 if (type->category == INFIX_TYPE_STRUCT) {
729 size_t current_offset = 0;
730 size_t max_alignment = 1;
731 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
732 infix_struct_member * member = &type->meta.aggregate_info.members[i];
733 size_t member_align = member->type->alignment;
734 if (member_align > max_alignment)
735 max_alignment = member_align;
736 current_offset = _infix_align_up(current_offset, member_align);
737 member->offset = current_offset;
738 current_offset += member->type->size;
739 }
740 type->alignment = max_alignment;
741 type->size = _infix_align_up(current_offset, max_alignment);
742 }
743 else if (type->category == INFIX_TYPE_UNION) {
744 size_t max_size = 0;
745 size_t max_alignment = 1;
746 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
747 infix_type * member_type = type->meta.aggregate_info.members[i].type;
748 if (member_type->size > max_size)
749 max_size = member_type->size;
750 if (member_type->alignment > max_alignment)
751 max_alignment = member_type->alignment;
752 }
753 type->alignment = max_alignment;
754 type->size = _infix_align_up(max_size, max_alignment);
755 }
756 else if (type->category == INFIX_TYPE_ARRAY) {
757 // Recalculate array size based on the (potentially updated) element size.
758 type->alignment = type->meta.array_info.element_type->alignment;
759 type->size = type->meta.array_info.element_type->size * type->meta.array_info.num_elements;
760 }
761}

772void _infix_type_recalculate_layout(infix_type * type) {
773 // Create a temporary arena solely for the visited list's lifetime.
774 infix_arena_t * temp_arena = infix_arena_create(1024);
775 if (!temp_arena)
776 return;
777
778 recalc_visited_node_t * visited_head = nullptr;
779 _infix_type_recalculate_layout_recursive(temp_arena, type, &visited_head);
780
781 infix_arena_destroy(temp_arena);
782}

792typedef struct memo_node_t {
793 const infix_type * src;
794 infix_type * dest;
795 struct memo_node_t * next;
796} memo_node_t;
797
813static infix_type * _copy_type_graph_to_arena_recursive(infix_arena_t * dest_arena,
814 const infix_type * src_type,
815 memo_node_t ** memo_head) {
816 if (src_type == nullptr)
817 return nullptr;
818
819 // Base case: Static types don't need to be copied; return the singleton pointer.
820 if (!src_type->is_arena_allocated)
821 return (infix_type *)src_type;
822
823 // Check memoization table: if we've already copied this node, return the existing copy.
824 // This correctly handles cycles and shared sub-graphs.
825 for (memo_node_t * node = *memo_head; node != NULL; node = node->next)
826 if (node->src == src_type)
827 return node->dest;
828
829 // Allocate the new type object in the destination arena.
830 infix_type * dest_type = infix_arena_calloc(dest_arena, 1, sizeof(infix_type), _Alignof(infix_type));
831 if (dest_type == nullptr)
832 return nullptr;
833
834 // Add this new pair to the memoization table BEFORE recursing. This is crucial
835 // for handling cycles: the recursive call will find this entry and return `dest_type`.
836 memo_node_t * new_memo_node = infix_arena_alloc(dest_arena, sizeof(memo_node_t), _Alignof(memo_node_t));
837 if (!new_memo_node)
838 return nullptr;
839 new_memo_node->src = src_type;
840 new_memo_node->dest = dest_type;
841 new_memo_node->next = *memo_head;
842 *memo_head = new_memo_node;
843
844 // Perform a shallow copy of the main struct, then recurse to deep copy child pointers.
845 *dest_type = *src_type;
846 dest_type->is_arena_allocated = true;
847
848 switch (src_type->category) {
849 case INFIX_TYPE_POINTER:
850 dest_type->meta.pointer_info.pointee_type =
851 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.pointer_info.pointee_type, memo_head);
852 break;
853 case INFIX_TYPE_ARRAY:
854 dest_type->meta.array_info.element_type =
855 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.array_info.element_type, memo_head);
856 break;
857 case INFIX_TYPE_STRUCT:
858 case INFIX_TYPE_UNION:
859 if (src_type->meta.aggregate_info.num_members > 0) {
860 // Copy the members array itself.
861 size_t members_size = sizeof(infix_struct_member) * src_type->meta.aggregate_info.num_members;
862 dest_type->meta.aggregate_info.members =
863 infix_arena_alloc(dest_arena, members_size, _Alignof(infix_struct_member));
864 if (dest_type->meta.aggregate_info.members == nullptr)
865 return nullptr;
866 // Now, recurse for each member's type and copy its name.
867 for (size_t i = 0; i < src_type->meta.aggregate_info.num_members; ++i) {
868 dest_type->meta.aggregate_info.members[i] = src_type->meta.aggregate_info.members[i];
869 dest_type->meta.aggregate_info.members[i].type = _copy_type_graph_to_arena_recursive(
870 dest_arena, src_type->meta.aggregate_info.members[i].type, memo_head);
871 const char * src_name = src_type->meta.aggregate_info.members[i].name;
872 if (src_name) {
873 size_t name_len = strlen(src_name) + 1;
874 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
875 if (!dest_name)
876 return nullptr;
877 infix_memcpy(dest_name, src_name, name_len);
878 dest_type->meta.aggregate_info.members[i].name = dest_name;
879 }
880 }
881 }
882 break;
883 case INFIX_TYPE_NAMED_REFERENCE:
884 {
885 const char * src_name = src_type->meta.named_reference.name;
886 if (src_name) {
887 size_t name_len = strlen(src_name) + 1;
888 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
889 if (!dest_name)
890 return nullptr;
891 infix_memcpy(dest_name, src_name, name_len);
892 dest_type->meta.named_reference.name = dest_name;
893 }
894 break;
895 }
896 case INFIX_TYPE_REVERSE_TRAMPOLINE:
897 dest_type->meta.func_ptr_info.return_type =
898 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.func_ptr_info.return_type, memo_head);
899 if (src_type->meta.func_ptr_info.num_args > 0) {
900 size_t args_size = sizeof(infix_function_argument) * src_type->meta.func_ptr_info.num_args;
901 dest_type->meta.func_ptr_info.args =
902 infix_arena_alloc(dest_arena, args_size, _Alignof(infix_function_argument));
903 if (dest_type->meta.func_ptr_info.args == nullptr)
904 return nullptr;
905 for (size_t i = 0; i < src_type->meta.func_ptr_info.num_args; ++i) {
906 dest_type->meta.func_ptr_info.args[i] = src_type->meta.func_ptr_info.args[i];
907 dest_type->meta.func_ptr_info.args[i].type = _copy_type_graph_to_arena_recursive(
908 dest_arena, src_type->meta.func_ptr_info.args[i].type, memo_head);
909 const char * src_name = src_type->meta.func_ptr_info.args[i].name;
910 if (src_name) {
911 size_t name_len = strlen(src_name) + 1;
912 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
913 if (!dest_name)
914 return nullptr;
915 infix_memcpy(dest_name, src_name, name_len);
916 dest_type->meta.func_ptr_info.args[i].name = dest_name;
917 }
918 }
919 }
920 break;
921 case INFIX_TYPE_ENUM:
922 dest_type->meta.enum_info.underlying_type =
923 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.enum_info.underlying_type, memo_head);
924 break;
925 case INFIX_TYPE_COMPLEX:
926 dest_type->meta.complex_info.base_type =
927 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.complex_info.base_type, memo_head);
928 break;
929 case INFIX_TYPE_VECTOR:
930 dest_type->meta.vector_info.element_type =
931 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.vector_info.element_type, memo_head);
932 break;
933 default:
934 // Other types like primitives have no child pointers to copy.
935 break;
936 }
937 return dest_type;
938}

948 memo_node_t * memo_head = nullptr;

957typedef struct estimate_visited_node_t {
958 const infix_type * type;
959 struct estimate_visited_node_t * next;
960} estimate_visited_node_t;
961
974static size_t _estimate_graph_size_recursive(infix_arena_t * temp_arena,
975 const infix_type * type,
976 estimate_visited_node_t ** visited_head) {
977 if (!type || !type->is_arena_allocated)
978 return 0;
979
980 // Cycle detection: if we've seen this node, it's already accounted for.
981 for (estimate_visited_node_t * v = *visited_head; v != NULL; v = v->next)
982 if (v->type == type)
983 return 0;
984
985 // Add this node to the visited list before recursing.
986 estimate_visited_node_t * visited_node =
987 infix_arena_alloc(temp_arena, sizeof(estimate_visited_node_t), _Alignof(estimate_visited_node_t));
988 if (!visited_node) {
989 // On allocation failure, we can't proceed with estimation. Return a large
990 // number to ensure the caller allocates a fallback-sized arena.
991 return 65536;
992 }
993 visited_node->type = type;
994 visited_node->next = *visited_head;
995 *visited_head = visited_node;
996
997 // The size includes the type object itself, plus a memoization node used by the copy algorithm.
998 size_t total_size = sizeof(infix_type) + sizeof(memo_node_t);
999
1000 switch (type->category) {
1001 case INFIX_TYPE_POINTER:
1002 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.pointer_info.pointee_type, visited_head);
1003 break;
1004 case INFIX_TYPE_ARRAY:
1005 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.array_info.element_type, visited_head);
1006 break;
1007 case INFIX_TYPE_STRUCT:
1008 case INFIX_TYPE_UNION:
1009 if (type->meta.aggregate_info.num_members > 0) {
1010 total_size += sizeof(infix_struct_member) * type->meta.aggregate_info.num_members;
1011 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
1012 const infix_struct_member * member = &type->meta.aggregate_info.members[i];
1013 if (member->name)
1014 total_size += strlen(member->name) + 1;
1015 total_size += _estimate_graph_size_recursive(temp_arena, member->type, visited_head);
1016 }
1017 }
1018 break;
1019 case INFIX_TYPE_NAMED_REFERENCE:
1020 if (type->meta.named_reference.name)
1021 total_size += strlen(type->meta.named_reference.name) + 1;
1022 break;
1023 case INFIX_TYPE_REVERSE_TRAMPOLINE:
1024 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.func_ptr_info.return_type, visited_head);
1025 if (type->meta.func_ptr_info.num_args > 0) {
1026 total_size += sizeof(infix_function_argument) * type->meta.func_ptr_info.num_args;
1027 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i) {
1028 const infix_function_argument * arg = &type->meta.func_ptr_info.args[i];
1029 if (arg->name)
1030 total_size += strlen(arg->name) + 1;
1031 total_size += _estimate_graph_size_recursive(temp_arena, arg->type, visited_head);
1032 }
1033 }
1034 break;
1035 case INFIX_TYPE_ENUM:
1036 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.enum_info.underlying_type, visited_head);
1037 break;
1038 case INFIX_TYPE_COMPLEX:
1039 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.complex_info.base_type, visited_head);
1040 break;
1041 case INFIX_TYPE_VECTOR:
1042 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.vector_info.element_type, visited_head);
1043 break;
1044 default:
1045 break;
1046 }
1047
1048 return total_size;
1049}

1058size_t _infix_estimate_graph_size(infix_arena_t * temp_arena, const infix_type * type) {
1059 if (!temp_arena || !type)
1060 return 0;
1061
1062 estimate_visited_node_t * visited_head = NULL;
1063 return _estimate_graph_size_recursive(temp_arena, type, &visited_head);
1064}
