219static infix_status _create_aggregate_setup(infix_arena_t * arena,
220 infix_type ** out_type,
221 infix_struct_member ** out_arena_members,
222 infix_struct_member * members,
223 size_t num_members) {
224 if (out_type == nullptr)
225 return INFIX_ERROR_INVALID_ARGUMENT;
226 // Pre-flight check: ensure all provided member types are valid.
227 for (size_t i = 0; i < num_members; ++i) {
228 if (members[i].type == nullptr) {
229 *out_type = nullptr;
230 return INFIX_ERROR_INVALID_ARGUMENT;
231 }
232 }
233 infix_type * type = infix_arena_calloc(arena, 1, sizeof(infix_type), _Alignof(infix_type));
234 if (type == nullptr) {
235 *out_type = nullptr;
236 return INFIX_ERROR_ALLOCATION_FAILED;
237 }
238
239 infix_struct_member * arena_members = nullptr;
240 if (num_members > 0) {
241 arena_members =
242 infix_arena_alloc(arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
243 if (arena_members == nullptr) {
244 *out_type = nullptr;
245 return INFIX_ERROR_ALLOCATION_FAILED;
246 }
247 infix_memcpy(arena_members, members, sizeof(infix_struct_member) * num_members);
248 }
249 *out_type = type;
250 *out_arena_members = arena_members;
251 return INFIX_SUCCESS;
252}

422c23_nodiscard infix_status infix_type_create_union(infix_arena_t * arena,
423 infix_type ** out_type,
424 infix_struct_member * members,
425 size_t num_members) {
426 infix_type * type = nullptr;
427 infix_struct_member * arena_members = nullptr;
428 infix_status status = _create_aggregate_setup(arena, &type, &arena_members, members, num_members);
429 if (status != INFIX_SUCCESS) {
430 *out_type = nullptr;
431 return status;
432 }
433 type->is_arena_allocated = true;
434 type->category = INFIX_TYPE_UNION;
435 type->meta.aggregate_info.members = arena_members;
436 type->meta.aggregate_info.num_members = num_members;
437
438 // A union's size is the size of its largest member, and its alignment is the
439 // alignment of its most-aligned member.
440 size_t max_size = 0;
441 size_t max_alignment = 1;
442 for (size_t i = 0; i < num_members; ++i) {
443 arena_members[i].offset = 0; // All union members have an offset of 0.
444 if (arena_members[i].type->size > max_size)
445 max_size = arena_members[i].type->size;
446 if (arena_members[i].type->alignment > max_alignment)
447 max_alignment = arena_members[i].type->alignment;
448 }
449 type->alignment = max_alignment;
450 // The total size is the size of the largest member, padded up to the required alignment.
451 type->size = _infix_align_up(max_size, max_alignment);
452 // Overflow check
453 if (type->size < max_size) {
454 *out_type = nullptr;
455 return INFIX_ERROR_INVALID_ARGUMENT;
456 }
457 *out_type = type;
458 return INFIX_SUCCESS;
459}

469c23_nodiscard infix_status infix_type_create_struct(infix_arena_t * arena,
470 infix_type ** out_type,
471 infix_struct_member * members,
472 size_t num_members) {
473 _infix_clear_error();
474 infix_type * type = nullptr;
475 infix_struct_member * arena_members = nullptr;
476 infix_status status = _create_aggregate_setup(arena, &type, &arena_members, members, num_members);
477 if (status != INFIX_SUCCESS) {
478 *out_type = nullptr;
479 return status;
480 }
481 type->is_arena_allocated = true;
482 type->category = INFIX_TYPE_STRUCT;
483 type->meta.aggregate_info.members = arena_members;
484 type->meta.aggregate_info.num_members = num_members;
485
486 // This performs a preliminary layout calculation based on standard C packing rules.
487 // Note: This layout may be incomplete if it contains unresolved named references.
488 // The final, correct layout will be computed by `_infix_type_recalculate_layout`.
489 size_t current_offset = 0;
490 size_t max_alignment = 1;
491 for (size_t i = 0; i < num_members; ++i) {
492 infix_struct_member * member = &arena_members[i];
493 size_t member_align = member->type->alignment;
494
495 // An unresolved named reference will have alignment 0 or 1. If it's 0, we treat it as 1 for
496 // this preliminary pass to avoid division by zero. The real alignment will be fixed later.
497 if (member_align == 0) {
498 if (member->type->category != INFIX_TYPE_NAMED_REFERENCE) {
499 // A zero-alignment type that isn't a named reference is invalid (e.g., a struct with a void member).
500 *out_type = nullptr;
501 return INFIX_ERROR_INVALID_ARGUMENT;
502 }
503 member_align = 1;
504 }
505
506 // Align the current offset up to the boundary required by the current member.
507 size_t aligned_offset = _infix_align_up(current_offset, member_align);
508 if (aligned_offset < current_offset) { // Overflow check
509 *out_type = nullptr;
510 return INFIX_ERROR_INVALID_ARGUMENT;
511 }
512 current_offset = aligned_offset;
513 member->offset = current_offset;
514
515 // Add the member's size to the current offset.
516 if (current_offset > SIZE_MAX - member->type->size) { // Overflow check
517 *out_type = nullptr;
518 return INFIX_ERROR_INVALID_ARGUMENT;
519 }
520 current_offset += member->type->size;
521 // Keep track of the largest alignment requirement.
522 if (member_align > max_alignment)
523 max_alignment = member_align;
524 }
525 // A struct's alignment is the alignment of its most-aligned member.
526 type->alignment = max_alignment;
527 // The struct's total size is its occupied space, padded up to its alignment.
528 type->size = _infix_align_up(current_offset, max_alignment);
529 if (type->size < current_offset) { // Overflow check
530 *out_type = nullptr;
531 return INFIX_ERROR_INVALID_ARGUMENT;
532 }
533 *out_type = type;
534 return INFIX_SUCCESS;
535}

661static void _infix_type_recalculate_layout_recursive(infix_type * type, recalc_visited_node_t ** visited_head) {
662 if (!type || !type->is_arena_allocated)
663 return; // Base case: Don't modify static singleton types.
664
665 // Cycle detection: If we have already visited this node in the current recursion
666 // path, we are in a cycle. Return immediately to break the loop. The layout of
667 // this node will be calculated when the recursion unwinds to its first visit.
668 for (recalc_visited_node_t * v = *visited_head; v != nullptr; v = v->next) {
669 if (v->type == type)
670 return;
671 }
672 // Add this node to the visited list before recursing into its children.
673 recalc_visited_node_t * visited_node = infix_malloc(sizeof(recalc_visited_node_t));
674 if (!visited_node)
675 return; // Cannot proceed without memory.
676 visited_node->type = type;
677 visited_node->next = *visited_head;
678 *visited_head = visited_node;
679
680 // Recurse into child types first (post-order traversal).
681 switch (type->category) {
682 case INFIX_TYPE_POINTER:
683 _infix_type_recalculate_layout_recursive(type->meta.pointer_info.pointee_type, visited_head);
684 break;
685 case INFIX_TYPE_ARRAY:
686 _infix_type_recalculate_layout_recursive(type->meta.array_info.element_type, visited_head);
687 break;
688 case INFIX_TYPE_STRUCT:
689 case INFIX_TYPE_UNION:
690 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
691 _infix_type_recalculate_layout_recursive(type->meta.aggregate_info.members[i].type, visited_head);
692 }
693 break;
694 default:
695 break; // Other types have no child types to recurse into.
696 }
697
698 // After children are updated, recalculate this type's layout.
699 if (type->category == INFIX_TYPE_STRUCT) {
700 size_t current_offset = 0;
701 size_t max_alignment = 1;
702 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
703 infix_struct_member * member = &type->meta.aggregate_info.members[i];
704 size_t member_align = member->type->alignment;
705 if (member_align > max_alignment)
706 max_alignment = member_align;
707 current_offset = _infix_align_up(current_offset, member_align);
708 member->offset = current_offset;
709 current_offset += member->type->size;
710 }
711 type->alignment = max_alignment;
712 type->size = _infix_align_up(current_offset, max_alignment);
713 }
714 else if (type->category == INFIX_TYPE_UNION) {
715 size_t max_size = 0;
716 size_t max_alignment = 1;
717 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
718 infix_type * member_type = type->meta.aggregate_info.members[i].type;
719 if (member_type->size > max_size)
720 max_size = member_type->size;
721 if (member_type->alignment > max_alignment)
722 max_alignment = member_type->alignment;
723 }
724 type->alignment = max_alignment;
725 type->size = _infix_align_up(max_size, max_alignment);
726 }
727 else if (type->category == INFIX_TYPE_ARRAY) {
728 // Recalculate array size based on the (potentially updated) element size.
729 type->alignment = type->meta.array_info.element_type->alignment;
730 type->size = type->meta.array_info.element_type->size * type->meta.array_info.num_elements;
731 }
732
733 // Pop this node from the visited list before returning up the call stack.
734 *visited_head = visited_node->next;
735 infix_free(visited_node);
736}

747void _infix_type_recalculate_layout(infix_type * type) {
748 recalc_visited_node_t * visited_head = nullptr;
749 _infix_type_recalculate_layout_recursive(type, &visited_head);
750}

781static infix_type * _copy_type_graph_to_arena_recursive(infix_arena_t * dest_arena,
782 const infix_type * src_type,
783 memo_node_t ** memo_head) {
784 if (src_type == nullptr)
785 return nullptr;
786
787 // Base case: Static types don't need to be copied; return the singleton pointer.
788 if (!src_type->is_arena_allocated)
789 return (infix_type *)src_type;
790
791 // Check memoization table: if we've already copied this node, return the existing copy.
792 // This correctly handles cycles and shared sub-graphs.
793 for (memo_node_t * node = *memo_head; node != NULL; node = node->next) {
794 if (node->src == src_type)
795 return node->dest;
796 }
797
798 // Allocate the new type object in the destination arena.
799 infix_type * dest_type = infix_arena_calloc(dest_arena, 1, sizeof(infix_type), _Alignof(infix_type));
800 if (dest_type == nullptr)
801 return nullptr;
802
803 // Add this new pair to the memoization table BEFORE recursing. This is crucial
804 // for handling cycles: the recursive call will find this entry and return `dest_type`.
805 memo_node_t * new_memo_node = infix_arena_alloc(dest_arena, sizeof(memo_node_t), _Alignof(memo_node_t));
806 if (!new_memo_node)
807 return nullptr;
808 new_memo_node->src = src_type;
809 new_memo_node->dest = dest_type;
810 new_memo_node->next = *memo_head;
811 *memo_head = new_memo_node;
812
813 // Perform a shallow copy of the main struct, then recurse to deep copy child pointers.
814 *dest_type = *src_type;
815 dest_type->is_arena_allocated = true;
816
817 switch (src_type->category) {
818 case INFIX_TYPE_POINTER:
819 dest_type->meta.pointer_info.pointee_type =
820 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.pointer_info.pointee_type, memo_head);
821 break;
822 case INFIX_TYPE_ARRAY:
823 dest_type->meta.array_info.element_type =
824 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.array_info.element_type, memo_head);
825 break;
826 case INFIX_TYPE_STRUCT:
827 case INFIX_TYPE_UNION:
828 if (src_type->meta.aggregate_info.num_members > 0) {
829 // Copy the members array itself.
830 size_t members_size = sizeof(infix_struct_member) * src_type->meta.aggregate_info.num_members;
831 dest_type->meta.aggregate_info.members =
832 infix_arena_alloc(dest_arena, members_size, _Alignof(infix_struct_member));
833 if (dest_type->meta.aggregate_info.members == nullptr)
834 return nullptr;
835 // Now, recurse for each member's type and copy its name.
836 for (size_t i = 0; i < src_type->meta.aggregate_info.num_members; ++i) {
837 dest_type->meta.aggregate_info.members[i] = src_type->meta.aggregate_info.members[i];
838 dest_type->meta.aggregate_info.members[i].type = _copy_type_graph_to_arena_recursive(
839 dest_arena, src_type->meta.aggregate_info.members[i].type, memo_head);
840 const char * src_name = src_type->meta.aggregate_info.members[i].name;
841 if (src_name) {
842 size_t name_len = strlen(src_name) + 1;
843 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
844 if (!dest_name)
845 return nullptr;
846 infix_memcpy(dest_name, src_name, name_len);
847 dest_type->meta.aggregate_info.members[i].name = dest_name;
848 }
849 }
850 }
851 break;
852 case INFIX_TYPE_NAMED_REFERENCE:
853 {
854 const char * src_name = src_type->meta.named_reference.name;
855 if (src_name) {
856 size_t name_len = strlen(src_name) + 1;
857 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
858 if (!dest_name)
859 return nullptr;
860 infix_memcpy(dest_name, src_name, name_len);
861 dest_type->meta.named_reference.name = dest_name;
862 }
863 break;
864 }
865 case INFIX_TYPE_REVERSE_TRAMPOLINE:
866 dest_type->meta.func_ptr_info.return_type =
867 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.func_ptr_info.return_type, memo_head);
868 if (src_type->meta.func_ptr_info.num_args > 0) {
869 size_t args_size = sizeof(infix_function_argument) * src_type->meta.func_ptr_info.num_args;
870 dest_type->meta.func_ptr_info.args =
871 infix_arena_alloc(dest_arena, args_size, _Alignof(infix_function_argument));
872 if (dest_type->meta.func_ptr_info.args == nullptr)
873 return nullptr;
874 for (size_t i = 0; i < src_type->meta.func_ptr_info.num_args; ++i) {
875 dest_type->meta.func_ptr_info.args[i] = src_type->meta.func_ptr_info.args[i];
876 dest_type->meta.func_ptr_info.args[i].type = _copy_type_graph_to_arena_recursive(
877 dest_arena, src_type->meta.func_ptr_info.args[i].type, memo_head);
878 const char * src_name = src_type->meta.func_ptr_info.args[i].name;
879 if (src_name) {
880 size_t name_len = strlen(src_name) + 1;
881 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
882 if (!dest_name)
883 return nullptr;
884 infix_memcpy(dest_name, src_name, name_len);
885 dest_type->meta.func_ptr_info.args[i].name = dest_name;
886 }
887 }
888 }
889 break;
890 case INFIX_TYPE_ENUM:
891 dest_type->meta.enum_info.underlying_type =
892 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.enum_info.underlying_type, memo_head);
893 break;
894 case INFIX_TYPE_COMPLEX:
895 dest_type->meta.complex_info.base_type =
896 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.complex_info.base_type, memo_head);
897 break;
898 case INFIX_TYPE_VECTOR:
899 dest_type->meta.vector_info.element_type =
900 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.vector_info.element_type, memo_head);
901 break;
902 default:
903 // Other types like primitives have no child pointers to copy.
904 break;
905 }
906 return dest_type;
907}

916infix_type * _copy_type_graph_to_arena(infix_arena_t * dest_arena, const infix_type * src_type) {
917 memo_node_t * memo_head = nullptr;
918 return _copy_type_graph_to_arena_recursive(dest_arena, src_type, &memo_head);
