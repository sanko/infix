167/*
168 * @internal
169 * This is a common setup function for creating any aggregate type (struct or union)
170 * from an arena. It handles argument validation, allocates the main `infix_type`
171 * struct, and crucially, copies the user-provided `members` array into the arena.
172 * This copy makes the final `infix_type` object self-contained and immune to use-after-free
173 * bugs if the original `members` array was stack-allocated.
174 */
175static infix_status _create_aggregate_setup(infix_arena_t * arena,
176 infix_type ** out_type,
177 infix_struct_member ** out_arena_members,
178 infix_struct_member * members,
179 size_t num_members) {
180 if (out_type == nullptr)
181 return INFIX_ERROR_INVALID_ARGUMENT;
182
183 // Validate that all member types are non-null before proceeding.
184 for (size_t i = 0; i < num_members; ++i) {
185 if (members[i].type == nullptr) {
186 *out_type = nullptr;
187 return INFIX_ERROR_INVALID_ARGUMENT;
188 }
189 }
190
191 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
192 if (type == nullptr) {
193 *out_type = nullptr;
194 return INFIX_ERROR_ALLOCATION_FAILED;
195 }
196
197 // Copy the caller's member data into the arena to make this type object self-contained.
198 infix_struct_member * arena_members = nullptr;
199 if (num_members > 0) {
200 arena_members =
201 infix_arena_alloc(arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
202 if (arena_members == nullptr) {
203 *out_type = nullptr;
204 return INFIX_ERROR_ALLOCATION_FAILED;
205 }
206 memcpy(arena_members, members, sizeof(infix_struct_member) * num_members);
207 // Deep copy the member names to make the type fully self-contained.
208 for (size_t i = 0; i < num_members; ++i) {
209 const char * src_name = members[i].name;
210 if (src_name) {
211 size_t name_len = strlen(src_name) + 1;
212 char * dest_name = infix_arena_alloc(arena, name_len, 1);
213 if (!dest_name) {
214 *out_type = nullptr;
215 return INFIX_ERROR_ALLOCATION_FAILED;
216 }
217 memcpy(dest_name, src_name, name_len);
218 arena_members[i].name = dest_name;
219 }
220 }
221 }
222
223 *out_type = type;
224 *out_arena_members = arena_members;
225 return INFIX_SUCCESS;
226}

251/*
252 * Implementation for infix_type_create_array.
253 * Calculates array layout, including a critical security check to prevent integer
254 * overflow when calculating the total size from untrusted inputs (e.g., a fuzzer).
255 */
256c23_nodiscard infix_status infix_type_create_array(infix_arena_t * arena,
257 infix_type ** out_type,
258 infix_type * element_type,
259 size_t num_elements) {
260 if (out_type == nullptr || element_type == nullptr)
261 return INFIX_ERROR_INVALID_ARGUMENT;
262
263 // Security: Check for integer overflow before calculating the total array size.
264 if (element_type->size > 0 && num_elements > SIZE_MAX / element_type->size) {
265 *out_type = nullptr;
266 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
267 return INFIX_ERROR_INVALID_ARGUMENT;
268 }
269
270 infix_type * type = infix_arena_alloc(arena, sizeof(infix_type), _Alignof(infix_type));
271 if (type == nullptr) {
272 *out_type = nullptr;
273 return INFIX_ERROR_ALLOCATION_FAILED;
274 }
275
276 type->is_arena_allocated = true;
277 type->category = INFIX_TYPE_ARRAY;
278 type->meta.array_info.element_type = element_type;
279 type->meta.array_info.num_elements = num_elements;
280 type->alignment = element_type->alignment;
281 type->size = element_type->size * num_elements;
282
283 *out_type = type;
284 return INFIX_SUCCESS;
285}

389/*
390 * Implementation for infix_type_create_union.
391 * This function calculates the size and alignment for a union according to standard
392 * C layout rules: size is the max member size (padded), alignment is the max member alignment.
393 */
394c23_nodiscard infix_status infix_type_create_union(infix_arena_t * arena,
395 infix_type ** out_type,
396 infix_struct_member * members,
397 size_t num_members) {
398 infix_type * type = nullptr;
399 infix_struct_member * arena_members = nullptr;
400 infix_status status = _create_aggregate_setup(arena, &type, &arena_members, members, num_members);
401 if (status != INFIX_SUCCESS) {
402 *out_type = nullptr;
403 return status;
404 }
405
406 type->is_arena_allocated = true;
407 type->category = INFIX_TYPE_UNION;
408 type->meta.aggregate_info.name = nullptr; // Initialize name for anonymous union
409 type->meta.aggregate_info.members = arena_members;
410 type->meta.aggregate_info.num_members = num_members;
411
412 size_t max_size = 0;
413 size_t max_alignment = 1;
414
415 // A union's size is determined by its largest member, and its alignment
416 // by the strictest alignment of any member.
417 for (size_t i = 0; i < num_members; ++i) {
418 arena_members[i].offset = 0; // All union members are at offset 0.
419 if (arena_members[i].type->size > max_size)
420 max_size = arena_members[i].type->size;
421 if (arena_members[i].type->alignment > max_alignment)
422 max_alignment = arena_members[i].type->alignment;
423 }
424 type->alignment = max_alignment;
425
426 // The final size is the size of the largest member, rounded up to a
427 // multiple of the union's overall alignment.
428 type->size = _infix_align_up(max_size, max_alignment);
429 if (type->size < max_size) { // Overflow check
430 *out_type = nullptr;
431 return INFIX_ERROR_INVALID_ARGUMENT;
432 }
433
434 INFIX_DEBUG_PRINTF("Created arena union type. Size: %llu, Alignment: %llu",
435 (unsigned long long)type->size,
436 (unsigned long long)type->alignment);
437
438 *out_type = type;
439 return INFIX_SUCCESS;
440}

442/*
443 * Implementation for infix_type_create_struct.
444 * This function calculates the size and alignment of the struct based on its
445 * members, adhering to standard C layout and padding rules.
446 */
447c23_nodiscard infix_status infix_type_create_struct(infix_arena_t * arena,
448 infix_type ** out_type,
449 infix_struct_member * members,
450 size_t num_members) {
451 _infix_clear_error();
452 infix_type * type = nullptr;
453 infix_struct_member * arena_members = nullptr;
454 infix_status status = _create_aggregate_setup(arena, &type, &arena_members, members, num_members);
455 if (status != INFIX_SUCCESS) {
456 *out_type = nullptr;
457 return status;
458 }
459
460 type->is_arena_allocated = true;
461 type->category = INFIX_TYPE_STRUCT;
462 type->meta.aggregate_info.name = nullptr; // Initialize name for anonymous struct
463 type->meta.aggregate_info.members = arena_members;
464 type->meta.aggregate_info.num_members = num_members;
465
466 size_t current_offset = 0;
467 size_t max_alignment = 1; // Empty structs have an alignment of 1.
468
469 // Calculate layout based on the safe, arena-allocated copy of members.
470 for (size_t i = 0; i < num_members; ++i) {
471 infix_struct_member * member = &arena_members[i];
472 size_t member_align = member->type->alignment;
473
474 if (member_align == 0) {
475 *out_type = nullptr;
476 return INFIX_ERROR_INVALID_ARGUMENT;
477 }
478
479 // Align the current offset for this member.
480 size_t aligned_offset = _infix_align_up(current_offset, member_align);
481
482 // Security: Check for integer overflow during alignment.
483 if (aligned_offset < current_offset) {
484 *out_type = nullptr;
485 return INFIX_ERROR_INVALID_ARGUMENT;
486 }
487 current_offset = aligned_offset;
488 member->offset = current_offset;
489
490 // Security: Check for integer overflow before adding the member's size.
491 if (current_offset > SIZE_MAX - member->type->size) {
492 *out_type = nullptr;
493 return INFIX_ERROR_INVALID_ARGUMENT;
494 }
495 current_offset += member->type->size;
496
497 if (member_align > max_alignment)
498 max_alignment = member_align;
499 }
500
501 type->alignment = max_alignment;
502
503 // The final size is the calculated offset rounded up to the nearest multiple
504 // of the struct's overall alignment to account for trailing padding.
505 type->size = _infix_align_up(current_offset, max_alignment);
506 // Security: Check for overflow during final alignment.
507 if (type->size < current_offset) {
508 *out_type = nullptr;
509 return INFIX_ERROR_INVALID_ARGUMENT;
510 }
511
512 *out_type = type;
513 return INFIX_SUCCESS;
514}

561 * Implementation for infix_type_create_named_reference.

563 * A type graph containing this node cannot be used to generate a trampoline.

566 infix_type ** out_type,
567 const char * name,
568 infix_aggregate_category_t agg_cat) {

573 if (type == nullptr) {

596 type->meta.named_reference.aggregate_category = agg_cat;

612infix_type * _copy_type_graph_to_arena(infix_arena_t * dest_arena, const infix_type * src_type) {
613 if (src_type == nullptr)
614 return nullptr;
615
616 // Optimization: If the source type is a static primitive, we don't need to copy it.
617 if (!src_type->is_arena_allocated)
618 return (infix_type *)src_type;
619
620 // Allocate space for the new type in the destination arena and copy the base struct.
621 infix_type * dest_type = infix_arena_alloc(dest_arena, sizeof(infix_type), _Alignof(infix_type));
622 if (dest_type == nullptr)
623 return nullptr;
624 memcpy(dest_type, src_type, sizeof(infix_type));
625
626 // Recursively copy any nested types based on the category.
627 switch (src_type->category) {
628 case INFIX_TYPE_POINTER:
629 dest_type->meta.pointer_info.pointee_type =
630 _copy_type_graph_to_arena(dest_arena, src_type->meta.pointer_info.pointee_type);
631 break;
632 case INFIX_TYPE_ARRAY:
633 dest_type->meta.array_info.element_type =
634 _copy_type_graph_to_arena(dest_arena, src_type->meta.array_info.element_type);
635 break;
636 case INFIX_TYPE_STRUCT:
637 case INFIX_TYPE_UNION:
638 {
639 // Deep copy the aggregate's own name, if it has one.
640 const char * src_agg_name = src_type->meta.aggregate_info.name;
641 if (src_agg_name) {
642 size_t name_len = strlen(src_agg_name) + 1;
643 char * dest_agg_name = infix_arena_alloc(dest_arena, name_len, 1);
644 if (!dest_agg_name)
645 return nullptr; // Allocation failed
646 memcpy(dest_agg_name, src_agg_name, name_len);
647 // We need to cast away const to write to the destination struct field.
648 *((const char **)&dest_type->meta.aggregate_info.name) = dest_agg_name;
649 }
650
651 if (src_type->meta.aggregate_info.num_members > 0) {
652 size_t members_size = sizeof(infix_struct_member) * src_type->meta.aggregate_info.num_members;
653 dest_type->meta.aggregate_info.members =
654 infix_arena_alloc(dest_arena, members_size, _Alignof(infix_struct_member));
655 if (dest_type->meta.aggregate_info.members == nullptr)
656 return nullptr;
657 memcpy(dest_type->meta.aggregate_info.members, src_type->meta.aggregate_info.members, members_size);
658
659 for (size_t i = 0; i < src_type->meta.aggregate_info.num_members; ++i) {
660 dest_type->meta.aggregate_info.members[i].type =
661 _copy_type_graph_to_arena(dest_arena, src_type->meta.aggregate_info.members[i].type);
662 // Deep copy the member name string to avoid dangling pointers.
663 const char * src_name = src_type->meta.aggregate_info.members[i].name;
664 if (src_name) {
665 size_t name_len = strlen(src_name) + 1;
666 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
667 if (!dest_name)
668 return nullptr;
669 memcpy(dest_name, src_name, name_len);
670 *((const char **)&dest_type->meta.aggregate_info.members[i].name) = dest_name;
671 }
672 }
673 }
674 }
675 break;
676 case INFIX_TYPE_REVERSE_TRAMPOLINE:
677 dest_type->meta.func_ptr_info.return_type =
678 _copy_type_graph_to_arena(dest_arena, src_type->meta.func_ptr_info.return_type);
679 if (src_type->meta.func_ptr_info.num_args > 0) {
680 size_t args_size = sizeof(infix_function_argument) * src_type->meta.func_ptr_info.num_args;
681 dest_type->meta.func_ptr_info.args =
682 infix_arena_alloc(dest_arena, args_size, _Alignof(infix_function_argument));
683 if (dest_type->meta.func_ptr_info.args == nullptr)
684 return nullptr;
685 memcpy(dest_type->meta.func_ptr_info.args, src_type->meta.func_ptr_info.args, args_size);
686
687 for (size_t i = 0; i < src_type->meta.func_ptr_info.num_args; ++i) {
688 dest_type->meta.func_ptr_info.args[i].type =
689 _copy_type_graph_to_arena(dest_arena, src_type->meta.func_ptr_info.args[i].type);
690 const char * src_name = src_type->meta.func_ptr_info.args[i].name;
691 if (src_name) {
692 size_t name_len = strlen(src_name) + 1;
693 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
694 if (!dest_name)
695 return nullptr;
696 memcpy(dest_name, src_name, name_len);
697 *((const char **)&dest_type->meta.func_ptr_info.args[i].name) = dest_name;
698 }
699 }
700 }
701 break;
702 default:
703 // For PRIMITIVE, VOID, ENUM, etc., the initial shallow copy is sufficient.
704 break;
705 }
706 return dest_type;
707}
