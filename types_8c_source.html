231static bool _layout_struct(infix_type * type) {
232 size_t current_byte_offset = 0;
233 size_t current_unit_offset = 0;
234 size_t current_unit_size = 0;
235 uint32_t current_unit_bits_used = 0;
236 size_t max_alignment = 1;
237 bool in_bitfield = false;
238
239 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
240 infix_struct_member * member = &type->meta.aggregate_info.members[i];
241 infix_type * mtype = member->type;
242
243 if (member->is_bitfield) {
244 size_t align = mtype->alignment;
245 if (align == 0)
246 align = 1;
247 if (align > max_alignment && !type->meta.aggregate_info.is_packed)
248 max_alignment = align;
249
250 // Zero-width bitfield or type mismatch or doesn't fit -> start new unit
251 if (member->bit_width == 0 || !in_bitfield || mtype->size != current_unit_size ||
252 current_unit_bits_used + member->bit_width > mtype->size * 8) {
253
254 // Align to start of new unit
255 size_t start = _infix_align_up(current_byte_offset, align);
256 if (start < current_byte_offset) {
257 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
258 return false;
259 }
260
261 current_unit_offset = start;
262 current_unit_size = mtype->size;
263 current_unit_bits_used = 0;
264 in_bitfield = true;
265
266 current_byte_offset = current_unit_offset;
267 }
268
269 member->offset = current_unit_offset + (current_unit_bits_used / 8);
270 member->bit_offset = (uint8_t)(current_unit_bits_used % 8);
271 current_unit_bits_used += member->bit_width;
272
273 size_t bytes_used = (current_unit_bits_used + 7) / 8;
274 if (current_unit_offset + bytes_used > current_byte_offset)
275 current_byte_offset = current_unit_offset + bytes_used;
276 }
277 else {
278 // Handle Flexible Array Members (FAM)
279 if (mtype->category == INFIX_TYPE_ARRAY && mtype->meta.array_info.is_flexible) {
280 in_bitfield = false;
281 size_t align = mtype->alignment;
282 if (align == 0)
283 align = 1;
284
285 if (align > max_alignment && !type->meta.aggregate_info.is_packed)
286 max_alignment = align;
287
288 size_t aligned = _infix_align_up(current_byte_offset, align);
289 if (aligned < current_byte_offset) {
290 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
291 return false;
292 }
293 current_byte_offset = aligned;
294 member->offset = current_byte_offset;
295 member->bit_offset = 0;
296 continue;
297 }
298
299 // Standard Member
300 in_bitfield = false;
301 size_t align = mtype->alignment;
302 if (align == 0)
303 align = 1;
304
305 if (align > max_alignment && !type->meta.aggregate_info.is_packed)
306 max_alignment = align;
307
308 size_t aligned = _infix_align_up(current_byte_offset, align);
309 if (aligned < current_byte_offset) {
310 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
311 return false;
312 }
313 current_byte_offset = aligned;
314 member->offset = current_byte_offset;
315 member->bit_offset = 0;
316
317 if (current_byte_offset > SIZE_MAX - mtype->size) {
318 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
319 return false;
320 }
321 current_byte_offset += mtype->size;
322 }
323 }
324
325 // If it is packed, the alignment is explicitly determined by the user.
326 if (type->meta.aggregate_info.is_packed)
327 max_alignment = type->alignment;
328
329 type->alignment = max_alignment;
330 type->size = _infix_align_up(current_byte_offset, max_alignment);
331 return true;
332}
350static infix_status _create_aggregate_setup(infix_arena_t * arena,
351 infix_type ** out_type,
352 infix_struct_member ** out_arena_members,
353 infix_struct_member * members,
354 size_t num_members) {
355 if (out_type == nullptr)
356 return INFIX_ERROR_INVALID_ARGUMENT;
357 // Pre-flight check: ensure all provided member types are valid.
358 for (size_t i = 0; i < num_members; ++i) {
359 if (members[i].type == nullptr) {
360 *out_type = nullptr;
361 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INVALID_MEMBER_TYPE, 0);
362 return INFIX_ERROR_INVALID_ARGUMENT;
363 }

370 }

372 if (num_members > 0) {

375 if (arena_members == nullptr) {
376 *out_type = nullptr;
377 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
378 return INFIX_ERROR_ALLOCATION_FAILED;
379 }

381 }

384 return INFIX_SUCCESS;

606INFIX_API c23_nodiscard infix_status infix_type_create_union(infix_arena_t * arena,
607 infix_type ** out_type,
608 infix_struct_member * members,
609 size_t num_members) {
610 infix_type * type = nullptr;
611 infix_struct_member * arena_members = nullptr;
612 infix_status status = _create_aggregate_setup(arena, &type, &arena_members, members, num_members);
613 if (status != INFIX_SUCCESS) {
614 *out_type = nullptr;
615 return status;
616 }
617 type->is_arena_allocated = true;
618 type->category = INFIX_TYPE_UNION;
619 type->meta.aggregate_info.members = arena_members;
620 type->meta.aggregate_info.num_members = num_members;
621 type->meta.aggregate_info.is_packed = false; // Unions don't use this flag currently
622 // A union's size is the size of its largest member, and its alignment is the
623 // alignment of its most-aligned member.
624 size_t max_size = 0;
625 size_t max_alignment = 1;
626 for (size_t i = 0; i < num_members; ++i) {
627 arena_members[i].offset = 0; // All union members have an offset of 0.
628 if (arena_members[i].type->size > max_size)
629 max_size = arena_members[i].type->size;
630 if (arena_members[i].type->alignment > max_alignment)
631 max_alignment = arena_members[i].type->alignment;
632 }
633 type->alignment = max_alignment;
634 // The total size is the size of the largest member, padded up to the required alignment.
635 type->size = _infix_align_up(max_size, max_alignment);
636 // Overflow check
637 if (type->size < max_size) {
638 *out_type = nullptr;
639 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
640 return INFIX_ERROR_INVALID_ARGUMENT;
641 }
642 *out_type = type;
643 return INFIX_SUCCESS;
644}
653INFIX_API c23_nodiscard infix_status infix_type_create_struct(infix_arena_t * arena,
654 infix_type ** out_type,
655 infix_struct_member * members,
656 size_t num_members) {
657 _infix_clear_error();
658 infix_type * type = nullptr;
659 infix_struct_member * arena_members = nullptr;
660 infix_status status = _create_aggregate_setup(arena, &type, &arena_members, members, num_members);
661 if (status != INFIX_SUCCESS) {
662 *out_type = nullptr;
663 return status;
664 }
665 type->is_arena_allocated = true;
666 type->category = INFIX_TYPE_STRUCT;
667 type->meta.aggregate_info.members = arena_members;
668 type->meta.aggregate_info.num_members = num_members;
669 type->meta.aggregate_info.is_packed = false;

671 // This performs a preliminary layout calculation.
672 // Note: This layout may be incomplete if it contains unresolved named references or flexible arrays.
673 // The final, correct layout will be computed by `_infix_type_recalculate_layout`.
674 // However, we must set a preliminary size/alignment here.
675
676 // We use the recalculate logic to do the heavy lifting, assuming a temporary arena can be made.
677 // But we can't create an arena here easily if we are in a strict context.
678 // So we do a simplified pass just like the old logic, ignoring complex bitfield rules for now.
679 // The proper bitfield layout happens in `_infix_type_recalculate_layout`.

682 infix_struct_member * member = &arena_members[i];
683 if (member->type->alignment == 0 && member->type->category != INFIX_TYPE_NAMED_REFERENCE &&
684 !(member->type->category == INFIX_TYPE_ARRAY && member->type->meta.array_info.is_flexible)) {
685 if (member->type->category != INFIX_TYPE_ARRAY) {
686 *out_type = nullptr;
687 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INVALID_MEMBER_TYPE, 0);
688 return INFIX_ERROR_INVALID_ARGUMENT;
689 }
690 }
691 }

694 if (!_layout_struct(type)) {
695 *out_type = nullptr;

699 *out_type = type;

712INFIX_API c23_nodiscard infix_status infix_type_create_packed_struct(infix_arena_t * arena,
713 infix_type ** out_type,
714 size_t total_size,
715 size_t alignment,
716 infix_struct_member * members,
717 size_t num_members) {
718 if (out_type == nullptr || (num_members > 0 && members == nullptr)) {
719 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_NULL_POINTER, 0);
720 return INFIX_ERROR_INVALID_ARGUMENT;
721 }
722 // Validate alignment is power-of-two AND within sane limits.
723 if (alignment == 0 || (alignment & (alignment - 1)) != 0) {
724 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_INVALID_ALIGNMENT, 0);
725 return INFIX_ERROR_INVALID_ARGUMENT;
726 }
727 if (alignment > INFIX_MAX_ALIGNMENT) {
728 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_TYPE_TOO_LARGE, 0);
729 return INFIX_ERROR_INVALID_ARGUMENT;
730 }
731 infix_type * type = infix_arena_calloc(arena, 1, sizeof(infix_type), _Alignof(infix_type));
732 if (type == nullptr) {
733 *out_type = nullptr;
734 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
735 return INFIX_ERROR_ALLOCATION_FAILED;
736 }
737 infix_struct_member * arena_members = nullptr;
738 if (num_members > 0) {
739 arena_members =
740 infix_arena_alloc(arena, sizeof(infix_struct_member) * num_members, _Alignof(infix_struct_member));
741 if (arena_members == nullptr) {
742 *out_type = nullptr;
743 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
744 return INFIX_ERROR_ALLOCATION_FAILED;
745 }
746 infix_memcpy(arena_members, members, sizeof(infix_struct_member) * num_members);
747 }
748 type->is_arena_allocated = true;
749 type->size = total_size;
750 type->alignment = alignment;
751 type->category = INFIX_TYPE_STRUCT; // Packed structs are still fundamentally structs.
752 type->meta.aggregate_info.members = arena_members;
753 type->meta.aggregate_info.num_members = num_members;
754 type->meta.aggregate_info.is_packed = true; // Marked as packed
755 *out_type = type;
756 return INFIX_SUCCESS;
757}

801// Internal Type Graph Management
814typedef struct recalc_visited_node_t {
815 infix_type * type;
816 struct recalc_visited_node_t * next;
817} recalc_visited_node_t;
840static void _infix_type_recalculate_layout_recursive(infix_arena_t * temp_arena,
841 infix_type * type,
842 recalc_visited_node_t ** visited_head) {
843 if (!type || !type->is_arena_allocated)
844 return; // Base case: Don't modify static singleton types.
845 // Cycle detection: If we have already visited this node in the current recursion
846 // path, we are in a cycle. Return immediately to break the loop. The layout of
847 // this node will be calculated when the recursion unwinds to its first visit.
848 for (recalc_visited_node_t * v = *visited_head; v != nullptr; v = v->next)
849 if (v->type == type)
850 return;
851 // Allocate the memoization node from a stable temporary arena.
852 recalc_visited_node_t * visited_node =
853 infix_arena_alloc(temp_arena, sizeof(recalc_visited_node_t), _Alignof(recalc_visited_node_t));
854 if (!visited_node)
855 return; // Cannot proceed without memory.
856 visited_node->type = type;
857 visited_node->next = *visited_head;
858 *visited_head = visited_node;
859 // Recurse into child types first (post-order traversal).
860 switch (type->category) {
861 case INFIX_TYPE_POINTER:
862 _infix_type_recalculate_layout_recursive(temp_arena, type->meta.pointer_info.pointee_type, visited_head);
863 break;
864 case INFIX_TYPE_ARRAY:
865 _infix_type_recalculate_layout_recursive(temp_arena, type->meta.array_info.element_type, visited_head);
866 break;
867 case INFIX_TYPE_STRUCT:
868 case INFIX_TYPE_UNION:
869 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
870 _infix_type_recalculate_layout_recursive(
871 temp_arena, type->meta.aggregate_info.members[i].type, visited_head);
872 }
873 break;
874 default:
875 break; // Other types have no child types to recurse into.
876 }
877 // After children are updated, recalculate this type's layout.
878 if (type->category == INFIX_TYPE_STRUCT)
879 _layout_struct(type);
880 else if (type->category == INFIX_TYPE_UNION) {
881 size_t max_size = 0;
882 size_t max_alignment = 1;
883 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
884 infix_type * member_type = type->meta.aggregate_info.members[i].type;
885 if (member_type->size > max_size)
886 max_size = member_type->size;
887 if (member_type->alignment > max_alignment)
888 max_alignment = member_type->alignment;
889 }
890 type->alignment = max_alignment;
891 type->size = _infix_align_up(max_size, max_alignment);
892 }
893 else if (type->category == INFIX_TYPE_ARRAY) {
894 // Flexible arrays have size 0 but inherit alignment.
895 // Fixed arrays calculate size normally.
896 if (type->meta.array_info.is_flexible) {
897 type->alignment = type->meta.array_info.element_type->alignment;
898 type->size = 0;
899 }
900 else {
901 type->alignment = type->meta.array_info.element_type->alignment;
902 type->size = type->meta.array_info.element_type->size * type->meta.array_info.num_elements;
903 }
904 }
905}
915void _infix_type_recalculate_layout(infix_type * type) {
916 // Create a temporary arena solely for the visited list's lifetime.
917 infix_arena_t * temp_arena = infix_arena_create(1024);
918 if (!temp_arena)
919 return;
920 recalc_visited_node_t * visited_head = nullptr;
921 _infix_type_recalculate_layout_recursive(temp_arena, type, &visited_head);
922 infix_arena_destroy(temp_arena);
923}
932typedef struct memo_node_t {
933 const infix_type * src;
934 infix_type * dest;
935 struct memo_node_t * next;
936} memo_node_t;
951static infix_type * _copy_type_graph_to_arena_recursive(infix_arena_t * dest_arena,
952 const infix_type * src_type,
953 memo_node_t ** memo_head) {
954 if (src_type == nullptr)
955 return nullptr;
956 // If the source type lives in the same arena as our destination, we can safely share the pointer instead of
957 // performing a deep copy.
958 if (src_type->arena == dest_arena)
959 return (infix_type *)src_type;
960 // Base case: Static types don't need to be copied; return the singleton pointer.
961 if (!src_type->is_arena_allocated)
962 return (infix_type *)src_type;
963 // Check memoization table: if we've already copied this node, return the existing copy.
964 // This correctly handles cycles and shared sub-graphs.
965 for (memo_node_t * node = *memo_head; node != NULL; node = node->next)
966 if (node->src == src_type)
967 return node->dest;
968 // Allocate the new type object in the destination arena.
969 infix_type * dest_type = infix_arena_calloc(dest_arena, 1, sizeof(infix_type), _Alignof(infix_type));
970 if (dest_type == nullptr)
971 return nullptr;
972 // Add this new pair to the memoization table BEFORE recursing. This is crucial
973 // for handling cycles: the recursive call will find this entry and return `dest_type`.
974 memo_node_t * new_memo_node = infix_arena_alloc(dest_arena, sizeof(memo_node_t), _Alignof(memo_node_t));
975 if (!new_memo_node)
976 return nullptr;
977 new_memo_node->src = src_type;
978 new_memo_node->dest = dest_type;
979 new_memo_node->next = *memo_head;
980 *memo_head = new_memo_node;
981 // Perform a shallow copy of the main struct, then recurse to deep copy child pointers.
982 *dest_type = *src_type;
983 dest_type->is_arena_allocated = true;
984 dest_type->is_incomplete = src_type->is_incomplete;
985 dest_type->arena = dest_arena; // The new type now belongs to the destination arena.
986 // Deep copy the semantic name string, if it exists.
987 if (src_type->name) {
988 size_t name_len = strlen(src_type->name) + 1;
989 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
990 if (!dest_name)
991 return nullptr; // Allocation failed
992 infix_memcpy((void *)dest_name, src_type->name, name_len);
993 dest_type->name = dest_name;
994 }
995 switch (src_type->category) {
996 case INFIX_TYPE_POINTER:
997 dest_type->meta.pointer_info.pointee_type =
998 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.pointer_info.pointee_type, memo_head);
999 break;
1000 case INFIX_TYPE_ARRAY:
1001 dest_type->meta.array_info.element_type =
1002 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.array_info.element_type, memo_head);
1003 // Explicitly copy the flexible flag to ensure it persists.
1004 dest_type->meta.array_info.is_flexible = src_type->meta.array_info.is_flexible;
1005 break;
1006 case INFIX_TYPE_STRUCT:
1007 case INFIX_TYPE_UNION:
1008 if (src_type->meta.aggregate_info.num_members > 0) {
1009 // Copy the members array itself.
1010 size_t members_size = sizeof(infix_struct_member) * src_type->meta.aggregate_info.num_members;
1011 dest_type->meta.aggregate_info.members =
1012 infix_arena_alloc(dest_arena, members_size, _Alignof(infix_struct_member));
1013 if (dest_type->meta.aggregate_info.members == nullptr)
1014 return nullptr;
1015 dest_type->meta.aggregate_info.is_packed = src_type->meta.aggregate_info.is_packed; // Copy packed flag
1016 // Now, recurse for each member's type and copy its name.
1017 for (size_t i = 0; i < src_type->meta.aggregate_info.num_members; ++i) {
1018 dest_type->meta.aggregate_info.members[i] = src_type->meta.aggregate_info.members[i];
1019 dest_type->meta.aggregate_info.members[i].type = _copy_type_graph_to_arena_recursive(
1020 dest_arena, src_type->meta.aggregate_info.members[i].type, memo_head);
1021 const char * src_name = src_type->meta.aggregate_info.members[i].name;
1022 if (src_name) {
1023 size_t name_len = strlen(src_name) + 1;
1024 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
1025 if (!dest_name)
1026 return nullptr;
1027 infix_memcpy((void *)dest_name, src_name, name_len);
1028 dest_type->meta.aggregate_info.members[i].name = dest_name;
1029 }
1030 // Copy bitfield properties
1031 dest_type->meta.aggregate_info.members[i].bit_width =
1032 src_type->meta.aggregate_info.members[i].bit_width;
1033 dest_type->meta.aggregate_info.members[i].bit_offset =
1034 src_type->meta.aggregate_info.members[i].bit_offset;
1035 dest_type->meta.aggregate_info.members[i].is_bitfield =
1036 src_type->meta.aggregate_info.members[i].is_bitfield;
1037 }
1038 }
1039 break;
1040 case INFIX_TYPE_NAMED_REFERENCE:
1041 {
1042 const char * src_name = src_type->meta.named_reference.name;
1043 if (src_name) {
1044 size_t name_len = strlen(src_name) + 1;
1045 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
1046 if (!dest_name)
1047 return nullptr;
1048 infix_memcpy((void *)dest_name, src_name, name_len);
1049 dest_type->meta.named_reference.name = dest_name;
1050 }
1051 break;
1052 }
1053 case INFIX_TYPE_REVERSE_TRAMPOLINE:
1054 dest_type->meta.func_ptr_info.return_type =
1055 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.func_ptr_info.return_type, memo_head);
1056 if (src_type->meta.func_ptr_info.num_args > 0) {
1057 size_t args_size = sizeof(infix_function_argument) * src_type->meta.func_ptr_info.num_args;
1058 dest_type->meta.func_ptr_info.args =
1059 infix_arena_alloc(dest_arena, args_size, _Alignof(infix_function_argument));
1060 if (dest_type->meta.func_ptr_info.args == nullptr)
1061 return nullptr;
1062 for (size_t i = 0; i < src_type->meta.func_ptr_info.num_args; ++i) {
1063 dest_type->meta.func_ptr_info.args[i] = src_type->meta.func_ptr_info.args[i];
1064 dest_type->meta.func_ptr_info.args[i].type = _copy_type_graph_to_arena_recursive(
1065 dest_arena, src_type->meta.func_ptr_info.args[i].type, memo_head);
1066 const char * src_name = src_type->meta.func_ptr_info.args[i].name;
1067 if (src_name) {
1068 size_t name_len = strlen(src_name) + 1;
1069 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
1070 if (!dest_name)
1071 return nullptr;
1072 infix_memcpy((void *)dest_name, src_name, name_len);
1073 dest_type->meta.func_ptr_info.args[i].name = dest_name;
1074 }
1075 }
1076 }
1077 break;
1078 case INFIX_TYPE_ENUM:
1079 dest_type->meta.enum_info.underlying_type =
1080 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.enum_info.underlying_type, memo_head);
1081 break;
1082 case INFIX_TYPE_COMPLEX:
1083 dest_type->meta.complex_info.base_type =
1084 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.complex_info.base_type, memo_head);
1085 break;
1086 case INFIX_TYPE_VECTOR:
1087 dest_type->meta.vector_info.element_type =
1088 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.vector_info.element_type, memo_head);
1089 break;
1090 default:
1091 // Other types like primitives have no child pointers to copy.
1092 break;
1093 }
1094 return dest_type;
1095}
1103infix_type * _copy_type_graph_to_arena(infix_arena_t * dest_arena, const infix_type * src_type) {
1104 memo_node_t * memo_head = nullptr;
1105 return _copy_type_graph_to_arena_recursive(dest_arena, src_type, &memo_head);
1106}
1112typedef struct estimate_visited_node_t {
1113 const infix_type * type;
1114 struct estimate_visited_node_t * next;
1115} estimate_visited_node_t;
1128static size_t _estimate_graph_size_recursive(infix_arena_t * temp_arena,
1129 const infix_type * type,
1130 estimate_visited_node_t ** visited_head) {
1131 if (!type || !type->is_arena_allocated)
1132 return 0;
1133 // Cycle detection: if we've seen this node, it's already accounted for.
1134 for (estimate_visited_node_t * v = *visited_head; v != NULL; v = v->next)
1135 if (v->type == type)
1136 return 0;
1137 // Add this node to the visited list before recursing.
1138 estimate_visited_node_t * visited_node =
1139 infix_arena_alloc(temp_arena, sizeof(estimate_visited_node_t), _Alignof(estimate_visited_node_t));
1140 if (!visited_node) {
1141 // On allocation failure, we can't proceed with estimation. Return a large
1142 // number to ensure the caller allocates a fallback-sized arena.
1143 return 65536;
1144 }
1145 visited_node->type = type;
1146 visited_node->next = *visited_head;
1147 *visited_head = visited_node;
1148 // The size includes the type object itself, a memoization node, and the name string if it exists.
1149 size_t total_size = sizeof(infix_type) + sizeof(memo_node_t);
1150 if (type->name)
1151 total_size += strlen(type->name) + 1;
1152 switch (type->category) {
1153 case INFIX_TYPE_POINTER:
1154 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.pointer_info.pointee_type, visited_head);
1155 break;
1156 case INFIX_TYPE_ARRAY:
1157 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.array_info.element_type, visited_head);
1158 break;
1159 case INFIX_TYPE_STRUCT:
1160 case INFIX_TYPE_UNION:
1161 if (type->meta.aggregate_info.num_members > 0) {
1162 total_size += sizeof(infix_struct_member) * type->meta.aggregate_info.num_members;
1163 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
1164 const infix_struct_member * member = &type->meta.aggregate_info.members[i];
1165 if (member->name)
1166 total_size += strlen(member->name) + 1;
1167 total_size += _estimate_graph_size_recursive(temp_arena, member->type, visited_head);
1168 }
1169 }
1170 break;
1171 case INFIX_TYPE_NAMED_REFERENCE:
1172 if (type->meta.named_reference.name)
1173 total_size += strlen(type->meta.named_reference.name) + 1;
1174 break;
1175 case INFIX_TYPE_REVERSE_TRAMPOLINE:
1176 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.func_ptr_info.return_type, visited_head);
1177 if (type->meta.func_ptr_info.num_args > 0) {
1178 total_size += sizeof(infix_function_argument) * type->meta.func_ptr_info.num_args;
1179 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i) {
1180 const infix_function_argument * arg = &type->meta.func_ptr_info.args[i];
1181 if (arg->name)
1182 total_size += strlen(arg->name) + 1;
1183 total_size += _estimate_graph_size_recursive(temp_arena, arg->type, visited_head);
1184 }
1185 }
1186 break;
1187 case INFIX_TYPE_ENUM:
1188 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.enum_info.underlying_type, visited_head);
1189 break;
1190 case INFIX_TYPE_COMPLEX:
1191 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.complex_info.base_type, visited_head);
1192 break;
1193 case INFIX_TYPE_VECTOR:
1194 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.vector_info.element_type, visited_head);
1195 break;
1196 default:
1197 break;
1198 }
1199 return total_size;
1200}
1208size_t _infix_estimate_graph_size(infix_arena_t * temp_arena, const infix_type * type) {
