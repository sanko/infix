424 infix_type ** out_type,

441 size_t max_alignment = 1;

443 arena_members[i].offset = 0; // All union members have an offset of 0.
444 if (arena_members[i].type->size > max_size)
445 max_size = arena_members[i].type->size;
446 if (arena_members[i].type->alignment > max_alignment)
447 max_alignment = arena_members[i].type->alignment;
448 }
449 type->alignment = max_alignment;
450 // The total size is the size of the largest member, padded up to the required alignment.
451 type->size = _infix_align_up(max_size, max_alignment);
452 // Overflow check
453 if (type->size < max_size) {
454 *out_type = nullptr;
455 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
456 return INFIX_ERROR_INVALID_ARGUMENT;
457 }
458 *out_type = type;

469c23_nodiscard infix_status infix_type_create_struct(infix_arena_t * arena,
470 infix_type ** out_type,
471 infix_struct_member * members,
472 size_t num_members) {
473 _infix_clear_error();
474 infix_type * type = nullptr;
475 infix_struct_member * arena_members = nullptr;
476 infix_status status = _create_aggregate_setup(arena, &type, &arena_members, members, num_members);
477 if (status != INFIX_SUCCESS) {
478 *out_type = nullptr;
479 return status;
480 }
481 type->is_arena_allocated = true;
482 type->category = INFIX_TYPE_STRUCT;
483 type->meta.aggregate_info.members = arena_members;
484 type->meta.aggregate_info.num_members = num_members;
485 // This performs a preliminary layout calculation based on standard C packing rules.
486 // Note: This layout may be incomplete if it contains unresolved named references.
487 // The final, correct layout will be computed by `_infix_type_recalculate_layout`.
488 size_t current_offset = 0;
489 size_t max_alignment = 1;
490 for (size_t i = 0; i < num_members; ++i) {
491 infix_struct_member * member = &arena_members[i];
492 size_t member_align = member->type->alignment;
493 // An unresolved named reference will have alignment 0 or 1. If it's 0, we treat it as 1 for
494 // this preliminary pass to avoid division by zero. The real alignment will be fixed later.
495 if (member_align == 0) {
496 if (member->type->category != INFIX_TYPE_NAMED_REFERENCE) {
497 // A zero-alignment type that isn't a named reference is invalid (e.g., a struct with a void member).
498 *out_type = nullptr;
499 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INVALID_MEMBER_TYPE, 0);
500 return INFIX_ERROR_INVALID_ARGUMENT;
501 }
502 member_align = 1;
503 }
504 // Align the current offset up to the boundary required by the current member.
505 size_t aligned_offset = _infix_align_up(current_offset, member_align);
506 if (aligned_offset < current_offset) { // Overflow check
507 *out_type = nullptr;
508 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
509 return INFIX_ERROR_INVALID_ARGUMENT;
510 }
511 current_offset = aligned_offset;
512 member->offset = current_offset;
513 // Add the member's size to the current offset.
514 if (current_offset > SIZE_MAX - member->type->size) { // Overflow check
515 *out_type = nullptr;
516 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
517 return INFIX_ERROR_INVALID_ARGUMENT;
518 }
519 current_offset += member->type->size;
520 // Keep track of the largest alignment requirement.
521 if (member_align > max_alignment)
522 max_alignment = member_align;
523 }
524 // A struct's alignment is the alignment of its most-aligned member.
525 type->alignment = max_alignment;
526 // The struct's total size is its occupied space, padded up to its alignment.
527 type->size = _infix_align_up(current_offset, max_alignment);
528 if (type->size < current_offset) { // Overflow check
529 *out_type = nullptr;
530 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_INTEGER_OVERFLOW, 0);
531 return INFIX_ERROR_INVALID_ARGUMENT;
532 }
533 *out_type = type;
534 return INFIX_SUCCESS;
535}

664static void _infix_type_recalculate_layout_recursive(infix_arena_t * temp_arena,
665 infix_type * type,
666 recalc_visited_node_t ** visited_head) {
667 if (!type || !type->is_arena_allocated)
668 return; // Base case: Don't modify static singleton types.
669 // Cycle detection: If we have already visited this node in the current recursion
670 // path, we are in a cycle. Return immediately to break the loop. The layout of
671 // this node will be calculated when the recursion unwinds to its first visit.
672 for (recalc_visited_node_t * v = *visited_head; v != nullptr; v = v->next)
673 if (v->type == type)
674 return;
675 // Allocate the memoization node from a stable temporary arena.
676 recalc_visited_node_t * visited_node =
677 infix_arena_alloc(temp_arena, sizeof(recalc_visited_node_t), _Alignof(recalc_visited_node_t));
678 if (!visited_node)
679 return; // Cannot proceed without memory.
680 visited_node->type = type;
681 visited_node->next = *visited_head;
682 *visited_head = visited_node;
683 // Recurse into child types first (post-order traversal).
684 switch (type->category) {
685 case INFIX_TYPE_POINTER:
686 _infix_type_recalculate_layout_recursive(temp_arena, type->meta.pointer_info.pointee_type, visited_head);
687 break;
688 case INFIX_TYPE_ARRAY:
689 _infix_type_recalculate_layout_recursive(temp_arena, type->meta.array_info.element_type, visited_head);
690 break;
691 case INFIX_TYPE_STRUCT:
692 case INFIX_TYPE_UNION:
693 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
694 _infix_type_recalculate_layout_recursive(
695 temp_arena, type->meta.aggregate_info.members[i].type, visited_head);
696 }
697 break;
698 default:
699 break; // Other types have no child types to recurse into.
700 }
701 // After children are updated, recalculate this type's layout.
702 if (type->category == INFIX_TYPE_STRUCT) {
703 size_t current_offset = 0;
704 size_t max_alignment = 1;
705 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
706 infix_struct_member * member = &type->meta.aggregate_info.members[i];
707 size_t member_align = member->type->alignment;
708 if (member_align > max_alignment)
709 max_alignment = member_align;
710 current_offset = _infix_align_up(current_offset, member_align);
711 member->offset = current_offset;
712 current_offset += member->type->size;
713 }
714 type->alignment = max_alignment;
715 type->size = _infix_align_up(current_offset, max_alignment);
716 }
717 else if (type->category == INFIX_TYPE_UNION) {
718 size_t max_size = 0;
719 size_t max_alignment = 1;
720 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
721 infix_type * member_type = type->meta.aggregate_info.members[i].type;
722 if (member_type->size > max_size)
723 max_size = member_type->size;
724 if (member_type->alignment > max_alignment)
725 max_alignment = member_type->alignment;
726 }
727 type->alignment = max_alignment;
728 type->size = _infix_align_up(max_size, max_alignment);
729 }
730 else if (type->category == INFIX_TYPE_ARRAY) {
731 // Recalculate array size based on the (potentially updated) element size.
732 type->alignment = type->meta.array_info.element_type->alignment;
733 type->size = type->meta.array_info.element_type->size * type->meta.array_info.num_elements;
734 }
735}

762typedef struct memo_node_t {
763 const infix_type * src;
764 infix_type * dest;
765 struct memo_node_t * next;
766} memo_node_t;
782static infix_type * _copy_type_graph_to_arena_recursive(infix_arena_t * dest_arena,
783 const infix_type * src_type,
784 memo_node_t ** memo_head) {
785 if (src_type == nullptr)
786 return nullptr;
787 // If the source type lives in the same arena as our destination, we can safely share the pointer instead of
788 // performing a deep copy.
789 if (src_type->arena == dest_arena)
790 return (infix_type *)src_type;
791 // Base case: Static types don't need to be copied; return the singleton pointer.
792 if (!src_type->is_arena_allocated)
793 return (infix_type *)src_type;
794 // Check memoization table: if we've already copied this node, return the existing copy.
795 // This correctly handles cycles and shared sub-graphs.
796 for (memo_node_t * node = *memo_head; node != NULL; node = node->next)
797 if (node->src == src_type)
798 return node->dest;
799 // Allocate the new type object in the destination arena.
800 infix_type * dest_type = infix_arena_calloc(dest_arena, 1, sizeof(infix_type), _Alignof(infix_type));
801 if (dest_type == nullptr)
802 return nullptr;
803 // Add this new pair to the memoization table BEFORE recursing. This is crucial
804 // for handling cycles: the recursive call will find this entry and return `dest_type`.
805 memo_node_t * new_memo_node = infix_arena_alloc(dest_arena, sizeof(memo_node_t), _Alignof(memo_node_t));
806 if (!new_memo_node)
807 return nullptr;
808 new_memo_node->src = src_type;
809 new_memo_node->dest = dest_type;
810 new_memo_node->next = *memo_head;
811 *memo_head = new_memo_node;
812 // Perform a shallow copy of the main struct, then recurse to deep copy child pointers.
813 *dest_type = *src_type;
814 dest_type->is_arena_allocated = true;
815 dest_type->arena = dest_arena; // The new type now belongs to the destination arena.
816 // Deep copy the semantic name string, if it exists.
817 if (src_type->name) {
818 size_t name_len = strlen(src_type->name) + 1;
819 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
820 if (!dest_name)
821 return nullptr; // Allocation failed
822 infix_memcpy((void *)dest_name, src_type->name, name_len);
823 dest_type->name = dest_name;
824 }
825 switch (src_type->category) {
826 case INFIX_TYPE_POINTER:
827 dest_type->meta.pointer_info.pointee_type =
828 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.pointer_info.pointee_type, memo_head);
829 break;
830 case INFIX_TYPE_ARRAY:
831 dest_type->meta.array_info.element_type =
832 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.array_info.element_type, memo_head);
833 break;
834 case INFIX_TYPE_STRUCT:
835 case INFIX_TYPE_UNION:
836 if (src_type->meta.aggregate_info.num_members > 0) {
837 // Copy the members array itself.
838 size_t members_size = sizeof(infix_struct_member) * src_type->meta.aggregate_info.num_members;
839 dest_type->meta.aggregate_info.members =
840 infix_arena_alloc(dest_arena, members_size, _Alignof(infix_struct_member));
841 if (dest_type->meta.aggregate_info.members == nullptr)
842 return nullptr;
843 // Now, recurse for each member's type and copy its name.
844 for (size_t i = 0; i < src_type->meta.aggregate_info.num_members; ++i) {
845 dest_type->meta.aggregate_info.members[i] = src_type->meta.aggregate_info.members[i];
846 dest_type->meta.aggregate_info.members[i].type = _copy_type_graph_to_arena_recursive(
847 dest_arena, src_type->meta.aggregate_info.members[i].type, memo_head);
848 const char * src_name = src_type->meta.aggregate_info.members[i].name;
849 if (src_name) {
850 size_t name_len = strlen(src_name) + 1;
851 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
852 if (!dest_name)
853 return nullptr;
854 infix_memcpy((void *)dest_name, src_name, name_len);
855 dest_type->meta.aggregate_info.members[i].name = dest_name;
856 }
857 }
858 }
859 break;
860 case INFIX_TYPE_NAMED_REFERENCE:
861 {
862 const char * src_name = src_type->meta.named_reference.name;
863 if (src_name) {
864 size_t name_len = strlen(src_name) + 1;
865 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
866 if (!dest_name)
867 return nullptr;
868 infix_memcpy((void *)dest_name, src_name, name_len);
869 dest_type->meta.named_reference.name = dest_name;
870 }
871 break;
872 }
873 case INFIX_TYPE_REVERSE_TRAMPOLINE:
874 dest_type->meta.func_ptr_info.return_type =
875 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.func_ptr_info.return_type, memo_head);
876 if (src_type->meta.func_ptr_info.num_args > 0) {
877 size_t args_size = sizeof(infix_function_argument) * src_type->meta.func_ptr_info.num_args;
878 dest_type->meta.func_ptr_info.args =
879 infix_arena_alloc(dest_arena, args_size, _Alignof(infix_function_argument));
880 if (dest_type->meta.func_ptr_info.args == nullptr)
881 return nullptr;
882 for (size_t i = 0; i < src_type->meta.func_ptr_info.num_args; ++i) {
883 dest_type->meta.func_ptr_info.args[i] = src_type->meta.func_ptr_info.args[i];
884 dest_type->meta.func_ptr_info.args[i].type = _copy_type_graph_to_arena_recursive(
885 dest_arena, src_type->meta.func_ptr_info.args[i].type, memo_head);
886 const char * src_name = src_type->meta.func_ptr_info.args[i].name;
887 if (src_name) {
888 size_t name_len = strlen(src_name) + 1;
889 char * dest_name = infix_arena_alloc(dest_arena, name_len, 1);
890 if (!dest_name)
891 return nullptr;
892 infix_memcpy((void *)dest_name, src_name, name_len);
893 dest_type->meta.func_ptr_info.args[i].name = dest_name;
894 }
895 }
896 }
897 break;
898 case INFIX_TYPE_ENUM:
899 dest_type->meta.enum_info.underlying_type =
900 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.enum_info.underlying_type, memo_head);
901 break;
902 case INFIX_TYPE_COMPLEX:
903 dest_type->meta.complex_info.base_type =
904 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.complex_info.base_type, memo_head);
905 break;
906 case INFIX_TYPE_VECTOR:
907 dest_type->meta.vector_info.element_type =
908 _copy_type_graph_to_arena_recursive(dest_arena, src_type->meta.vector_info.element_type, memo_head);
909 break;
910 default:
911 // Other types like primitives have no child pointers to copy.
912 break;
913 }
914 return dest_type;
915}

932typedef struct estimate_visited_node_t {
933 const infix_type * type;
934 struct estimate_visited_node_t * next;
935} estimate_visited_node_t;
948static size_t _estimate_graph_size_recursive(infix_arena_t * temp_arena,
949 const infix_type * type,
950 estimate_visited_node_t ** visited_head) {
951 if (!type || !type->is_arena_allocated)
952 return 0;
953 // Cycle detection: if we've seen this node, it's already accounted for.
954 for (estimate_visited_node_t * v = *visited_head; v != NULL; v = v->next)
955 if (v->type == type)
956 return 0;
957 // Add this node to the visited list before recursing.
958 estimate_visited_node_t * visited_node =
959 infix_arena_alloc(temp_arena, sizeof(estimate_visited_node_t), _Alignof(estimate_visited_node_t));
960 if (!visited_node) {
961 // On allocation failure, we can't proceed with estimation. Return a large
962 // number to ensure the caller allocates a fallback-sized arena.
963 return 65536;
964 }
965 visited_node->type = type;
966 visited_node->next = *visited_head;
967 *visited_head = visited_node;
968 // The size includes the type object itself, a memoization node, and the name string if it exists.
969 size_t total_size = sizeof(infix_type) + sizeof(memo_node_t);
970 if (type->name)
971 total_size += strlen(type->name) + 1;
972 switch (type->category) {
973 case INFIX_TYPE_POINTER:
974 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.pointer_info.pointee_type, visited_head);
975 break;
976 case INFIX_TYPE_ARRAY:
977 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.array_info.element_type, visited_head);
978 break;
979 case INFIX_TYPE_STRUCT:
980 case INFIX_TYPE_UNION:
981 if (type->meta.aggregate_info.num_members > 0) {
982 total_size += sizeof(infix_struct_member) * type->meta.aggregate_info.num_members;
983 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
984 const infix_struct_member * member = &type->meta.aggregate_info.members[i];
985 if (member->name)
986 total_size += strlen(member->name) + 1;
987 total_size += _estimate_graph_size_recursive(temp_arena, member->type, visited_head);
988 }
989 }
990 break;
991 case INFIX_TYPE_NAMED_REFERENCE:
992 if (type->meta.named_reference.name)
993 total_size += strlen(type->meta.named_reference.name) + 1;
994 break;
995 case INFIX_TYPE_REVERSE_TRAMPOLINE:
996 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.func_ptr_info.return_type, visited_head);
997 if (type->meta.func_ptr_info.num_args > 0) {
998 total_size += sizeof(infix_function_argument) * type->meta.func_ptr_info.num_args;
999 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i) {
1000 const infix_function_argument * arg = &type->meta.func_ptr_info.args[i];
1001 if (arg->name)
1002 total_size += strlen(arg->name) + 1;
1003 total_size += _estimate_graph_size_recursive(temp_arena, arg->type, visited_head);
1004 }
1005 }
1006 break;
1007 case INFIX_TYPE_ENUM:
1008 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.enum_info.underlying_type, visited_head);
1009 break;
1010 case INFIX_TYPE_COMPLEX:
1011 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.complex_info.base_type, visited_head);
1012 break;
1013 case INFIX_TYPE_VECTOR:
1014 total_size += _estimate_graph_size_recursive(temp_arena, type->meta.vector_info.element_type, visited_head);
1015 break;
1016 default:
1017 break;
1018 }
1019 return total_size;
