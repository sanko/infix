116c23_nodiscard void * infix_arena_alloc(infix_arena_t * arena, size_t size, size_t alignment) {
117 if (arena == nullptr || arena->error)
118 return nullptr;
119
120 // Alignment must be a power of two for the bitwise alignment trick to work.
121 if (alignment == 0 || (alignment & (alignment - 1)) != 0) {
122 arena->error = true;
123 return nullptr;
124 }
125
126 // A zero-size allocation simply returns the current aligned pointer without advancing it.
127 if (size == 0)
128 return (void *)(arena->buffer + arena->current_offset);
129
130 // Calculate the next offset that meets the alignment requirement.
131 size_t aligned_offset = _infix_align_up(arena->current_offset, alignment);
132
133 // Security: Check for integer overflow on the alignment calculation.
134 if (aligned_offset < arena->current_offset) {
135 arena->error = true;
136 return nullptr;
137 }
138
139 // Security: Check for integer overflow on the final size calculation and for capacity.
140 if (SIZE_MAX - size < aligned_offset || aligned_offset + size > arena->capacity) {
141 arena->error = true;
142 return nullptr;
143 }
144
145 void * ptr = arena->buffer + aligned_offset;
146 arena->current_offset = aligned_offset + size;
147 return ptr;
148}
