119c23_nodiscard void * infix_arena_alloc(infix_arena_t * arena, size_t size, size_t alignment) {
120 if (arena == nullptr || arena->error)
121 return nullptr;
122
123 // Alignment must be a power of two for the bitwise alignment trick to work.
124 if (alignment == 0 || (alignment & (alignment - 1)) != 0) {
125 arena->error = true;
126 // This is a programmatic error. `INFIX_CODE_UNKNOWN` is the best fit from existing codes.
127 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_UNKNOWN, 0);
128 return nullptr;
129 }
130
131 // A zero-size allocation simply returns the current aligned pointer without advancing it.
132 if (size == 0)
133 return (void *)(arena->buffer + arena->current_offset);
134
135 // Calculate the next offset that meets the alignment requirement.
136 size_t aligned_offset = _infix_align_up(arena->current_offset, alignment);
137
138 // Security: Check for integer overflow on the alignment calculation.
139 if (aligned_offset < arena->current_offset) {
140 arena->error = true;
141 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_INTEGER_OVERFLOW, 0);
142 return nullptr;
143 }
144
145 // Security: Check for integer overflow on the final size calculation and for capacity.
146 if (SIZE_MAX - size < aligned_offset || aligned_offset + size > arena->capacity) {
147 arena->error = true;
148 // This is an out-of-memory condition for this specific arena.
149 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
150 return nullptr;
151 }
152
153 void * ptr = arena->buffer + aligned_offset;
154 arena->current_offset = aligned_offset + size;
155 return ptr;
156}
