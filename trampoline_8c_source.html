110void code_buffer_init(code_buffer * buf, infix_arena_t * arena) {
111 buf->capacity = 64; // Start with a small initial capacity.
112 buf->arena = arena;
113 buf->code = infix_arena_alloc(arena, buf->capacity, 16);
114 buf->size = 0;
115 buf->error = (buf->code == nullptr);
116}

124void code_buffer_append(code_buffer * buf, const void * data, size_t len) {
125 if (buf->error)
126 return; // If already in an error state, do nothing.
127
128 if (len > SIZE_MAX - buf->size) {
129 buf->error = true;
130 return;
131 }
132
133 if (buf->size + len > buf->capacity) {
134 size_t new_capacity = buf->capacity;
135 while (new_capacity < buf->size + len) {
136 if (new_capacity > SIZE_MAX / 2) {
137 buf->error = true;
138 return;
139 }
140 new_capacity *= 2;
141 }
142
143 void * new_code = infix_arena_alloc(buf->arena, new_capacity, 16);
144 if (new_code == nullptr) {
145 buf->error = true;
146 return;
147 }
148 infix_memcpy(new_code, buf->code, buf->size);
149 buf->code = new_code;
150 buf->capacity = new_capacity;
151 }
152 infix_memcpy(buf->code + buf->size, data, len);
153 buf->size += len;
154}

171// Internal Type System Helpers
180static bool _is_type_graph_resolved(infix_type * type) {
181 if (!type)
182 return true;
183 switch (type->category) {
184 case INFIX_TYPE_NAMED_REFERENCE:
185 return false;
186 case INFIX_TYPE_POINTER:
187 return _is_type_graph_resolved(type->meta.pointer_info.pointee_type);
188 case INFIX_TYPE_ARRAY:
189 return _is_type_graph_resolved(type->meta.array_info.element_type);
190 case INFIX_TYPE_STRUCT:
191 case INFIX_TYPE_UNION:
192 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i)
193 if (!_is_type_graph_resolved(type->meta.aggregate_info.members[i].type))
194 return false;
195 return true;
196 case INFIX_TYPE_REVERSE_TRAMPOLINE:
197 if (!_is_type_graph_resolved(type->meta.func_ptr_info.return_type))
198 return false;
199 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i)
200 if (!_is_type_graph_resolved(type->meta.func_ptr_info.args[i].type))
201 return false;
202 return true;
203 default:
204 return true;
205 }
206}

247c23_nodiscard infix_status _infix_forward_create_internal(infix_forward_t ** out_trampoline,
248 infix_type * return_type,
249 infix_type ** arg_types,
250 size_t num_args,
251 size_t num_fixed_args,
252 infix_arena_t * source_arena,
253 void * target_fn) {
254 if (out_trampoline == nullptr || (arg_types == nullptr && num_args > 0))
255 return INFIX_ERROR_INVALID_ARGUMENT;
256
257 // Validate the type graphs to ensure they don't contain unresolved placeholders.
258 if (!_is_type_graph_resolved(return_type)) {
259 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
260 return INFIX_ERROR_INVALID_ARGUMENT;
261 }
262 for (size_t i = 0; i < num_args; ++i) {
263 if (arg_types[i] == nullptr || !_is_type_graph_resolved(arg_types[i]))
264 return INFIX_ERROR_INVALID_ARGUMENT;
265 }
266
267 const infix_forward_abi_spec * spec = get_current_forward_abi_spec();
268 if (spec == nullptr)
269 return INFIX_ERROR_UNSUPPORTED_ABI;
270
271 infix_status status = INFIX_SUCCESS;
272 infix_call_frame_layout * layout = nullptr;
273 infix_forward_t * handle = nullptr;
274 // Use a temporary arena for layout calculations and code generation.
275 infix_arena_t * temp_arena = infix_arena_create(65536);
276 if (!temp_arena)
277 return INFIX_ERROR_ALLOCATION_FAILED;
278
279 code_buffer buf;
280 code_buffer_init(&buf, temp_arena);
281
282 // JIT Compilation Pipeline
283 status = spec->prepare_forward_call_frame(
284 temp_arena, &layout, return_type, arg_types, num_args, num_fixed_args, target_fn);
285 if (status != INFIX_SUCCESS)
286 goto cleanup;
287 status = spec->generate_forward_prologue(&buf, layout);
288 if (status != INFIX_SUCCESS)
289 goto cleanup;
290 status = spec->generate_forward_argument_moves(&buf, layout, arg_types, num_args, num_fixed_args);
291 if (status != INFIX_SUCCESS)
292 goto cleanup;
293
294 status = spec->generate_forward_call_instruction(&buf, layout);
295 if (status != INFIX_SUCCESS)
296 goto cleanup;
297
298 status = spec->generate_forward_epilogue(&buf, layout, return_type);
299 if (status != INFIX_SUCCESS)
300 goto cleanup;
301
302 if (buf.error || temp_arena->error) {
303 status = INFIX_ERROR_ALLOCATION_FAILED;
304 goto cleanup;
305 }
306
307 // Final Trampoline Handle Assembly
308 handle = infix_calloc(1, sizeof(infix_forward_t));
309 if (handle == nullptr) {
310 status = INFIX_ERROR_ALLOCATION_FAILED;
311 goto cleanup;
312 }
313
314 // "Just-Right" Arena Sizing Optimization
315 size_t required_size = source_arena ? source_arena->current_offset : 8192; // Fallback size
316 handle->arena = infix_arena_create(required_size + INFIX_TRAMPOLINE_HEADROOM);
317 if (handle->arena == nullptr) {
318 status = INFIX_ERROR_ALLOCATION_FAILED;
319 goto cleanup;
320 }
321
322 // Deep copy all type info into the handle's own arena to make it self-contained.
323 handle->return_type = _copy_type_graph_to_arena(handle->arena, return_type);
324 handle->arg_types = infix_arena_alloc(handle->arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *));
325 if ((handle->return_type == nullptr && return_type != nullptr) || (num_args > 0 && handle->arg_types == nullptr)) {
326 status = INFIX_ERROR_ALLOCATION_FAILED;
327 goto cleanup;
328 }
329 for (size_t i = 0; i < num_args; ++i) {
330 handle->arg_types[i] = _copy_type_graph_to_arena(handle->arena, arg_types[i]);
331 if (handle->arg_types[i] == nullptr && arg_types[i] != nullptr) {
332 status = INFIX_ERROR_ALLOCATION_FAILED;
333 goto cleanup;
334 }
335 }
336 handle->num_args = num_args;
337 handle->num_fixed_args = num_fixed_args;
338 handle->target_fn = target_fn;
339
340 // Allocate executable memory and copy the generated code into it.
341 handle->exec = infix_executable_alloc(buf.size);
342 if (handle->exec.rw_ptr == nullptr) {
343 status = INFIX_ERROR_ALLOCATION_FAILED;
344 goto cleanup;
345 }
346 infix_memcpy(handle->exec.rw_ptr, buf.code, buf.size);
347
348 if (!infix_executable_make_executable(handle->exec)) {
349 status = INFIX_ERROR_PROTECTION_FAILED;
350 goto cleanup;
351 }
352
353 infix_dump_hex(handle->exec.rx_ptr, handle->exec.size, "Forward Trampoline Machine Code");
354 *out_trampoline = handle;
355
356cleanup:
357 if (status != INFIX_SUCCESS && handle != nullptr)
358 infix_forward_destroy(handle);
359 infix_arena_destroy(temp_arena);
360 return status;
361}

424/*
425 * Implementation for infix_reverse_create_manual.
426 * This is a large, complex function that orchestrates the entire process of
427 * creating a reverse trampoline (callback).
428 */
429c23_nodiscard infix_status infix_reverse_create_manual(infix_reverse_t ** out_context,
430 infix_type * return_type,
431 infix_type ** arg_types,
432 size_t num_args,
433 size_t num_fixed_args,
434 void * user_callback_fn,
435 void * user_data) {
436 if (out_context == nullptr || num_fixed_args > num_args)
437 return INFIX_ERROR_INVALID_ARGUMENT;
438
439 // Validate the user-provided type graphs.
440 if (!_is_type_graph_resolved(return_type))
441 return INFIX_ERROR_INVALID_ARGUMENT;
442 if (arg_types == nullptr && num_args > 0)
443 return INFIX_ERROR_INVALID_ARGUMENT;
444 for (size_t i = 0; i < num_args; ++i) {
445 if (arg_types[i] == nullptr || !_is_type_graph_resolved(arg_types[i]))
446 return INFIX_ERROR_INVALID_ARGUMENT;
447 }
448
449 const infix_reverse_abi_spec * spec = get_current_reverse_abi_spec();
450 if (spec == nullptr)
451 return INFIX_ERROR_UNSUPPORTED_ABI;
452
453 infix_status status = INFIX_SUCCESS;
454 infix_reverse_call_frame_layout * layout = nullptr;
455 infix_reverse_t * context = nullptr;
456 infix_arena_t * temp_arena = nullptr; // For layout calculations and code generation.
457 infix_protected_t prot = {.rw_ptr = nullptr, .size = 0};
458 code_buffer buf;
459
460 temp_arena = infix_arena_create(65536);
461 if (!temp_arena)
462 return INFIX_ERROR_ALLOCATION_FAILED;
463
464 code_buffer_init(&buf, temp_arena);
465
466 // Allocate page-aligned memory for the context struct itself, so it can be hardened.
467 size_t page_size = get_page_size();
468 size_t context_alloc_size = (sizeof(infix_reverse_t) + page_size - 1) & ~(page_size - 1);
469 prot = infix_protected_alloc(context_alloc_size);
470 if (prot.rw_ptr == nullptr) {
471 status = INFIX_ERROR_ALLOCATION_FAILED;
472 goto cleanup;
473 }
474 context = (infix_reverse_t *)prot.rw_ptr;
475 infix_memset(context, 0, context_alloc_size);
476
477 // Create the persistent arena that will be owned by the context.
478 context->arena = infix_arena_create(8192);
479 if (context->arena == nullptr) {
480 status = INFIX_ERROR_ALLOCATION_FAILED;
481 goto cleanup;
482 }
483
484 // Initialize the context fields.
485 context->protected_ctx = prot;
486 context->num_args = num_args;
487 context->num_fixed_args = num_fixed_args;
488 context->is_variadic = (num_fixed_args < num_args);
489 context->user_callback_fn = user_callback_fn;
490 context->user_data = user_data;
491 context->internal_dispatcher = infix_internal_dispatch_callback_fn_impl;
492
493 // Deep copy all type information into the context's own arena.
494 context->return_type = _copy_type_graph_to_arena(context->arena, return_type);
495 context->arg_types = infix_arena_alloc(context->arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *));
496 if ((context->return_type == nullptr && return_type != nullptr) ||
497 (num_args > 0 && context->arg_types == nullptr)) {
498 status = INFIX_ERROR_ALLOCATION_FAILED;
499 goto cleanup;
500 }
501 for (size_t i = 0; i < num_args; ++i) {
502 context->arg_types[i] = _copy_type_graph_to_arena(context->arena, arg_types[i]);
503 if (context->arg_types[i] == nullptr && arg_types[i] != nullptr) {
504 status = INFIX_ERROR_ALLOCATION_FAILED;
505 goto cleanup;
506 }
507 }
508
509 // The user's handler expects `context*` as its first argument. We must create a new
510 // argument type list that reflects this for the cached forward trampoline.
511 infix_type ** callback_arg_types =
512 infix_arena_alloc(temp_arena, (1 + num_args) * sizeof(infix_type *), _Alignof(infix_type *));
513 if (callback_arg_types == nullptr) {
514 status = INFIX_ERROR_ALLOCATION_FAILED;
515 goto cleanup;
516 }
517 callback_arg_types[0] = infix_type_create_pointer(); // The context pointer
518 if (context->num_args > 0)
519 infix_memcpy(&callback_arg_types[1], context->arg_types, context->num_args * sizeof(infix_type *));
520
521 // Generate the cached forward trampoline with the augmented argument list.
522 status = infix_forward_create_manual(&context->cached_forward_trampoline,
523 context->return_type,
524 callback_arg_types,
525 context->num_args + 1,
526 context->num_fixed_args + 1,
527 user_callback_fn);
528 if (status != INFIX_SUCCESS)
529 goto cleanup;
530
531 // JIT Compilation Pipeline for Reverse Trampoline
532 status = spec->prepare_reverse_call_frame(temp_arena, &layout, context);
533 if (status != INFIX_SUCCESS)
534 goto cleanup;
535 status = spec->generate_reverse_prologue(&buf, layout);
536 if (status != INFIX_SUCCESS)
537 goto cleanup;
538 status = spec->generate_reverse_argument_marshalling(&buf, layout, context);
539 if (status != INFIX_SUCCESS)
540 goto cleanup;
541 status = spec->generate_reverse_dispatcher_call(&buf, layout, context);
542 if (status != INFIX_SUCCESS)
543 goto cleanup;
544 status = spec->generate_reverse_epilogue(&buf, layout, context);
545 if (status != INFIX_SUCCESS)
546 goto cleanup;
547
548 if (buf.error || temp_arena->error) {
549 status = INFIX_ERROR_ALLOCATION_FAILED;
550 goto cleanup;
551 }
552
553 // Allocate executable memory and finalize the JIT code.
554 context->exec = infix_executable_alloc(buf.size);
555 if (context->exec.rw_ptr == nullptr) {
556 status = INFIX_ERROR_ALLOCATION_FAILED;
557 goto cleanup;
558 }
559 infix_memcpy(context->exec.rw_ptr, buf.code, buf.size);
560
561 if (!infix_executable_make_executable(context->exec)) {
562 status = INFIX_ERROR_PROTECTION_FAILED;
563 goto cleanup;
564 }
565
566 // Harden the context struct itself to be read-only as a security measure.
567 if (!infix_protected_make_readonly(context->protected_ctx)) {
568 status = INFIX_ERROR_PROTECTION_FAILED;
569 goto cleanup;
570 }
571
572 infix_dump_hex(context->exec.rx_ptr, buf.size, "Reverse Trampoline Machine Code");
573 *out_context = context;
574
575cleanup:
576 if (status != INFIX_SUCCESS && context != nullptr)
577 infix_reverse_destroy(context);
578 infix_arena_destroy(temp_arena);
579 return status;
580}
