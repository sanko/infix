126void code_buffer_init(code_buffer * buf, infix_arena_t * arena) {
127 buf->capacity = 64; // Start with a small initial capacity.
128 buf->arena = arena;
129 buf->code = infix_arena_alloc(arena, buf->capacity, 16);
130 buf->size = 0;
131 buf->error = (buf->code == nullptr);
132}

146void code_buffer_append(code_buffer * buf, const void * data, size_t len) {
147 if (buf->error)
148 return;
149
150 if (len > SIZE_MAX - buf->size) { // Overflow check
151 buf->error = true;
152 return;
153 }
154
155 if (buf->size + len > buf->capacity) {
156 size_t new_capacity = buf->capacity;
157 while (new_capacity < buf->size + len) {
158 if (new_capacity > SIZE_MAX / 2) { // Overflow check
159 buf->error = true;
160 return;
161 }
162 new_capacity *= 2;
163 }
164
165 void * new_code = infix_arena_alloc(buf->arena, new_capacity, 16);
166 if (new_code == nullptr) {
167 buf->error = true;
168 return;
169 }
170 infix_memcpy(new_code, buf->code, buf->size);
171 buf->code = new_code;
172 buf->capacity = new_capacity;
173 }
174 infix_memcpy(buf->code + buf->size, data, len);
175 buf->size += len;
176}

190typedef struct visited_node_t {
191 const infix_type * type;
192 struct visited_node_t * next;
193} visited_node_t;
194
207static bool _is_type_graph_resolved_recursive(const infix_type * type, visited_node_t * visited_head) {
208 if (!type)
209 return true;
210
211 // Cycle detection: if we've seen this node before, we can assume it's resolved
212 // for the purpose of this check, as we'll validate it on the first visit.
213 for (visited_node_t * v = visited_head; v != NULL; v = v->next)
214 if (v->type == type)
215 return true;
216
217 visited_node_t current_visited_node = {.type = type, .next = visited_head};
218
219 switch (type->category) {
220 case INFIX_TYPE_NAMED_REFERENCE:
221 return false; // Base case: an unresolved reference.
222 case INFIX_TYPE_POINTER:
223 return _is_type_graph_resolved_recursive(type->meta.pointer_info.pointee_type, &current_visited_node);
224 case INFIX_TYPE_ARRAY:
225 return _is_type_graph_resolved_recursive(type->meta.array_info.element_type, &current_visited_node);
226 case INFIX_TYPE_STRUCT:
227 case INFIX_TYPE_UNION:
228 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i)
229 if (!_is_type_graph_resolved_recursive(type->meta.aggregate_info.members[i].type, &current_visited_node))
230 return false;
231 return true;
232 case INFIX_TYPE_REVERSE_TRAMPOLINE:
233 if (!_is_type_graph_resolved_recursive(type->meta.func_ptr_info.return_type, &current_visited_node))
234 return false;
235 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i)
236 if (!_is_type_graph_resolved_recursive(type->meta.func_ptr_info.args[i].type, &current_visited_node))
237 return false;
238 return true;
239 default:
240 return true; // Primitives, void, etc., are always resolved.
241 }
242}

248static bool _is_type_graph_resolved(const infix_type * type) { return _is_type_graph_resolved_recursive(type, NULL); }

264static size_t _estimate_metadata_size(infix_arena_t * temp_arena,
265 infix_type * return_type,
266 infix_type ** arg_types,
267 size_t num_args) {
268 size_t total_size = 0;
269 total_size += _infix_estimate_graph_size(temp_arena, return_type);
270
271 if (arg_types != nullptr) {
272 // Add space for the arg_types pointer array itself.
273 total_size += sizeof(infix_type *) * num_args;
274 for (size_t i = 0; i < num_args; ++i)
275 total_size += _infix_estimate_graph_size(temp_arena, arg_types[i]);
276 }
277 return total_size;
278}

316c23_nodiscard infix_status _infix_forward_create_internal(infix_forward_t ** out_trampoline,
317 infix_type * return_type,
318 infix_type ** arg_types,
319 size_t num_args,
320 size_t num_fixed_args,
321 void * target_fn) {
322 if (out_trampoline == nullptr || return_type == nullptr || (arg_types == nullptr && num_args > 0)) {
323 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_UNKNOWN, 0);
324 return INFIX_ERROR_INVALID_ARGUMENT;
325 }
326
327 // Pre-flight check: ensure all types are resolved before passing to ABI layer.
328 if (!_is_type_graph_resolved(return_type)) {
329 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
330 return INFIX_ERROR_INVALID_ARGUMENT;
331 }
332 for (size_t i = 0; i < num_args; ++i) {
333 if (arg_types[i] == nullptr || !_is_type_graph_resolved(arg_types[i])) {
334 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
335 return INFIX_ERROR_INVALID_ARGUMENT;
336 }
337 }
338
339 const infix_forward_abi_spec * spec = get_current_forward_abi_spec();
340 if (spec == nullptr) {
341 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNSUPPORTED_ABI, 0);
342 return INFIX_ERROR_UNSUPPORTED_ABI;
343 }
344
345 infix_status status = INFIX_SUCCESS;
346 infix_call_frame_layout * layout = nullptr;
347 infix_forward_t * handle = nullptr;
348
349 // Use a temporary arena for all intermediate allocations during code generation.
350 infix_arena_t * temp_arena = infix_arena_create(65536);
351 if (!temp_arena) {
352 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
353 return INFIX_ERROR_ALLOCATION_FAILED;
354 }
355
356 code_buffer buf;
357 code_buffer_init(&buf, temp_arena);
358
359 // JIT Compilation Pipeline
360 // 1. Prepare: Classify arguments and create the layout blueprint.
361 status = spec->prepare_forward_call_frame(
362 temp_arena, &layout, return_type, arg_types, num_args, num_fixed_args, target_fn);
363 if (status != INFIX_SUCCESS)
364 goto cleanup;
365
366 // 2. Generate: Emit machine code based on the layout.
367 status = spec->generate_forward_prologue(&buf, layout);
368 if (status != INFIX_SUCCESS)
369 goto cleanup;
370
371 status = spec->generate_forward_argument_moves(&buf, layout, arg_types, num_args, num_fixed_args);
372 if (status != INFIX_SUCCESS)
373 goto cleanup;
374
375 status = spec->generate_forward_call_instruction(&buf, layout);
376 if (status != INFIX_SUCCESS)
377 goto cleanup;
378
379 status = spec->generate_forward_epilogue(&buf, layout, return_type);
380 if (status != INFIX_SUCCESS)
381 goto cleanup;
382
383 if (buf.error || temp_arena->error) {
384 status = INFIX_ERROR_ALLOCATION_FAILED;
385 goto cleanup;
386 }
387
388 // Finalize Handle
389 handle = infix_calloc(1, sizeof(infix_forward_t));
390 if (handle == nullptr) {
391 status = INFIX_ERROR_ALLOCATION_FAILED;
392 goto cleanup;
393 }
394
395 // "Estimate" stage: Calculate the exact size needed for the handle's private arena.
396 size_t required_metadata_size = _estimate_metadata_size(temp_arena, return_type, arg_types, num_args);
397
398 // Create the handle's private arena with the calculated size plus some headroom for safety.
399 handle->arena = infix_arena_create(required_metadata_size + INFIX_TRAMPOLINE_HEADROOM);
400
401
402 if (handle->arena == nullptr) {
403 status = INFIX_ERROR_ALLOCATION_FAILED;
404 goto cleanup;
405 }
406
407 // "Copy" stage: Deep copy all type info into the handle's private arena.
408 handle->return_type = _copy_type_graph_to_arena(handle->arena, return_type);
409 if (num_args > 0) {
410 handle->arg_types = infix_arena_alloc(handle->arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *));
411 if (handle->arg_types == nullptr) {
412 status = INFIX_ERROR_ALLOCATION_FAILED;
413 goto cleanup;
414 }
415 for (size_t i = 0; i < num_args; ++i) {
416 handle->arg_types[i] = _copy_type_graph_to_arena(handle->arena, arg_types[i]);
417 // Check for allocation failure during copy
418 if (arg_types[i] != nullptr && handle->arg_types[i] == nullptr && !handle->arena->error) {
419 status = INFIX_ERROR_ALLOCATION_FAILED;
420 goto cleanup;
421 }
422 }
423 }
424
425 handle->num_args = num_args;
426 handle->num_fixed_args = num_fixed_args;
427 handle->target_fn = target_fn;
428
429 // Allocate and finalize executable memory.
430 handle->exec = infix_executable_alloc(buf.size);
431 if (handle->exec.rw_ptr == nullptr) {
432 status = INFIX_ERROR_ALLOCATION_FAILED;
433 goto cleanup;
434 }
435 infix_memcpy(handle->exec.rw_ptr, buf.code, buf.size);
436
437 if (!infix_executable_make_executable(handle->exec)) {
438 status = INFIX_ERROR_PROTECTION_FAILED;
439 goto cleanup;
440 }
441
442 infix_dump_hex(handle->exec.rx_ptr, handle->exec.size, "Forward Trampoline Machine Code");
443 *out_trampoline = handle;
444
445cleanup:
446 // If any step failed, ensure the partially created handle is fully destroyed.
447 if (status != INFIX_SUCCESS && handle != nullptr)
448 infix_forward_destroy(handle);
449 // The temporary arena is always destroyed.
450 infix_arena_destroy(temp_arena);
451 return status;
452}

568static infix_status _infix_reverse_create_internal(infix_reverse_t ** out_context,
569 infix_type * return_type,
570 infix_type ** arg_types,
571 size_t num_args,
572 size_t num_fixed_args,
573 void * user_callback_fn,
574 void * user_data,
575 bool is_callback) {
576 if (out_context == nullptr || return_type == nullptr || num_fixed_args > num_args) {
577 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_UNKNOWN, 0);
578 return INFIX_ERROR_INVALID_ARGUMENT;
579 }
580
581 // Pre-flight check: ensure all types are fully resolved.
582 if (!_is_type_graph_resolved(return_type)) {
583 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
584 return INFIX_ERROR_INVALID_ARGUMENT;
585 }
586 if (arg_types == nullptr && num_args > 0) {
587 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
588 return INFIX_ERROR_INVALID_ARGUMENT;
589 }
590 for (size_t i = 0; i < num_args; ++i) {
591 if (arg_types[i] == nullptr || !_is_type_graph_resolved(arg_types[i])) {
592 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
593 return INFIX_ERROR_INVALID_ARGUMENT;
594 }
595 }
596
597 const infix_reverse_abi_spec * spec = get_current_reverse_abi_spec();
598 if (spec == nullptr) {
599 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNSUPPORTED_ABI, 0);
600 return INFIX_ERROR_UNSUPPORTED_ABI;
601 }
602
603 infix_status status = INFIX_SUCCESS;
604 infix_reverse_call_frame_layout * layout = nullptr;
605 infix_reverse_t * context = nullptr;
606 infix_arena_t * temp_arena = nullptr;
607 infix_protected_t prot = {.rw_ptr = nullptr, .size = 0};
608 code_buffer buf;
609
610 temp_arena = infix_arena_create(65536);
611 if (!temp_arena) {
612 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
613 return INFIX_ERROR_ALLOCATION_FAILED;
614 }
615
616 code_buffer_init(&buf, temp_arena);
617
618 // Security Hardening: Allocate the context struct itself in special, page-aligned
619 // memory that can be made read-only after initialization.
620 size_t page_size = get_page_size();
621 size_t context_alloc_size = (sizeof(infix_reverse_t) + page_size - 1) & ~(page_size - 1);
622 prot = infix_protected_alloc(context_alloc_size);
623 if (prot.rw_ptr == nullptr) {
624 status = INFIX_ERROR_ALLOCATION_FAILED;
625 goto cleanup;
626 }
627 context = (infix_reverse_t *)prot.rw_ptr;
628 infix_memset(context, 0, context_alloc_size);
629
630 // "Estimate" stage: Calculate the exact size needed for the context's private arena.
631 size_t required_metadata_size = _estimate_metadata_size(temp_arena, return_type, arg_types, num_args);
632
633 // Create the context's private arena with the calculated size plus some headroom for safety.
634 context->arena = infix_arena_create(required_metadata_size + INFIX_TRAMPOLINE_HEADROOM);
635
636 if (context->arena == nullptr) {
637 status = INFIX_ERROR_ALLOCATION_FAILED;
638 goto cleanup;
639 }
640
641 // Populate the context fields.
642 context->protected_ctx = prot;
643 context->num_args = num_args;
644 context->num_fixed_args = num_fixed_args;
645 context->is_variadic = (num_fixed_args < num_args);
646 context->user_callback_fn = user_callback_fn;
647 context->user_data = user_data;
648 context->internal_dispatcher = infix_internal_dispatch_callback_fn_impl;
649 context->cached_forward_trampoline = nullptr;
650
651 // "Copy" stage: deep copy all types into the context's private arena.
652 context->return_type = _copy_type_graph_to_arena(context->arena, return_type);
653 if (num_args > 0) {
654 context->arg_types = infix_arena_alloc(context->arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *));
655 if (context->arg_types == nullptr) {
656 status = INFIX_ERROR_ALLOCATION_FAILED;
657 goto cleanup;
658 }
659 for (size_t i = 0; i < num_args; ++i) {
660 context->arg_types[i] = _copy_type_graph_to_arena(context->arena, arg_types[i]);
661 if (arg_types[i] != nullptr && context->arg_types[i] == nullptr) {
662 status = INFIX_ERROR_ALLOCATION_FAILED;
663 goto cleanup;
664 }
665 }
666 }
667
668 // Special step for type-safe callbacks: generate and cache a forward trampoline
669 // that will be used to call the user's type-safe C handler.
670 if (is_callback) {
671 status = infix_forward_create_manual(&context->cached_forward_trampoline,
672 context->return_type,
673 context->arg_types,
674 context->num_args,
675 context->num_fixed_args,
676 user_callback_fn);
677 if (status != INFIX_SUCCESS)
678 goto cleanup;
679 }
680
681 // JIT Compilation Pipeline for Reverse Stub
682 status = spec->prepare_reverse_call_frame(temp_arena, &layout, context);
683 if (status != INFIX_SUCCESS)
684 goto cleanup;
685 status = spec->generate_reverse_prologue(&buf, layout);
686 if (status != INFIX_SUCCESS)
687 goto cleanup;
688 status = spec->generate_reverse_argument_marshalling(&buf, layout, context);
689 if (status != INFIX_SUCCESS)
690 goto cleanup;
691 status = spec->generate_reverse_dispatcher_call(&buf, layout, context);
692 if (status != INFIX_SUCCESS)
693 goto cleanup;
694 status = spec->generate_reverse_epilogue(&buf, layout, context);
695 if (status != INFIX_SUCCESS)
696 goto cleanup;
697 // End of Pipeline
698
699 if (buf.error || temp_arena->error) {
700 status = INFIX_ERROR_ALLOCATION_FAILED;
701 goto cleanup;
702 }
703
704 context->exec = infix_executable_alloc(buf.size);
705 if (context->exec.rw_ptr == nullptr) {
706 status = INFIX_ERROR_ALLOCATION_FAILED;
707 goto cleanup;
708 }
709 infix_memcpy(context->exec.rw_ptr, buf.code, buf.size);
710
711 if (!infix_executable_make_executable(context->exec)) {
712 status = INFIX_ERROR_PROTECTION_FAILED;
713 goto cleanup;
714 }
715
716 // Security Hardening: Make the context memory read-only to prevent runtime corruption.
717 if (!infix_protected_make_readonly(context->protected_ctx)) {
718 status = INFIX_ERROR_PROTECTION_FAILED;
719 goto cleanup;
720 }
721
722 infix_dump_hex(context->exec.rx_ptr, buf.size, "Reverse Trampoline Machine Code");
723 *out_context = context;
724
725cleanup:
726 if (status != INFIX_SUCCESS) {
727 // If allocation of the context itself failed, prot.rw_ptr will be null.
728 if (prot.rw_ptr != nullptr)
729 infix_reverse_destroy(context);
730 }
731 infix_arena_destroy(temp_arena);
732 return status;
733}

824c23_nodiscard infix_status infix_forward_create(infix_forward_t ** out_trampoline,
825 const char * signature,
826 void * target_function,
827 infix_registry_t * registry) {
828 _infix_clear_error();
829 if (!signature) {
830 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_UNKNOWN, 0);
831 return INFIX_ERROR_INVALID_ARGUMENT;
832 }
833
834 infix_arena_t * arena = nullptr;
835 infix_type * ret_type = nullptr;
836 infix_function_argument * args = nullptr;
837 size_t num_args = 0, num_fixed = 0;
838 infix_type ** arg_types = nullptr;
839 infix_status status;
840 if (signature[0] == '@') {
841 if (registry == nullptr) {
842 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_UNKNOWN, 0); // Using @Name requires a registry
843 return INFIX_ERROR_INVALID_ARGUMENT;
844 }
845
846 const infix_type * func_type = infix_registry_lookup_type(registry, &signature[1]);
847 if (func_type == NULL) {
848 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
849 return INFIX_ERROR_INVALID_ARGUMENT;
850 }
851
852 if (func_type->category != INFIX_TYPE_REVERSE_TRAMPOLINE) {
853 // The user provided a name for a non-function type (e.g., "@Point")
854 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, 0);
855 return INFIX_ERROR_INVALID_ARGUMENT;
856 }
857
858 // We have a valid function type from the registry. Now, unpack its components.
859 ret_type = func_type->meta.func_ptr_info.return_type;
860 num_args = func_type->meta.func_ptr_info.num_args;
861 num_fixed = func_type->meta.func_ptr_info.num_fixed_args;
862 args = func_type->meta.func_ptr_info.args;
863
864 // The Manual API needs a temporary arena to hold the arg_types array.
865 infix_arena_t * temp_arena = infix_arena_create(sizeof(infix_type *) * num_args + 128);
866 if (!temp_arena) {
867 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
868 return INFIX_ERROR_ALLOCATION_FAILED;
869 }
870
871 if (num_args > 0) {
872 arg_types = infix_arena_alloc(temp_arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *));
873 if (!arg_types) {
874 infix_arena_destroy(temp_arena);
875 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
876 return INFIX_ERROR_ALLOCATION_FAILED;
877 }
878 for (size_t i = 0; i < num_args; ++i)
879 arg_types[i] = args[i].type;
880 }
881 }
882 else {
883 // This is a high-level wrapper. It uses the parser to build the type info first.
884 status = infix_signature_parse(signature, &arena, &ret_type, &args, &num_args, &num_fixed, registry);
885 if (status != INFIX_SUCCESS) {
886 infix_arena_destroy(arena);
887 return status;
888 }
889 // Extract the `infix_type*` array from the parsed `infix_function_argument` array.
890 arg_types = (num_args > 0) ? infix_arena_alloc(arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *))
891 : nullptr;
892 if (num_args > 0 && !arg_types) {
893 infix_arena_destroy(arena);
894 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
895 return INFIX_ERROR_ALLOCATION_FAILED;
896 }
897 for (size_t i = 0; i < num_args; ++i)
898 arg_types[i] = args[i].type;
899 }
900 // Call the core internal implementation with the parsed types.
901 status = _infix_forward_create_internal(out_trampoline, ret_type, arg_types, num_args, num_fixed, target_function);
902 infix_arena_destroy(arena); // Clean up the temporary arena from parsing.
903 return status;
904}
