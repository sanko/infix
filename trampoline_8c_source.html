126void code_buffer_init(code_buffer * buf, infix_arena_t * arena) {
127 buf->capacity = 64; // Start with a small initial capacity.
128 buf->arena = arena;
129 buf->code = infix_arena_alloc(arena, buf->capacity, 16);
130 buf->size = 0;
131 buf->error = (buf->code == nullptr);
132}

146void code_buffer_append(code_buffer * buf, const void * data, size_t len) {
147 if (buf->error)
148 return;
149
150 if (len > SIZE_MAX - buf->size) { // Overflow check
151 buf->error = true;
152 return;
153 }
154
155 if (buf->size + len > buf->capacity) {
156 size_t new_capacity = buf->capacity;
157 while (new_capacity < buf->size + len) {
158 if (new_capacity > SIZE_MAX / 2) { // Overflow check
159 buf->error = true;
160 return;
161 }
162 new_capacity *= 2;
163 }
164
165 void * new_code = infix_arena_alloc(buf->arena, new_capacity, 16);
166 if (new_code == nullptr) {
167 buf->error = true;
168 return;
169 }
170 infix_memcpy(new_code, buf->code, buf->size);
171 buf->code = new_code;
172 buf->capacity = new_capacity;
173 }
174 infix_memcpy(buf->code + buf->size, data, len);
175 buf->size += len;
176}

196typedef struct visited_node_t {
197 const infix_type * type;
198 struct visited_node_t * next;
199} visited_node_t;
200
213static bool _is_type_graph_resolved_recursive(const infix_type * type, visited_node_t * visited_head) {
214 if (!type)
215 return true;
216
217 // Cycle detection: if we've seen this node before, we can assume it's resolved
218 // for the purpose of this check, as we'll validate it on the first visit.
219 for (visited_node_t * v = visited_head; v != NULL; v = v->next) {
220 if (v->type == type) {
221 return true;
222 }
223 }
224
225 visited_node_t current_visited_node = {.type = type, .next = visited_head};
226
227 switch (type->category) {
228 case INFIX_TYPE_NAMED_REFERENCE:
229 return false; // Base case: an unresolved reference.
230 case INFIX_TYPE_POINTER:
231 return _is_type_graph_resolved_recursive(type->meta.pointer_info.pointee_type, &current_visited_node);
232 case INFIX_TYPE_ARRAY:
233 return _is_type_graph_resolved_recursive(type->meta.array_info.element_type, &current_visited_node);
234 case INFIX_TYPE_STRUCT:
235 case INFIX_TYPE_UNION:
236 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
237 if (!_is_type_graph_resolved_recursive(type->meta.aggregate_info.members[i].type, &current_visited_node))
238 return false;
239 }
240 return true;
241 case INFIX_TYPE_REVERSE_TRAMPOLINE:
242 if (!_is_type_graph_resolved_recursive(type->meta.func_ptr_info.return_type, &current_visited_node))
243 return false;
244 for (size_t i = 0; i < type->meta.func_ptr_info.num_args; ++i) {
245 if (!_is_type_graph_resolved_recursive(type->meta.func_ptr_info.args[i].type, &current_visited_node))
246 return false;
247 }
248 return true;
249 default:
250 return true; // Primitives, void, etc., are always resolved.
251 }
252}

258static bool _is_type_graph_resolved(const infix_type * type) {
259 return _is_type_graph_resolved_recursive(type, NULL);
260}

299c23_nodiscard infix_status _infix_forward_create_internal(infix_forward_t ** out_trampoline,
300 infix_type * return_type,
301 infix_type ** arg_types,
302 size_t num_args,
303 size_t num_fixed_args,
304 infix_arena_t * source_arena,
305 void * target_fn) {
306 if (out_trampoline == nullptr || (arg_types == nullptr && num_args > 0))
307 return INFIX_ERROR_INVALID_ARGUMENT;
308
309 // Pre-flight check: ensure all types are resolved before passing to ABI layer.
310 if (!_is_type_graph_resolved(return_type)) {
311 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
312 return INFIX_ERROR_INVALID_ARGUMENT;
313 }
314 for (size_t i = 0; i < num_args; ++i) {
315 if (arg_types[i] == nullptr || !_is_type_graph_resolved(arg_types[i])) {
316 _infix_set_error(INFIX_CATEGORY_ABI, INFIX_CODE_UNRESOLVED_NAMED_TYPE, 0);
317 return INFIX_ERROR_INVALID_ARGUMENT;
318 }
319 }
320
321 const infix_forward_abi_spec * spec = get_current_forward_abi_spec();
322 if (spec == nullptr)
323 return INFIX_ERROR_UNSUPPORTED_ABI;
324
325 infix_status status = INFIX_SUCCESS;
326 infix_call_frame_layout * layout = nullptr;
327 infix_forward_t * handle = nullptr;
328
329 // Use a temporary arena for all intermediate allocations during code generation.
330 infix_arena_t * temp_arena = infix_arena_create(65536);
331 if (!temp_arena)
332 return INFIX_ERROR_ALLOCATION_FAILED;
333
334 code_buffer buf;
335 code_buffer_init(&buf, temp_arena);
336
337 // JIT Compilation Pipeline
338 // 1. Prepare: Classify arguments and create the layout blueprint.
339 status = spec->prepare_forward_call_frame(
340 temp_arena, &layout, return_type, arg_types, num_args, num_fixed_args, target_fn);
341 if (status != INFIX_SUCCESS)
342 goto cleanup;
343
344 // 2. Generate: Emit machine code based on the layout.
345 status = spec->generate_forward_prologue(&buf, layout);
346 if (status != INFIX_SUCCESS)
347 goto cleanup;
348 status = spec->generate_forward_argument_moves(&buf, layout, arg_types, num_args, num_fixed_args);
349 if (status != INFIX_SUCCESS)
350 goto cleanup;
351 status = spec->generate_forward_call_instruction(&buf, layout);
352 if (status != INFIX_SUCCESS)
353 goto cleanup;
354 status = spec->generate_forward_epilogue(&buf, layout, return_type);
355 if (status != INFIX_SUCCESS)
356 goto cleanup;
357 // End of Pipeline
358
359 if (buf.error || temp_arena->error) {
360 status = INFIX_ERROR_ALLOCATION_FAILED;
361 goto cleanup;
362 }
363
364 // Finalize Handle
365 // Allocate the handle that will be returned to the user.
366 handle = infix_calloc(1, sizeof(infix_forward_t));
367 if (handle == nullptr) {
368 status = INFIX_ERROR_ALLOCATION_FAILED;
369 goto cleanup;
370 }
371
372 // Create a private arena for the handle to own its type metadata.
373 size_t required_size = source_arena ? source_arena->current_offset : 8192;
374 handle->arena = infix_arena_create(required_size + INFIX_TRAMPOLINE_HEADROOM);
375 if (handle->arena == nullptr) {
376 status = INFIX_ERROR_ALLOCATION_FAILED;
377 goto cleanup;
378 }
379
380 // "Copy" stage: Deep copy all type info into the handle's private arena.
381 handle->return_type = _copy_type_graph_to_arena(handle->arena, return_type);
382 handle->arg_types = infix_arena_alloc(handle->arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *));
383 if ((handle->return_type == nullptr && return_type != nullptr) || (num_args > 0 && handle->arg_types == nullptr)) {
384 status = INFIX_ERROR_ALLOCATION_FAILED;
385 goto cleanup;
386 }
387 for (size_t i = 0; i < num_args; ++i) {
388 handle->arg_types[i] = _copy_type_graph_to_arena(handle->arena, arg_types[i]);
389 if (handle->arg_types[i] == nullptr && arg_types[i] != nullptr) {
390 status = INFIX_ERROR_ALLOCATION_FAILED;
391 goto cleanup;
392 }
393 }
394 handle->num_args = num_args;
395 handle->num_fixed_args = num_fixed_args;
396 handle->target_fn = target_fn;
397
398 // Allocate and finalize executable memory.
399 handle->exec = infix_executable_alloc(buf.size);
400 if (handle->exec.rw_ptr == nullptr) {
401 status = INFIX_ERROR_ALLOCATION_FAILED;
402 goto cleanup;
403 }
404 infix_memcpy(handle->exec.rw_ptr, buf.code, buf.size);
405
406 if (!infix_executable_make_executable(handle->exec)) {
407 status = INFIX_ERROR_PROTECTION_FAILED;
408 goto cleanup;
409 }
410
411 infix_dump_hex(handle->exec.rx_ptr, handle->exec.size, "Forward Trampoline Machine Code");
412 *out_trampoline = handle;
413
414cleanup:
415 // If any step failed, ensure the partially created handle is fully destroyed.
416 if (status != INFIX_SUCCESS && handle != nullptr)
417 infix_forward_destroy(handle);
418 // The temporary arena is always destroyed.
419 infix_arena_destroy(temp_arena);
420 return status;
421}

536static infix_status _infix_reverse_create_internal(infix_reverse_t ** out_context,
537 infix_type * return_type,
538 infix_type ** arg_types,
539 size_t num_args,
540 size_t num_fixed_args,
541 void * user_callback_fn,
542 void * user_data,
543 bool is_callback) {
544 if (out_context == nullptr || num_fixed_args > num_args)
545 return INFIX_ERROR_INVALID_ARGUMENT;
546
547 // Pre-flight check: ensure all types are fully resolved.
548 if (!_is_type_graph_resolved(return_type))
549 return INFIX_ERROR_INVALID_ARGUMENT;
550 if (arg_types == nullptr && num_args > 0)
551 return INFIX_ERROR_INVALID_ARGUMENT;
552 for (size_t i = 0; i < num_args; ++i) {
553 if (arg_types[i] == nullptr || !_is_type_graph_resolved(arg_types[i]))
554 return INFIX_ERROR_INVALID_ARGUMENT;
555 }
556
557 const infix_reverse_abi_spec * spec = get_current_reverse_abi_spec();
558 if (spec == nullptr)
559 return INFIX_ERROR_UNSUPPORTED_ABI;
560
561 infix_status status = INFIX_SUCCESS;
562 infix_reverse_call_frame_layout * layout = nullptr;
563 infix_reverse_t * context = nullptr;
564 infix_arena_t * temp_arena = nullptr;
565 infix_protected_t prot = {.rw_ptr = nullptr, .size = 0};
566 code_buffer buf;
567
568 temp_arena = infix_arena_create(65536);
569 if (!temp_arena)
570 return INFIX_ERROR_ALLOCATION_FAILED;
571
572 code_buffer_init(&buf, temp_arena);
573
574 // Security Hardening: Allocate the context struct itself in special, page-aligned
575 // memory that can be made read-only after initialization.
576 size_t page_size = get_page_size();
577 size_t context_alloc_size = (sizeof(infix_reverse_t) + page_size - 1) & ~(page_size - 1);
578 prot = infix_protected_alloc(context_alloc_size);
579 if (prot.rw_ptr == nullptr) {
580 status = INFIX_ERROR_ALLOCATION_FAILED;
581 goto cleanup;
582 }
583 context = (infix_reverse_t *)prot.rw_ptr;
584 infix_memset(context, 0, context_alloc_size);
585
586 // Create the private arena for this context's type metadata.
587 context->arena = infix_arena_create(8192);
588 if (context->arena == nullptr) {
589 status = INFIX_ERROR_ALLOCATION_FAILED;
590 goto cleanup;
591 }
592
593 // Populate the context fields.
594 context->protected_ctx = prot;
595 context->num_args = num_args;
596 context->num_fixed_args = num_fixed_args;
597 context->is_variadic = (num_fixed_args < num_args);
598 context->user_callback_fn = user_callback_fn;
599 context->user_data = user_data;
600 context->internal_dispatcher = infix_internal_dispatch_callback_fn_impl;
601 context->cached_forward_trampoline = nullptr;
602
603 // "Copy" stage: deep copy all types into the context's private arena.
604 context->return_type = _copy_type_graph_to_arena(context->arena, return_type);
605 context->arg_types = infix_arena_alloc(context->arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *));
606 if ((context->return_type == nullptr && return_type != nullptr) ||
607 (num_args > 0 && context->arg_types == nullptr)) {
608 status = INFIX_ERROR_ALLOCATION_FAILED;
609 goto cleanup;
610 }
611 for (size_t i = 0; i < num_args; ++i) {
612 context->arg_types[i] = _copy_type_graph_to_arena(context->arena, arg_types[i]);
613 if (context->arg_types[i] == nullptr && arg_types[i] != nullptr) {
614 status = INFIX_ERROR_ALLOCATION_FAILED;
615 goto cleanup;
616 }
617 }
618
619 // Special step for type-safe callbacks: generate and cache a forward trampoline
620 // that will be used to call the user's type-safe C handler.
621 if (is_callback) {
622 status = infix_forward_create_manual(&context->cached_forward_trampoline,
623 context->return_type,
624 context->arg_types,
625 context->num_args,
626 context->num_fixed_args,
627 user_callback_fn);
628 if (status != INFIX_SUCCESS)
629 goto cleanup;
630 }
631
632 // JIT Compilation Pipeline for Reverse Stub
633 status = spec->prepare_reverse_call_frame(temp_arena, &layout, context);
634 if (status != INFIX_SUCCESS)
635 goto cleanup;
636 status = spec->generate_reverse_prologue(&buf, layout);
637 if (status != INFIX_SUCCESS)
638 goto cleanup;
639 status = spec->generate_reverse_argument_marshalling(&buf, layout, context);
640 if (status != INFIX_SUCCESS)
641 goto cleanup;
642 status = spec->generate_reverse_dispatcher_call(&buf, layout, context);
643 if (status != INFIX_SUCCESS)
644 goto cleanup;
645 status = spec->generate_reverse_epilogue(&buf, layout, context);
646 if (status != INFIX_SUCCESS)
647 goto cleanup;
648 // End of Pipeline
649
650 if (buf.error || temp_arena->error) {
651 status = INFIX_ERROR_ALLOCATION_FAILED;
652 goto cleanup;
653 }
654
655 context->exec = infix_executable_alloc(buf.size);
656 if (context->exec.rw_ptr == nullptr) {
657 status = INFIX_ERROR_ALLOCATION_FAILED;
658 goto cleanup;
659 }
660 infix_memcpy(context->exec.rw_ptr, buf.code, buf.size);
661
662 if (!infix_executable_make_executable(context->exec)) {
663 status = INFIX_ERROR_PROTECTION_FAILED;
664 goto cleanup;
665 }
666
667 // Security Hardening: Make the context memory read-only to prevent runtime corruption.
668 if (!infix_protected_make_readonly(context->protected_ctx)) {
669 status = INFIX_ERROR_PROTECTION_FAILED;
670 goto cleanup;
671 }
672
673 infix_dump_hex(context->exec.rx_ptr, buf.size, "Reverse Trampoline Machine Code");
674 *out_context = context;
675
676cleanup:
677 if (status != INFIX_SUCCESS && context != nullptr)
678 infix_reverse_destroy(context);
679 infix_arena_destroy(temp_arena);
680 return status;
681}

770c23_nodiscard infix_status infix_forward_create(infix_forward_t ** out_trampoline,
771 const char * signature,
772 void * target_function,
773 infix_registry_t * registry) {
774 infix_arena_t * arena = nullptr;
775 infix_type * ret_type = nullptr;
776 infix_function_argument * args = nullptr;
777 size_t num_args = 0, num_fixed = 0;
778 // This is a high-level wrapper. It uses the parser to build the type info first.
779 infix_status status = infix_signature_parse(signature, &arena, &ret_type, &args, &num_args, &num_fixed, registry);
780 if (status != INFIX_SUCCESS) {
781 infix_arena_destroy(arena);
782 return status;
783 }
784 // Extract the `infix_type*` array from the parsed `infix_function_argument` array.
785 infix_type ** arg_types =
786 (num_args > 0) ? infix_arena_alloc(arena, sizeof(infix_type *) * num_args, _Alignof(infix_type *)) : nullptr;
787 if (num_args > 0 && !arg_types) {
788 infix_arena_destroy(arena);
789 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
790 return INFIX_ERROR_ALLOCATION_FAILED;
791 }
792 for (size_t i = 0; i < num_args; ++i)
793 arg_types[i] = args[i].type;
794
795 // Call the core internal implementation with the parsed types.
796 status = _infix_forward_create_internal(
797 out_trampoline, ret_type, arg_types, num_args, num_fixed, arena, target_function);
798 infix_arena_destroy(arena); // Clean up the temporary arena from parsing.
799 return status;
800}

809 const char * signature,

813 infix_type * ret_type = nullptr;

817 if (status != INFIX_SUCCESS) {
