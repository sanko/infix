90static void skip_whitespace(parser_state * state) {
91 while (true) {
92 while (isspace((unsigned char)*state->p))
93 state->p++;
94 if (*state->p == '#') { // C-style line comments
95 while (*state->p != '\n' && *state->p != '\0')
96 state->p++;
97 }
98 else
99 break;
100 }
101}

110static bool parse_size_t(parser_state * state, size_t * out_val) {
111 const char * start = state->p;
112 char * end;
113 unsigned long long val = strtoull(start, &end, 10);
114 if (end == start) {
115 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
116 return false;
117 }
118 *out_val = (size_t)val;
119 state->p = end;
120 return true;
121}

131static const char * parse_identifier(parser_state * state) {
132 skip_whitespace(state);
133 const char * start = state->p;
134 if (!isalpha((unsigned char)*start) && *start != '_')
135 return nullptr;
136 while (isalnum((unsigned char)*state->p) || *state->p == '_' || *state->p == ':') {
137 if (*state->p == ':' && state->p[1] != ':')
138 break; // A single ':' is not part of an identifier.
139 if (*state->p == ':')
140 state->p++; // Consume first ':' of '::'
141 state->p++;
142 }
143 size_t len = state->p - start;
144 if (len == 0)
145 return nullptr;
146 char * name = infix_arena_calloc(state->arena, 1, len + 1, 1);
147 if (!name) {
148 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, (size_t)(state->p - state->start));
149 return nullptr;
150 }
151 infix_memcpy(name, start, len);
152 name[len] = '\0';
153 return name;
154}

166static bool consume_keyword(parser_state * state, const char * keyword) {
167 skip_whitespace(state);
168 size_t len = strlen(keyword);
169 if (strncmp(state->p, keyword, len) == 0) {
170 // Ensure it's not a prefix of a longer word (e.g., "int" vs "integer").
171 if (isalnum((unsigned char)state->p[len]) || state->p[len] == '_')
172 return false;
173 state->p += len;
174 skip_whitespace(state);
175 return true;
176 }
177 return false;
178}

189static const char * parse_optional_name_prefix(parser_state * state) {
190 skip_whitespace(state);
191 // Save the current position in case we need to backtrack.
192 const char * p_before = state->p;
193 const char * name = parse_identifier(state);
194 if (name) {
195 skip_whitespace(state);
196 if (*state->p == ':') { // Found "identifier:", so consume the colon and return the name.
197 state->p++;
198 return name;
199 }
200 }
201 // If it wasn't a `name:`, backtrack to the original position.
202 state->p = p_before;
203 return nullptr;
204}

220static bool is_function_signature_ahead(const parser_state * state) {
221 const char * p = state->p;
222 if (*p != '(')
223 return false;
224 p++;
225 // Find the matching ')' by tracking nesting depth.
226 int depth = 1;
227 while (*p != '\0' && depth > 0) {
228 if (*p == '(')
229 depth++;
230 else if (*p == ')')
231 depth--;
232 p++;
233 }
234 if (depth != 0)
235 return false; // Mismatched parentheses.
236 // Skip any whitespace or comments after the ')'
237 while (isspace((unsigned char)*p) || *p == '#') {
238 if (*p == '#') {
239 while (*p != '\n' && *p != '\0')
240 p++;
241 }
242 else
243 p++;
244 }
245 // Check for the '->' arrow.
246 return (p[0] == '-' && p[1] == '>');
247}

262static infix_struct_member * parse_aggregate_members(parser_state * state, char end_char, size_t * out_num_members) {
263 // Use a temporary linked list to collect members, as the count is unknown in a single pass.
264 typedef struct member_node {
265 infix_struct_member m;
266 struct member_node * next;
267 } member_node;
268 member_node *head = nullptr, *tail = nullptr;
269 size_t num_members = 0;
270
271 skip_whitespace(state);
272 if (*state->p != end_char) {
273 while (1) {
274 const char * p_before_member = state->p;
275 const char * name = parse_optional_name_prefix(state);
276 // Disallow an empty member definition like `name,` without a type.
277 if (name && (*state->p == ',' || *state->p == end_char)) {
278 state->p = p_before_member + strlen(name); // Position error at end of name
279 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
280 return nullptr;
281 }
282
283 infix_type * member_type = parse_type(state);
284 if (!member_type)
285 return nullptr;
286 // Structs and unions cannot have `void` members.
287 if (member_type->category == INFIX_TYPE_VOID) {
288 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
289 return nullptr;
290 }
291
292 member_node * node = infix_arena_calloc(state->arena, 1, sizeof(member_node), _Alignof(member_node));
293 if (!node) {
294 _infix_set_error(
295 INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, (size_t)(state->p - state->start));
296 return nullptr;
297 }
298 // The member offset is not calculated here; it will be done later
299 // by `infix_type_create_struct` or `_infix_type_recalculate_layout`.
300 node->m = infix_type_create_member(name, member_type, 0);
301 node->next = nullptr;
302 if (!head)
303 head = tail = node;
304 else {
305 tail->next = node;
306 tail = node;
307 }
308 num_members++;
309 // Check for next token: ',' or end_char
310 skip_whitespace(state);
311
312 if (*state->p == ',') {
313 state->p++; // Consume comma.
314 skip_whitespace(state);
315 // A trailing comma like `{int,}` is a syntax error.
316 if (*state->p == end_char) {
317 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
318 return nullptr;
319 }
320 }
321 else if (*state->p == end_char)
322 break;
323 else { // Unexpected token (e.g., missing comma).
324 if (*state->p == '\0')
325 return nullptr; // EOF is handled by the caller.
326 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
327 return nullptr;
328 }
329 }
330 }
331
332 *out_num_members = num_members;
333 if (num_members == 0)
334 return nullptr;
335
336 // Convert the temporary linked list to a flat array in the arena.
337 infix_struct_member * members =
338 infix_arena_calloc(state->arena, num_members, sizeof(infix_struct_member), _Alignof(infix_struct_member));
339 if (!members) {
340 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, (size_t)(state->p - state->start));
341 return nullptr;
342 }
343 member_node * current = head;
344 for (size_t i = 0; i < num_members; i++) {
345 members[i] = current->m;
346 current = current->next;
347 }
348 return members;
349}

359static infix_type * parse_aggregate(parser_state * state, char start_char, char end_char) {
360 if (state->depth >= MAX_RECURSION_DEPTH) {
361 set_parser_error(state, INFIX_CODE_RECURSION_DEPTH_EXCEEDED);
362 return nullptr;
363 }
364 state->depth++;
365
366 if (*state->p != start_char) {
367 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
368 state->depth--;
369 return nullptr;
370 }
371 state->p++;
372
373 size_t num_members = 0;
374 infix_struct_member * members = parse_aggregate_members(state, end_char, &num_members);
375 // If member parsing failed, an error is already set. Propagate the failure.
376 if (!members && infix_get_last_error().code != INFIX_CODE_SUCCESS) {
377 state->depth--;
378 return nullptr;
379 }
380
381 if (*state->p != end_char) {
382 set_parser_error(state, INFIX_CODE_UNTERMINATED_AGGREGATE);
383 state->depth--;
384 return nullptr;
385 }
386 state->p++;
387
388 infix_type * agg_type = nullptr;
389 infix_status status = (start_char == '{') ? infix_type_create_struct(state->arena, &agg_type, members, num_members)
390 : infix_type_create_union(state->arena, &agg_type, members, num_members);
391
392 if (status != INFIX_SUCCESS) {
393 _infix_set_error(INFIX_CATEGORY_GENERAL,
394 (status == INFIX_ERROR_ALLOCATION_FAILED) ? INFIX_CODE_OUT_OF_MEMORY : INFIX_CODE_UNKNOWN,
395 (size_t)(state->p - state->start));
396 state->depth--;
397 return nullptr;
398 }
399
400 state->depth--;
401 return agg_type;
402}

410static infix_type * parse_packed_struct(parser_state * state) {
411 size_t alignment = 1; // Default alignment for `!{...}` is 1.
412 if (*state->p == '!') {
413 state->p++;
414 if (isdigit((unsigned char)*state->p)) {
415 // This is the `!N:{...}` form with an explicit alignment.
416 if (!parse_size_t(state, &alignment))
417 return nullptr;
418 if (*state->p != ':') {
419 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
420 return nullptr;
421 }
422 state->p++;
423 }
424 }
425
426 skip_whitespace(state);
427 if (*state->p != '{') {
428 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
429 return nullptr;
430 }
431 state->p++;
432 size_t num_members = 0;
433 infix_struct_member * members = parse_aggregate_members(state, '}', &num_members);
434 if (!members && infix_get_last_error().code != INFIX_CODE_SUCCESS)
435 return nullptr;
436
437 if (*state->p != '}') {
438 set_parser_error(state, INFIX_CODE_UNTERMINATED_AGGREGATE);
439 return nullptr;
440 }
441 state->p++;
442
443 infix_type * packed_type = nullptr;
444 // For packed structs, the total size is simply the sum of member sizes without padding.
445 // The user of `infix_type_create_packed_struct` must provide pre-calculated offsets.
446 // Since our parser doesn't know the offsets, we pass a preliminary size. The final
447 // layout pass will fix this if needed, but for packed structs, the user's offsets
448 // are king.
449 size_t total_size = 0;
450 for (size_t i = 0; i < num_members; ++i)
451 total_size += members[i].type->size;
452
453 infix_status status =
454 infix_type_create_packed_struct(state->arena, &packed_type, total_size, alignment, members, num_members);
455
456 if (status != INFIX_SUCCESS) {
457 _infix_set_error(INFIX_CATEGORY_GENERAL,
458 (status == INFIX_ERROR_ALLOCATION_FAILED) ? INFIX_CODE_OUT_OF_MEMORY : INFIX_CODE_UNKNOWN,
459 (size_t)(state->p - state->start));
460 return nullptr;
461 }
462 return packed_type;
463}

477static infix_type * parse_primitive(parser_state * state) {
478 if (consume_keyword(state, "sint8") || consume_keyword(state, "int8"))
479 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT8);
480 if (consume_keyword(state, "uint8"))
481 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT8);
482 if (consume_keyword(state, "sint16") || consume_keyword(state, "int16"))
483 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT16);
484 if (consume_keyword(state, "uint16"))
485 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT16);
486 if (consume_keyword(state, "sint32") || consume_keyword(state, "int32"))
487 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT32);
488 if (consume_keyword(state, "uint32"))
489 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT32);
490 if (consume_keyword(state, "sint64") || consume_keyword(state, "int64"))
491 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT64);
492 if (consume_keyword(state, "uint64"))
493 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT64);
494 if (consume_keyword(state, "sint128") || consume_keyword(state, "int128"))
495 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT128);
496 if (consume_keyword(state, "uint128"))
497 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT128);
498 if (consume_keyword(state, "float32"))
499 return infix_type_create_primitive(INFIX_PRIMITIVE_FLOAT);
500 if (consume_keyword(state, "float64"))
501 return infix_type_create_primitive(INFIX_PRIMITIVE_DOUBLE);
502 if (consume_keyword(state, "bool"))
503 return infix_type_create_primitive(INFIX_PRIMITIVE_BOOL);
504 if (consume_keyword(state, "void"))
505 return infix_type_create_void();
506
507 // C-style convenience aliases
508 if (consume_keyword(state, "uchar"))
509 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT8);
510 if (consume_keyword(state, "char"))
511 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT8);
512 if (consume_keyword(state, "ushort"))
513 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT16);
514 if (consume_keyword(state, "short"))
515 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT16);
516 if (consume_keyword(state, "uint"))
517 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT32);
518 if (consume_keyword(state, "int"))
519 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT32);
520 if (consume_keyword(state, "ulonglong"))
521 return infix_type_create_primitive(INFIX_PRIMITIVE_UINT64);
522 if (consume_keyword(state, "longlong"))
523 return infix_type_create_primitive(INFIX_PRIMITIVE_SINT64);
524 // `long` is platform-dependent, so we use `sizeof` to pick the correct size.
525 if (consume_keyword(state, "ulong"))
526 return infix_type_create_primitive(sizeof(unsigned long) == 8 ? INFIX_PRIMITIVE_UINT64
527 : INFIX_PRIMITIVE_UINT32);
528 if (consume_keyword(state, "long"))
529 return infix_type_create_primitive(sizeof(long) == 8 ? INFIX_PRIMITIVE_SINT64 : INFIX_PRIMITIVE_SINT32);
530 if (consume_keyword(state, "double"))
531 return infix_type_create_primitive(INFIX_PRIMITIVE_DOUBLE);
532 if (consume_keyword(state, "float"))
533 return infix_type_create_primitive(INFIX_PRIMITIVE_FLOAT);
534 if (consume_keyword(state, "longdouble"))
535 return infix_type_create_primitive(INFIX_PRIMITIVE_LONG_DOUBLE);
536
537 return nullptr;
538}

550static infix_type * parse_type(parser_state * state) {
551 if (state->depth >= MAX_RECURSION_DEPTH) {
552 set_parser_error(state, INFIX_CODE_RECURSION_DEPTH_EXCEEDED);
553 return nullptr;
554 }
555 state->depth++;
556 skip_whitespace(state);
557 infix_type * result_type = nullptr;
558 const char * p_before_type = state->p;
559
560 if (*state->p == '@') { // Named type reference: `@MyStruct`
561 state->p++;
562 const char * name = parse_identifier(state);
563 if (!name) {
564 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
565 state->depth--;
566 return nullptr;
567 }
568 if (infix_type_create_named_reference(state->arena, &result_type, name, INFIX_AGGREGATE_STRUCT) !=
569 INFIX_SUCCESS)
570 result_type = nullptr;
571 }
572 else if (*state->p == '*') { // Pointer type: `*int`
573 state->p++;
574 skip_whitespace(state);
575 infix_type * pointee_type = parse_type(state);
576 if (!pointee_type) {
577 state->depth--;
578 return nullptr;
579 }
580 if (infix_type_create_pointer_to(state->arena, &result_type, pointee_type) != INFIX_SUCCESS)
581 result_type = nullptr;
582 }
583 else if (*state->p == '(') { // Grouped type `(type)` or function pointer `(...) -> type`
584 if (is_function_signature_ahead(state)) {
585 infix_type * ret_type = nullptr;
586 infix_function_argument * args = nullptr;
587 size_t num_args = 0, num_fixed = 0;
588 if (parse_function_signature_details(state, &ret_type, &args, &num_args, &num_fixed) != INFIX_SUCCESS) {
589 state->depth--;
590 return nullptr;
591 }
592 // Manually construct a function pointer type object.
593 // This is represented internally as a pointer-like type with extra metadata.
594 infix_type * func_type = infix_arena_calloc(state->arena, 1, sizeof(infix_type), _Alignof(infix_type));
595 if (!func_type) {
596 _infix_set_error(
597 INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, (size_t)(state->p - state->start));
598 state->depth--;
599 return nullptr;
600 }
601 *func_type = *infix_type_create_pointer(); // Function pointers have the layout of a generic pointer.
602 func_type->is_arena_allocated = true;
603 func_type->category = INFIX_TYPE_REVERSE_TRAMPOLINE; // Special category for function types.
604 func_type->meta.func_ptr_info.return_type = ret_type;
605 func_type->meta.func_ptr_info.args = args;
606 func_type->meta.func_ptr_info.num_args = num_args;
607 func_type->meta.func_ptr_info.num_fixed_args = num_fixed;
608 result_type = func_type;
609 }
610 else { // Grouped type: `(type)`
611 state->p++;
612 skip_whitespace(state);
613 result_type = parse_type(state);
614 if (!result_type) {
615 state->depth--;
616 return nullptr;
617 }
618 skip_whitespace(state);
619 if (*state->p != ')') {
620 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
621 result_type = nullptr;
622 }
623 else
624 state->p++;
625 }
626 }
627 else if (*state->p == '[') { // Array type: `[size:type]`
628 state->p++;
629 skip_whitespace(state);
630 size_t num_elements;
631 if (!parse_size_t(state, &num_elements)) {
632 state->depth--;
633 return nullptr;
634 }
635 skip_whitespace(state);
636 if (*state->p != ':') {
637 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
638 state->depth--;
639 return nullptr;
640 }
641 state->p++;
642 skip_whitespace(state);
643 infix_type * element_type = parse_type(state);
644 if (!element_type) {
645 state->depth--;
646 return nullptr;
647 }
648 if (element_type->category == INFIX_TYPE_VOID) { // An array of `void` is illegal in C.
649 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
650 state->depth--;
651 return nullptr;
652 }
653 skip_whitespace(state);
654 if (*state->p != ']') {
655 set_parser_error(state, INFIX_CODE_UNTERMINATED_AGGREGATE);
656 state->depth--;
657 return nullptr;
658 }
659 state->p++;
660 if (infix_type_create_array(state->arena, &result_type, element_type, num_elements) != INFIX_SUCCESS)
661 result_type = nullptr;
662 }
663 else if (*state->p == '!') // Packed struct
664 result_type = parse_packed_struct(state);
665 else if (*state->p == '{') // Struct
666 result_type = parse_aggregate(state, '{', '}');
667 else if (*state->p == '<') // Union
668 result_type = parse_aggregate(state, '<', '>');
669 else if (*state->p == 'e' && state->p[1] == ':') { // Enum: `e:type`
670 state->p += 2;
671 skip_whitespace(state);
672 infix_type * underlying_type = parse_type(state);
673 if (!underlying_type || underlying_type->category != INFIX_TYPE_PRIMITIVE) {
674 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
675 state->depth--;
676 return nullptr;
677 }
678 if (infix_type_create_enum(state->arena, &result_type, underlying_type) != INFIX_SUCCESS)
679 result_type = nullptr;
680 }
681 else if (*state->p == 'c' && state->p[1] == '[') { // Complex: `c[type]`
682 state->p += 2;
683 skip_whitespace(state);
684 infix_type * base_type = parse_type(state);
685 if (!base_type) {
686 state->depth--;
687 return nullptr;
688 }
689 skip_whitespace(state);
690 if (*state->p != ']') {
691 set_parser_error(state, INFIX_CODE_UNTERMINATED_AGGREGATE);
692 state->depth--;
693 return nullptr;
694 }
695 state->p++;
696 if (infix_type_create_complex(state->arena, &result_type, base_type) != INFIX_SUCCESS)
697 result_type = nullptr;
698 }
699 else if (*state->p == 'v' && state->p[1] == '[') { // Vector: `v[size:type]`
700 state->p += 2;
701 skip_whitespace(state);
702 size_t num_elements;
703 if (!parse_size_t(state, &num_elements)) {
704 state->depth--;
705 return nullptr;
706 }
707 if (*state->p != ':') {
708 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
709 state->depth--;
710 return nullptr;
711 }
712 state->p++;
713 infix_type * element_type = parse_type(state);
714 if (!element_type) {
715 state->depth--;
716 return nullptr;
717 }
718 if (*state->p != ']') {
719 set_parser_error(state, INFIX_CODE_UNTERMINATED_AGGREGATE);
720 state->depth--;
721 return nullptr;
722 }
723 state->p++;
724 if (infix_type_create_vector(state->arena, &result_type, element_type, num_elements) != INFIX_SUCCESS)
725 result_type = nullptr;
726 }
727 else { // Primitive type or error
728 result_type = parse_primitive(state);
729 if (!result_type) {
730 // If no error was set by a failed `consume_keyword`, set a generic one.
731 if (infix_get_last_error().code == INFIX_CODE_SUCCESS) {
732 state->p = p_before_type;
733 if (isalpha((unsigned char)*state->p) || *state->p == '_')
734 set_parser_error(state, INFIX_CODE_INVALID_KEYWORD);
735 else
736 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
737 }
738 }
739 }
740
741 state->depth--;
742 return result_type;
743}

758static infix_status parse_function_signature_details(parser_state * state,
759 infix_type ** out_ret_type,
760 infix_function_argument ** out_args,
761 size_t * out_num_args,
762 size_t * out_num_fixed_args) {
763 if (*state->p != '(') {
764 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
765 return INFIX_ERROR_INVALID_ARGUMENT;
766 }
767 state->p++;
768 skip_whitespace(state);
769
770 // Use a temporary linked list to collect arguments.
771 typedef struct arg_node {
772 infix_function_argument arg;
773 struct arg_node * next;
774 } arg_node;
775 arg_node *head = nullptr, *tail = nullptr;
776 size_t num_args = 0;
777
778 // Parse Fixed Arguments
779 if (*state->p != ')' && *state->p != ';') {
780 while (1) {
781 skip_whitespace(state);
782 if (*state->p == ')' || *state->p == ';')
783 break;

786 if (!arg_type)

793 node->arg.name = name;
794 node->next = nullptr;

796 head = tail = node;
797 else {
798 tail->next = node;
799 tail = node;
800 }

803 if (*state->p == ',') {

805 skip_whitespace(state);

807 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
808 return INFIX_ERROR_INVALID_ARGUMENT;
809 }
810 }
811 else if (*state->p != ')' && *state->p != ';') {
812 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
813 return INFIX_ERROR_INVALID_ARGUMENT;
814 }
815 else
816 break;
817 }
818 }
819 *out_num_fixed_args = num_args;
820
821 // Parse Variadic Arguments
822 if (*state->p == ';') {
823 state->p++;
824 if (*state->p != ')') {
825 while (1) {
826 skip_whitespace(state);
827 if (*state->p == ')')
828 break;
829 const char * name = parse_optional_name_prefix(state);
830 infix_type * arg_type = parse_type(state);
831 if (!arg_type)
832 return INFIX_ERROR_INVALID_ARGUMENT;
833

835 if (!node)
836 return INFIX_ERROR_ALLOCATION_FAILED;
837 node->arg.type = arg_type;
838 node->arg.name = name;
839 node->next = nullptr;
840 if (!head)
841 head = tail = node;
842 else {
843 tail->next = node;
844 tail = node;
845 }
846 num_args++;
847 skip_whitespace(state);
848 if (*state->p == ',') {
849 state->p++;
850 skip_whitespace(state);
851 if (*state->p == ')') { // Trailing comma error.
852 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
853 return INFIX_ERROR_INVALID_ARGUMENT;
854 }
855 }
856 else if (*state->p != ')') {
857 set_parser_error(state, INFIX_CODE_UNEXPECTED_TOKEN);
858 return INFIX_ERROR_INVALID_ARGUMENT;
859 }
860 else
861 break;
862 }
863 }
864 }
865 skip_whitespace(state);
866 if (*state->p != ')') {
867 set_parser_error(state, INFIX_CODE_UNTERMINATED_AGGREGATE);
868 return INFIX_ERROR_INVALID_ARGUMENT;
869 }
870 state->p++;
871
872 // Parse Return Type
873 skip_whitespace(state);
874 if (state->p[0] != '-' || state->p[1] != '>') {
875 set_parser_error(state, INFIX_CODE_MISSING_RETURN_TYPE);
876 return INFIX_ERROR_INVALID_ARGUMENT;
877 }
878 state->p += 2;
879 *out_ret_type = parse_type(state);
880 if (!*out_ret_type)
881 return INFIX_ERROR_INVALID_ARGUMENT;
882
883 // Convert linked list of args to a flat array.
884 infix_function_argument * args = (num_args > 0)
885 ? infix_arena_calloc(state->arena, num_args, sizeof(infix_function_argument), _Alignof(infix_function_argument))
886 : nullptr;
887 if (num_args > 0 && !args)
888 return INFIX_ERROR_ALLOCATION_FAILED;
889 arg_node * current = head;
890 for (size_t i = 0; i < num_args; i++) {
891 args[i] = current->arg;
892 current = current->next;
893 }
894 *out_args = args;
895 *out_num_args = num_args;
896 return INFIX_SUCCESS;
897}

917c23_nodiscard infix_status _infix_parse_type_internal(infix_type ** out_type,
918 infix_arena_t ** out_arena,
919 const char * signature) {

921 _infix_set_error(INFIX_CATEGORY_GENERAL, INFIX_CODE_UNKNOWN, 0);
922 return INFIX_ERROR_INVALID_ARGUMENT;
923 }
924 // The top-level public API is responsible for setting g_infix_last_signature_context.
925 *out_arena = infix_arena_create(4096);

928 return INFIX_ERROR_ALLOCATION_FAILED;
929 }

936 set_parser_error(&state, INFIX_CODE_UNEXPECTED_TOKEN);
937 type = nullptr;
938 }
939 }
940 if (!type) {
941 // If parsing failed at any point, clean up the temporary arena.
942 infix_arena_destroy(*out_arena);
943 *out_arena = nullptr;
944 *out_type = nullptr;
945 return INFIX_ERROR_INVALID_ARGUMENT;
946 }
947 *out_type = type;
948 return INFIX_SUCCESS;
949}

964c23_nodiscard infix_status infix_type_from_signature(infix_type ** out_type,
965 infix_arena_t ** out_arena,
966 const char * signature,
967 infix_registry_t * registry) {
968 _infix_clear_error();
969 g_infix_last_signature_context = signature; // Set context for rich error reporting.
970
971 // 1. "Parse" stage: Create a raw, unresolved type graph in a temporary arena.
972 infix_type * raw_type = nullptr;
973 infix_arena_t * parser_arena = nullptr;
974 infix_status status = _infix_parse_type_internal(&raw_type, &parser_arena, signature);
975 if (status != INFIX_SUCCESS)
976 return status;
977
978 // Create the final arena that will be returned to the caller.
979 *out_arena = infix_arena_create(4096);
980 if (!*out_arena) {
981 infix_arena_destroy(parser_arena);
982 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
983 return INFIX_ERROR_ALLOCATION_FAILED;
984 }
985
986 // 2. "Copy" stage: Deep copy the raw graph into the final arena.
987 infix_type * final_type = _copy_type_graph_to_arena(*out_arena, raw_type);
988 infix_arena_destroy(parser_arena); // The temporary graph is no longer needed.
989 if (!final_type) {
990 infix_arena_destroy(*out_arena);
991 *out_arena = nullptr;
992 return INFIX_ERROR_ALLOCATION_FAILED;
993 }
994
995 // 3. "Resolve" stage: Replace all named references (`@Name`) with concrete types.
996 status = _infix_resolve_type_graph_inplace(&final_type, registry);
997 if (status != INFIX_SUCCESS) {
998 infix_arena_destroy(*out_arena);
999 *out_arena = nullptr;
1000 *out_type = nullptr;
1001 }
1002 else {
1003 // 4. "Layout" stage: Calculate the final size, alignment, and member offsets.
1004 _infix_type_recalculate_layout(final_type);
1005 *out_type = final_type;
1006 }
1007 return status;
1008}

1027c23_nodiscard infix_status infix_signature_parse(const char * signature,
1028 infix_arena_t ** out_arena,
1029 infix_type ** out_ret_type,
1030 infix_function_argument ** out_args,
1031 size_t * out_num_args,
1032 size_t * out_num_fixed_args,
1033 infix_registry_t * registry) {

1036 return INFIX_ERROR_INVALID_ARGUMENT;
1037 g_infix_last_signature_context = signature;
1038
1039 // 1. "Parse" stage
1040 infix_type * raw_func_type = nullptr;
1041 infix_arena_t * parser_arena = nullptr;
1042 infix_status status = _infix_parse_type_internal(&raw_func_type, &parser_arena, signature);
1043 if (status != INFIX_SUCCESS)
1044 return status;
1045
1046 if (raw_func_type->category != INFIX_TYPE_REVERSE_TRAMPOLINE) {
1047 infix_arena_destroy(parser_arena);
1048 _infix_set_error(INFIX_CATEGORY_PARSER, INFIX_CODE_UNEXPECTED_TOKEN, 0);
1049 return INFIX_ERROR_INVALID_ARGUMENT;
1050 }
1051
1052 // Create final arena
1053 *out_arena = infix_arena_create(8192);
1054 if (!*out_arena) {
1055 infix_arena_destroy(parser_arena);
1056 _infix_set_error(INFIX_CATEGORY_ALLOCATION, INFIX_CODE_OUT_OF_MEMORY, 0);
1057 return INFIX_ERROR_ALLOCATION_FAILED;
1058 }
1059
1060 // 2. "Copy" stage
1061 infix_type * final_func_type = _copy_type_graph_to_arena(*out_arena, raw_func_type);
1062 infix_arena_destroy(parser_arena);
1063 if (!final_func_type) {
1064 infix_arena_destroy(*out_arena);
1065 *out_arena = nullptr;
1066 return INFIX_ERROR_ALLOCATION_FAILED;
1067 }
1068
1069 // 3. "Resolve" and 4. "Layout" stages
1070 status = _infix_resolve_type_graph_inplace(&final_func_type, registry);
1071 if (status != INFIX_SUCCESS) {
1072 infix_arena_destroy(*out_arena);
1073 *out_arena = nullptr;
1074 return INFIX_ERROR_INVALID_ARGUMENT;
1075 }
1076 _infix_type_recalculate_layout(final_func_type);
1077
1078 // Unpack the results for the caller from the final, processed function type object.
1079 *out_ret_type = final_func_type->meta.func_ptr_info.return_type;
1080 *out_args = final_func_type->meta.func_ptr_info.args;
1081 *out_num_args = final_func_type->meta.func_ptr_info.num_args;
1082 *out_num_fixed_args = final_func_type->meta.func_ptr_info.num_fixed_args;
1083
1084 return INFIX_SUCCESS;
1085}

1108static void _print(printer_state * state, const char * fmt, ...) {
1109 if (state->status != INFIX_SUCCESS)
1110 return;
1111 va_list args;
1112 va_start(args, fmt);
1113 int written = vsnprintf(state->p, state->remaining, fmt, args);
1114 va_end(args);
1115 if (written < 0 || (size_t)written >= state->remaining)
1116 // If snprintf failed or would have overflowed, mark an error.
1117 state->status = INFIX_ERROR_INVALID_ARGUMENT;
1118 else {
1119 state->p += written;
1120 state->remaining -= written;
1121 }
1122}

1130static void _infix_type_print_signature_recursive(printer_state * state, const infix_type * type) {
1131 if (state->status != INFIX_SUCCESS || !type) {
1132 if (state->status == INFIX_SUCCESS)
1133 state->status = INFIX_ERROR_INVALID_ARGUMENT;
1134 return;
1135 }
1136 switch (type->category) {
1137 case INFIX_TYPE_VOID:
1138 _print(state, "void");
1139 break;
1140 case INFIX_TYPE_NAMED_REFERENCE:
1141 _print(state, "@%s", type->meta.named_reference.name);
1142 break;
1143 case INFIX_TYPE_POINTER:
1144 _print(state, "*");
1145 // Special handling for `void*` or recursive pointers to avoid infinite recursion.
1146 if (type->meta.pointer_info.pointee_type == type || type->meta.pointer_info.pointee_type == nullptr ||
1147 type->meta.pointer_info.pointee_type->category == INFIX_TYPE_VOID)
1148 _print(state, "void");
1149 else
1150 _infix_type_print_signature_recursive(state, type->meta.pointer_info.pointee_type);
1151 break;
1152 case INFIX_TYPE_ARRAY:
1153 _print(state, "[%zu:", type->meta.array_info.num_elements);
1154 _infix_type_print_signature_recursive(state, type->meta.array_info.element_type);
1155 _print(state, "]");
1156 break;
1157 case INFIX_TYPE_STRUCT:
1158 // If a struct was created from a registry, it will have a name. Print the name
1159 // instead of the full body for a canonical representation.
1160 if (type->meta.aggregate_info.name)
1161 _print(state, "@%s", type->meta.aggregate_info.name);
1162 else {
1163 _print(state, "{");
1164 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
1165 if (i > 0)
1166 _print(state, ",");
1167 const infix_struct_member * member = &type->meta.aggregate_info.members[i];
1168 if (member->name)
1169 _print(state, "%s:", member->name);
1170 _infix_type_print_signature_recursive(state, member->type);
1171 }
1172 _print(state, "}");
1173 }
1174 break;
1175 case INFIX_TYPE_UNION:
1176 if (type->meta.aggregate_info.name)
1177 _print(state, "@%s", type->meta.aggregate_info.name);
1178 else {
1179 _print(state, "<");
1180 for (size_t i = 0; i < type->meta.aggregate_info.num_members; ++i) {
1181 if (i > 0)
1182 _print(state, ",");
1183 const infix_struct_member * member = &type->meta.aggregate_info.members[i];
1184 if (member->name)
1185 _print(state, "%s:", member->name);

1187 }
1188 _print(state, ">");
1189 }
1190 break;
1191 case INFIX_TYPE_REVERSE_TRAMPOLINE:
1192 _print(state, "(");
1193 for (size_t i = 0; i < type->meta.func_ptr_info.num_fixed_args; ++i) {
1194 if (i > 0)

1202 _print(state, ";");

1207 if (arg->name)
1208 _print(state, "%s:", arg->name);
1209 _infix_type_print_signature_recursive(state, arg->type);
1210 }
1211 }
1212 _print(state, ")->");
1213 _infix_type_print_signature_recursive(state, type->meta.func_ptr_info.return_type);
1214 break;
1215 case INFIX_TYPE_ENUM:

1218 break;
1219 case INFIX_TYPE_COMPLEX:
