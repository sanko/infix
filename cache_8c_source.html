41static uint64_t _cache_hash(const char * sig, void * target_fn, bool is_safe) {
42 uint64_t h = 5381;
43 int c;
44 while ((c = *sig++))
45 h = ((h << 5) + h) + c;
46 h ^= (uint64_t)(uintptr_t)target_fn;
47 if (is_safe)
48 h ^= 0x123456789ABCDEF0ULL;
49 return h;

130static bool _cache_remove_no_lock(infix_forward_t * trampoline) {
131 if (!trampoline->signature)
132 return false;
133 uint64_t h = _cache_hash(trampoline->signature, trampoline->target_fn, trampoline->is_safe);
134 size_t index = h % CACHE_BUCKETS;

137 while (*p) {
138 if ((*p)->trampoline == trampoline) {
139 _cache_entry_t * to_free = *p;
140 *p = to_free->next;
141 infix_free(to_free);

143 return true;
144 }
